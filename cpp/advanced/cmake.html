CMake

CMake is the de facto standard build system for C++ projects. It generates native build files (Makefiles, Ninja, Visual Studio projects) from a platform-independent CMakeLists.txt configuration. Modern CMake (3.15+) promotes target-based configuration over directory-level variables.

CMake Basics

# Minimum CMake version (always set this first)
cmake_minimum_required(VERSION 3.20)

# Project name and language
project(MyProject VERSION 1.0.0 LANGUAGES CXX)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)  # Use -std=c++17, not -std=gnu++17

# Export compile_commands.json (needed by clang-tidy, clangd)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

Multi-Target CMakeLists.txt

cmake_minimum_required(VERSION 3.20)
project(RanStack VERSION 2.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXPORT_COMPILE_COMMANDS ON)

# --- Compile options (modern CMake: use target_*, not add_definitions) ---
add_library(ran_compile_options INTERFACE)
target_compile_options(ran_compile_options INTERFACE
    -Wall -Wextra -Wpedantic
    $<$<CMAKE_BUILD_TYPE:Debug>:-g -O0>
    $<$<CMAKE_BUILD_TYPE:Release>:-O2 -DNDEBUG>
)

# --- Library target ---
add_library(ran_protocol
    src/packet.cpp
    src/rlc_layer.cpp
    src/pdcp_layer.cpp
)
target_include_directories(ran_protocol PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
    $<INSTALL_INTERFACE:include>
)
target_link_libraries(ran_protocol
    PUBLIC Threads::Threads
    PRIVATE ran_compile_options
)

# --- Executable target ---
add_executable(ran_simulator
    src/main.cpp
    src/simulator.cpp
)
target_link_libraries(ran_simulator
    PRIVATE ran_protocol ran_compile_options
)

# --- Test target ---
add_executable(ran_tests
    tests/main.cpp
    tests/packet_test.cpp
    tests/rlc_test.cpp
)
target_link_libraries(ran_tests
    PRIVATE ran_protocol GTest::gtest_main ran_compile_options
)
enable_testing()
add_test(NAME protocol_tests COMMAND ran_tests)

# --- Static analysis integration ---
find_program(CPPCHECK cppcheck)
if(CPPCHECK)
    add_custom_target(cppcheck
        COMMAND ${CPPCHECK} --enable=all --std=c++17 --suppress=missingInclude
                ${CMAKE_CURRENT_SOURCE_DIR}/src
        WORKING_DIRECTORY ${CMAKE_BINARY_DIR}
    )
endif()

find_program(CLANG_TIDY clang-tidy)
if(CLANG_TIDY)
    set_target_properties(ran_protocol ran_simulator PROPERTIES
        CXX_CLANG_TIDY "${CLANG_TIDY};--config-file=${CMAKE_SOURCE_DIR}/.clang-tidy"
    )
endif()

Key CMake Commands

CommandPurposeExample
add_executableDefine an executable targetadd_executable(app main.cpp)
add_libraryDefine a library (STATIC, SHARED, OBJECT, INTERFACE)add_library(lib STATIC lib.cpp)
target_link_librariesLink libraries and transitively propagate their settingstarget_link_libraries(app PRIVATE lib)
target_include_directoriesAdd include search paths for a targettarget_include_directories(lib PUBLIC include/)
target_compile_optionsAdd compiler flags for a targettarget_compile_options(app PRIVATE -Wall)
target_compile_definitionsAdd preprocessor definitionstarget_compile_definitions(app PRIVATE DEBUG)
find_packageLocate and import external librariesfind_package(Threads REQUIRED)
enable_testing / add_testRegister tests with CTestadd_test(NAME mytest COMMAND app)

Modern CMake: Link Visibility

# PUBLIC: headers and link dependencies propagate to consumers
target_include_directories(mylib PUBLIC include/)     # consumers see headers
target_link_libraries(mylib PUBLIC Threads::Threads)  # consumers also link Threads

# PRIVATE: only for this target, not propagated to consumers
target_compile_options(mylib PRIVATE -Wno-old-style-cast)

# INTERFACE: only propagated to consumers, not used by this target itself
add_library(mylib_interface INTERFACE)
target_link_libraries(mylib_interface INTERFACE
    Threads::Threads
    nlohmann_json::nlohmann_json
)

Legacy vs Modern: Avoid include_directories(), link_directories(), add_definitions(), and link_libraries() (no target prefix). These set directory-level properties that are not target-scoped and cause hard-to-debug dependency issues.

Building with CMake

# Configure (out-of-source build)
cmake -B build -DCMAKE_BUILD_TYPE=Release

# Build
cmake --build build --parallel 8

# Run tests
ctest --test-dir build --output-on-failure

# Debug build
cmake -B build-debug -DCMAKE_BUILD_TYPE=Debug -DCMAKE_CXX_FLAGS="-fsanitize=address"
cmake --build build-debug

Best practice: Always use out-of-source builds (-B build). Keep compile_commands.json enabled for cppcheck and clang-tidy integration. Use target-based commands exclusively. Use Make as the generator backend or Ninja for faster builds.
