Observer Pattern

The Observer pattern defines a one-to-many dependency between objects: when one object (the subject) changes state, all its dependents (observers) are notified automatically. This is the foundation of event systems, MVC architecture, and reactive programming. In C++, observers can be implemented via virtual functions, std::function callbacks, or the std::variant+visitor approach.

Prerequisites: Singleton Pattern, Factory Pattern, std::function & std::bind, Lambda Expressions.

Classic Subject/Observer with Virtual Functions

#include <iostream>
#include <string>
#include <vector>
#include <memory>

// Observer interface
class Observer {
public:
    virtual ~Observer() = default;
    virtual void on_notify(const std::string& event, int data) = 0;
};

// Subject (observable)
class Subject {
    std::vector<Observer*> observers_;
public:
    void attach(Observer* obs) { observers_.push_back(obs); }
    void detach(Observer* obs) {
        observers_.erase(
            std::remove(observers_.begin(), observers_.end(), obs),
            observers_.end()
        );
    }

    void notify(const std::string& event, int data) {
        for (auto* obs : observers_) {
            obs->on_notify(event, data);
        }
    }
};

// Concrete observer
class UeMonitor : public Observer {
    std::string name_;
public:
    explicit UeMonitor(std::string name) : name_(std::move(name)) {}

    void on_notify(const std::string& event, int data) override {
        std::cout << "[" << name_ << "] Event: " << event
                  << ", UE-ID: " << data << "\n";
    }
};

class MetricsCollector : public Observer {
    int event_count_ = 0;
public:
    void on_notify(const std::string& event, int /*data*/) override {
        ++event_count_;
        std::cout << "[Metrics] Total events: " << event_count_ << "\n";
    }
};

int main() {
    Subject event_bus;

    UeMonitor monitor_a("CU-1");
    UeMonitor monitor_b("DU-1");
    MetricsCollector metrics;

    event_bus.attach(&monitor_a);
    event_bus.attach(&monitor_b);
    event_bus.attach(&metrics);

    event_bus.notify("HANDOVER_START", 42);
    event_bus.notify("HANDOVER_COMPLETE", 42);
    event_bus.notify("UE_RELEASE", 99);
}

Callback-Based Event System (std::function)

Using std::function removes the need for an observer base class and allows lambdas as callbacks:

#include <iostream>
#include <string>
#include <functional>
#include <unordered_map>
#include <vector>
#include <mutex>

class EventBus {
    using Callback = std::function<void(int)>;
    std::unordered_map<std::string, std::vector<Callback>> subscribers_;
    mutable std::mutex mutex_;

public:
    // Subscribe to an event type
    void on(const std::string& event, Callback cb) {
        std::lock_guard lock(mutex_);
        subscribers_[event].push_back(std::move(cb));
    }

    // Emit an event
    void emit(const std::string& event, int data) {
        std::lock_guard lock(mutex_);
        auto it = subscribers_.find(event);
        if (it != subscribers_.end()) {
            for (const auto& cb : it->second) {
                cb(data);
            }
        }
    }
};

int main() {
    EventBus bus;

    // Subscribe with lambdas
    bus.on("UE_ATTACH", [](int ue_id) {
        std::cout << "UE attached: " << ue_id << "\n";
    });

    bus.on("UE_ATTACH", [](int ue_id) {
        std::cout << "Logging: UE " << ue_id << " connected\n";
    });

    int handover_count = 0;
    bus.on("HANDOVER", [&handover_count](int ue_id) {
        ++handover_count;
        std::cout << "Handover #" << handover_count
                  << " for UE " << ue_id << "\n";
    });

    bus.emit("UE_ATTACH", 1001);
    bus.emit("UE_ATTACH", 1002);
    bus.emit("HANDOVER", 1001);
}

Type-Safe Events with std::variant

#include <iostream>
#include <string>
#include <variant>
#include <vector>
#include <functional>

// Define event types
struct UeAttachEvent { int ue_id; std::string cell; };
struct UeReleaseEvent { int ue_id; int cause; };
struct HandoverEvent { int ue_id; std::string source; std::string target; };

using Event = std::variant<UeAttachEvent, UeReleaseEvent, HandoverEvent>;

// Type-safe event bus
class TypedEventBus {
    std::vector<std::function<void(const Event&)>> handlers_;
public:
    void subscribe(std::function<void(const Event&)> handler) {
        handlers_.push_back(std::move(handler));
    }

    void publish(const Event& event) {
        for (const auto& h : handlers_) {
            h(event);
        }
    }
};

int main() {
    TypedEventBus bus;

    bus.subscribe([](const Event& e) {
        std::visit([](auto&& arg) {
            using T = std::decay_t<decltype(arg)>;
            if constexpr (std::is_same_v<T, UeAttachEvent>) {
                std::cout << "Attach: UE " << arg.ue_id
                          << " at " << arg.cell << "\n";
            } else if constexpr (std::is_same_v<T, UeReleaseEvent>) {
                std::cout << "Release: UE " << arg.ue_id
                          << " cause=" << arg.cause << "\n";
            } else if constexpr (std::is_same_v<T, HandoverEvent>) {
                std::cout << "Handover: UE " << arg.ue_id
                          << " " << arg.source << " -> " << arg.target << "\n";
            }
        }, e);
    });

    bus.publish(UeAttachEvent{1001, "Cell-A"});
    bus.publish(HandoverEvent{1001, "Cell-A", "Cell-B"});
    bus.publish(UeReleaseEvent{1001, 0});
}

Implementation Comparison

ApproachProsCons
Virtual functionsType-safe, polymorphicRequires inheritance, rigid interface
std::functionFlexible, lambdas, no inheritanceType erasure overhead, no type safety
std::variantType-safe, extensible, no vtableAll event types in one variant, verbose
Signals/Slots (Qt-style)Very flexible, thread-safeLibrary dependency

Best practice: Use std::function callbacks for simple event systems. Use std::variant when you need compile-time type safety for events. Always make the event bus thread-safe if observers are subscribed/unsubscribed from different threads. See Observer Pattern Practice for exercises.
