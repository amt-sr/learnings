Producer-Consumer Pattern

The Producer-Consumer pattern decouples data generation from data processing using a shared buffer. Producer threads push data into the buffer; consumer threads pop and process it. This pattern is fundamental in concurrent systems: network servers accept connections (producer) and worker threads handle requests (consumer), real-time systems collect sensor data (producer) and process it asynchronously (consumer).

Prerequisites: Mutex & Locks, Condition Variable, Thread-Safe Data Structures, Lock-Free Programming.

Bounded Buffer with mutex + condition_variable

#include <iostream>
#include <queue>
#include <thread>
#include <mutex>
#include <condition_variable>
#include <string>
#include <chrono>

// Thread-safe bounded buffer (fixed capacity)
template<typename T>
class BoundedBuffer {
    std::queue<T> queue_;
    std::mutex mutex_;
    std::condition_variable not_full_;   // signals when space available
    std::condition_variable not_empty_;  // signals when data available
    std::size_t capacity_;

public:
    explicit BoundedBuffer(std::size_t capacity) : capacity_(capacity) {}

    void push(T item) {
        std::unique_lock lock(mutex_);
        // Wait until there is space in the buffer
        not_full_.wait(lock, [this] { return queue_.size() < capacity_; });
        queue_.push(std::move(item));
        lock.unlock();
        // Notify one consumer that data is available
        not_empty_.notify_one();
    }

    T pop() {
        std::unique_lock lock(mutex_);
        // Wait until buffer is not empty
        not_empty_.wait(lock, [this] { return !queue_.empty(); });
        T item = std::move(queue_.front());
        queue_.pop();
        lock.unlock();
        // Notify one producer that space is available
        not_full_.notify_one();
        return item;
    }

    // Non-blocking try_pop
    bool try_pop(T& item) {
        std::lock_guard lock(mutex_);
        if (queue_.empty()) return false;
        item = std::move(queue_.front());
        queue_.pop();
        not_full_.notify_one();
        return true;
    }

    std::size_t size() const {
        std::lock_guard lock(mutex_);
        return queue_.size();
    }
};

Thread-Safe Queue (General Purpose)

#include <queue>
#include <mutex>
#include <condition_variable>
#include <optional>

// Unbounded thread-safe queue with shutdown support
template<typename T>
class ThreadSafeQueue {
    std::queue<T> queue_;
    mutable std::mutex mutex_;
    std::condition_variable cv_;
    bool shutdown_ = false;

public:
    void push(T item) {
        {
            std::lock_guard lock(mutex_);
            if (shutdown_) return;
            queue_.push(std::move(item));
        }
        cv_.notify_one();
    }

    // Blocking pop — returns nullopt on shutdown
    std::optional<T> pop() {
        std::unique_lock lock(mutex_);
        cv_.wait(lock, [this] { return shutdown_ || !queue_.empty(); });

        if (shutdown_ && queue_.empty()) return std::nullopt;

        T item = std::move(queue_.front());
        queue_.pop();
        return item;
    }

    // Shutdown: wake all waiting consumers
    void shutdown() {
        {
            std::lock_guard lock(mutex_);
            shutdown_ = true;
        }
        cv_.notify_all();
    }

    std::size_t size() const {
        std::lock_guard lock(mutex_);
        return queue_.size();
    }
};

Multiple Producers, Multiple Consumers

#include <iostream>
#include <thread>
#include <vector>
#include <atomic>
#include <chrono>

BoundedBuffer<int> buffer(10);  // capacity 10
std::atomic<bool> done{false};
std::atomic<int> total_produced{0};
std::atomic<int> total_consumed{0};

void producer(int id) {
    for (int i = 0; i < 20; ++i) {
        buffer.push(i + id * 100);
        total_produced.fetch_add(1);
        std::cout << "Producer " << id << " pushed " << (i + id * 100) << "\n";
        std::this_thread::sleep_for(std::chrono::milliseconds(10));
    }
}

void consumer(int id) {
    while (true) {
        int item = buffer.pop();
        total_consumed.fetch_add(1);
        std::cout << "Consumer " << id << " popped " << item << "\n";
        std::this_thread::sleep_for(std::chrono::milliseconds(25));

        if (done.load() && total_consumed.load() >= total_produced.load()) {
            break;
        }
    }
}

int main() {
    const int num_producers = 3;
    const int num_consumers = 2;

    std::vector<std::thread> producers;
    std::vector<std::thread> consumers;

    for (int i = 0; i < num_producers; ++i)
        producers.emplace_back(producer, i);
    for (int i = 0; i < num_consumers; ++i)
        consumers.emplace_back(consumer, i);

    for (auto& t : producers) t.join();

    done.store(true);
    for (auto& t : consumers) t.join();

    std::cout << "Total produced: " << total_produced << "\n";
    std::cout << "Total consumed: " << total_consumed << "\n";
}

Connection to Thread Pools

A thread pool is a fixed set of consumer threads with an unbounded task queue. Producers submit tasks; worker threads execute them. The producer-consumer pattern is the backbone of every thread pool implementation:

ComponentProducer-Consumer Mapping
Task queueShared buffer (usually unbounded)
Worker threadsConsumers — pop tasks and execute
Task submissionProducer — push tasks into queue
ShutdownStop accepting tasks, drain queue, join workers

Best practice: Use condition variables with predicates (lambda in wait()) to avoid spurious wakeups. Always pair notify_one/notify_all with actual state changes while holding the lock. For high-throughput scenarios, consider Lock-Free Programming approaches to avoid mutex contention.

Common pitfall — missed wakeup: If the producer notifies before the consumer waits, the notification is lost. The predicate-based wait() (e.g., wait(lock, [this]{ return !queue_.empty(); })) prevents this by checking the condition first.

Related topics: Mutex & Locks (mutual exclusion), Condition Variable (efficient waiting/notification), Thread-Safe Data Structures (concurrent containers), Lock-Free Programming (mutex-free alternatives), Thread-Safe Queue Practice.
