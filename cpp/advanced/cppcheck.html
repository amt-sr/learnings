cppcheck

cppcheck is a free, open-source static analysis tool for C and C++ code. Unlike compiler-based tools, it does not require compilation -- it performs its own parsing and analysis. This makes it fast and easy to integrate into any build system or CI pipeline.

Installing cppcheck

# macOS (Homebrew)
brew install cppcheck

# Ubuntu/Debian
sudo apt install cppcheck

# Windows (vcpkg or download from github.com/danmar/cppcheck)
vcpkg install cppcheck

# Verify installation
cppcheck --version
# cppcheck 2.14.0

Running Analysis

# Basic scan on source directory
cppcheck src/

# Enable all checks with detailed output
cppcheck --enable=all --std=c++17 --suppress=missingInclude src/

# Check specific files
cppcheck --enable=all src/main.cpp src/network.cpp

# Generate XML report for CI integration
cppcheck --enable=all --xml 2> report.xml src/

# Parallel analysis (use all cores)
cppcheck -j 8 --enable=all src/

Suppression Files

Suppress false positives with inline comments or a suppression file:

// Inline suppression (single line)
int* p = new int[10]; // cppcheck-suppress memoryLeak

// File suppression (suppress for an entire file)
// cppcheck-suppress-file unusedFunction

// suppression.txt file format:
# Suppress by file and line
memleak:src/legacy.cpp:42

# Suppress by file and check name
unusedVariable:src/helpers.cpp

# Suppress by check name globally
missingInclude

# Run with suppression file
cppcheck --enable=all --suppressions-list=suppressions.txt src/

CMake Integration

# CMakeLists.txt -- add cppcheck as a custom target
find_program(CPPCHECK cppcheck)

if(CPPCHECK)
    add_custom_target(
        cppcheck
        COMMAND ${CPPCHECK}
            --enable=all
            --suppress=missingInclude
            --std=c++17
            --language=c++
            -j ${PARALLEL_JOBS}
            ${PROJECT_SOURCE_DIR}/src
        WORKING_DIRECTORY ${PROJECT_BINARY_DIR}
        COMMENT "Running cppcheck static analysis"
    )
endif()

CI Pipeline Integration

# GitHub Actions step
- name: Run cppcheck
  run: |
    cppcheck --enable=all \
      --suppress=missingInclude \
      --std=c++17 \
      --error-exitcode=1 \
      --xml 2> cppcheck.xml \
      src/

- name: Upload cppcheck report
  if: always()
  uses: actions/upload-artifact@v4
  with:
    name: cppcheck-report
    path: cppcheck.xml

Tip: Use --error-exitcode=1 in CI so the build fails when errors are found. Use --force to check all configurations (all #ifdef branches) if you have platform-specific code.

Best practice: Combine cppcheck with clang-tidy and Clang Static Analyzer for comprehensive coverage. cppcheck excels at catching memory leaks that other tools miss. Add sanitizers for runtime verification.
