Placement New & Memory Pools

Placement new allows you to construct an object at a specific memory address. Combined with memory pools (pre-allocated blocks of memory), this enables deterministic allocation, zero fragmentation, and precise control over object lifetimes. This is the low-level foundation beneath the Custom Allocators topic.

Placement New Syntax

#include <new>

// Regular new: allocates memory AND constructs the object
MyClass* obj1 = new MyClass(args);

// Placement new: constructs the object at a specific address (NO allocation)
alignas(MyClass) char buffer[sizeof(MyClass)];
MyClass* obj2 = new (buffer) MyClass(args);

// IMPORTANT: must manually call destructor (do NOT delete placement-new'd objects)
obj2->~MyClass();

Critical rule: Never call delete on an object created with placement new. The memory was not allocated by new, so delete will try to free invalid memory. Always call the destructor explicitly instead.

Basic Placement New Usage

#include <new>
#include <iostream>
#include <cstddef>
#include <cstring>

struct RrcConnectionRequest {
    std::uint64_t ue_id = 0;
    std::uint8_t establishment_cause = 0;
    std::uint8_t spare[2]{};

    RrcConnectionRequest(std::uint64_t id, std::uint8_t cause)
        : ue_id(id), establishment_cause(cause) {}

    void print() const {
        std::cout << "UE ID: " << ue_id
                  << ", Cause: " << static_cast<int>(establishment_cause) << "\n";
    }
};

int main() {
    // Pre-allocate raw memory buffer
    alignas(RrcConnectionRequest) char buffer[sizeof(RrcConnectionRequest)];

    // Construct object in the buffer using placement new
    auto* req = new (buffer) RrcConnectionRequest(12345, 1);
    req->print();  // UE ID: 12345, Cause: 1

    // Manually call destructor
    req->~RrcConnectionRequest();

    // Buffer can now be reused for a different object
    std::memset(buffer, 0, sizeof(buffer));
    auto* req2 = new (buffer) RrcConnectionRequest(67890, 2);
    req2->print();  // UE ID: 67890, Cause: 2
    req2->~RrcConnectionRequest();

    return 0;
}

Memory Pool Implementation

#include <cstddef>
#include <new>
#include <stack>
#include <vector>
#include <iostream>
#include <stdexcept>

template<typename T, std::size_t MaxObjects = 1024>
class MemoryPool {
    alignas(alignof(T)) char buffer_[MaxObjects * sizeof(T)];
    std::stack<T*, std::vector<T*>> free_stack_;
    std::size_t allocated_count_ = 0;

    void init() {
        // Push all slots onto the free stack
        for (std::size_t i = 0; i < MaxObjects; ++i) {
            free_stack_.push(reinterpret_cast<T*>(buffer_ + i * sizeof(T)));
        }
    }

public:
    MemoryPool() { init(); }

    // Non-copyable
    MemoryPool(const MemoryPool&) = delete;
    MemoryPool& operator=(const MemoryPool&) = delete;

    template<typename... Args>
    T* construct(Args&&... args) {
        if (free_stack_.empty()) {
            throw std::runtime_error("MemoryPool exhausted");
        }
        T* ptr = free_stack_.top();
        free_stack_.pop();
        ++allocated_count_;
        return new (ptr) T(std::forward<Args>(args)...);
    }

    void destroy(T* ptr) {
        if (!ptr) return;
        ptr->~T();
        free_stack_.push(ptr);
        --allocated_count_;
    }

    [[nodiscard]] std::size_t available() const noexcept {
        return free_stack_.size();
    }

    [[nodiscard]] std::size_t allocated() const noexcept {
        return allocated_count_;
    }
};

// Usage
struct UeContext {
    std::uint32_t ue_id = 0;
    std::uint32_t cell_id = 0;
    uint8_t state = 0;

    UeContext(std::uint32_t uid, std::uint32_t cid)
        : ue_id(uid), cell_id(cid) {}
};

int main() {
    MemoryPool<UeContext, 100> pool;
    std::cout << "Available: " << pool.available() << "\n";  // 100

    auto* ctx1 = pool.construct(1, 100);
    auto* ctx2 = pool.construct(2, 100);
    std::cout << "Available: " << pool.available() << "\n";  // 98

    pool.destroy(ctx1);
    std::cout << "Available after destroy: " << pool.available() << "\n";  // 99

    // Object reuse — no heap allocation
    auto* ctx3 = pool.construct(3, 101);  // reuses ctx1's slot
}

Object Reuse Patterns

In telecom systems, objects representing protocol contexts (UE sessions, bearer contexts, message buffers) are frequently created and destroyed. Memory pools allow object slots to be reused without heap allocation:

#include <new>
#include <optional>
#include <functional>

// A generic object pool with RAII guard
template<typename T>
class ObjectPool {
    struct Slot {
        alignas(alignof(T)) char storage[sizeof(T)];
        bool occupied = false;
    };

    std::vector<Slot> slots_;

public:
    explicit ObjectPool(std::size_t capacity) : slots_(capacity) {}

    class Handle {
        T* ptr_ = nullptr;
        ObjectPool* pool_ = nullptr;
        std::size_t index_ = 0;

    public:
        Handle(T* ptr, ObjectPool* pool, std::size_t idx)
            : ptr_(ptr), pool_(pool), index_(idx) {}

        ~Handle() {
            if (ptr_ && pool_) {
                ptr_->~T();
                pool_->slots_[index_].occupied = false;
            }
        }

        // Non-copyable, movable
        Handle(const Handle&) = delete;
        Handle& operator=(const Handle&) = delete;
        Handle(Handle&& other) noexcept
            : ptr_(other.ptr_), pool_(other.pool_), index_(other.index_) {
            other.ptr_ = nullptr;
            other.pool_ = nullptr;
        }

        T& operator*() { return *ptr_; }
        T* operator->() { return ptr_; }
        explicit operator bool() const { return ptr_ != nullptr; }
    };

    template<typename... Args>
    std::optional<Handle> acquire(Args&&... args) {
        for (std::size_t i = 0; i < slots_.size(); ++i) {
            if (!slots_[i].occupied) {
                T* ptr = new (slots_[i].storage) T(std::forward<Args>(args)...);
                slots_[i].occupied = true;
                return Handle(ptr, this, i);
            }
        }
        return std::nullopt;
    }

    [[nodiscard]] std::size_t capacity() const { return slots_.size(); }
    [[nodiscard]] std::size_t in_use() const {
        return std::count_if(slots_.begin(), slots_.end(),
                              [](const Slot& s) { return s.occupied; });
    }
};

// Usage: RAII handles automatically return objects to the pool
void handle_ue_connection() {
    ObjectPool<UeContext> pool(1000);

    {
        auto handle = pool.acquire(42, 100);
        if (handle) {
            handle->state = 1;
            // handle's destructor returns the slot to the pool
        }
    }  // slot returned here

    auto handle2 = pool.acquire(43, 101);
    // Reuses the slot that was just returned
}

Telecom Relevance

5G base stations manage thousands of simultaneous UE contexts. Each context involves protocol state machines, timers, and buffer allocations. Creating these on the heap for every connection setup and tearing them down on release causes fragmentation and non-deterministic latency. Memory pools with placement new provide:

  • Deterministic allocation: Fixed-time allocation and deallocation, no OS involvement
  • Zero fragmentation: Fixed-size slots from pre-allocated memory
  • Cache locality: All objects in a pool are contiguous in memory (see Cache-Oriented Programming)
  • Object reuse: Destroyed slots are immediately available for new connections

Common mistake: Forgetting to call the destructor before reuse. If a slot's previous occupant had a non-trivial destructor (held a mutex, file handle, or allocated sub-resources), skipping destruction causes resource leaks. Always pair placement new with an explicit destructor call.
