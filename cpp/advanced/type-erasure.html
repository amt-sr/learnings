Type Erasure

Type erasure decouples the interface from the concrete type, allowing objects of different types to be stored in the same container and manipulated through a common interface — all without requiring a common base class or virtual functions. It provides value semantics (copyable, movable) unlike raw pointers to base classes.

Contrast with CRTP: CRTP requires the concrete type to be known at compile time. Type erasure defers the concrete type selection to runtime while maintaining value semantics.

std::any

std::any (C++17) is the simplest form of type erasure — it can hold any copyable type and you retrieve values with std::any_cast:

#include <any>
#include <iostream>
#include <string>
#include <vector>

int main() {
    std::any value = 42;                // holds an int
    std::cout << std::any_cast<int>(value) << "\n";  // 42

    value = std::string("hello");       // now holds a string
    std::cout << std::any_cast<std::string>(value) << "\n";  // hello

    // Wrong cast: throws std::bad_any_cast
    try {
        auto x = std::any_cast<double>(value);
    } catch (const std::bad_any_cast& e) {
        std::cout << "Bad cast: " << e.what() << "\n";
    }

    // Check type before casting
    if (value.type() == typeid(std::string)) {
        std::cout << "It's a string!\n";
    }

    // Container of mixed types
    std::vector<std::any> mixed = {1, "text"s, 3.14, true};
}

std::any limitations: No interface — you can only store and retrieve. Type safety is checked at runtime via any_cast. Heap allocation for large objects. For a typed interface, build a custom type-erased wrapper.

Custom Type-Erased Wrapper

#include <memory>
#include <utility>
#include <string>
#include <iostream>
#include <functional>

// Type-erased callable wrapper (similar to std::function)
class AnyCallback {
    struct ICallable {
        virtual ~ICallable() = default;
        virtual void invoke() = 0;
        virtual std::unique_ptr<ICallable> clone() const = 0;
    };

    template<typename F>
    struct CallableImpl : ICallable {
        F func;
        explicit CallableImpl(F f) : func(std::move(f)) {}

        void invoke() override { func(); }

        std::unique_ptr<ICallable> clone() const override {
            return std::make_unique<CallableImpl<F>>(func);
        }
    };

    std::unique_ptr<ICallable> impl_;

public:
    AnyCallback() = default;

    template<typename F>
    explicit AnyCallback(F f)
        : impl_(std::make_unique<CallableImpl<std::decay_t<F>>>(std::move(f))) {}

    // Copyable
    AnyCallback(const AnyCallback& other)
        : impl_(other.impl_ ? other.impl_->clone() : nullptr) {}

    AnyCallback& operator=(const AnyCallback& other) {
        if (this != &other) {
            impl_ = other.impl_ ? other.impl_->clone() : nullptr;
        }
        return *this;
    }

    // Movable
    AnyCallback(AnyCallback&&) noexcept = default;
    AnyCallback& operator=(AnyCallback&&) noexcept = default;

    void operator()() {
        if (impl_) impl_->invoke();
    }

    explicit operator bool() const { return impl_ != nullptr; }
};

// Usage: heterogeneous container of callbacks
int main() {
    std::vector<AnyCallback> callbacks;

    callbacks.emplace_back([]() { std::cout << "Lambda callback\n"; });
    callbacks.emplace_back([]() { std::cout << "Another callback\n"; });

    // Store a function pointer
    void (*func_ptr)() = []() { std::cout << "Function pointer\n"; };
    callbacks.emplace_back(func_ptr);

    for (auto& cb : callbacks) {
        cb();
    }
}

Type-Erased Message Handler

#include <memory>
#include <string>
#include <vector>
#include <iostream>

// A type-erased message handler: any handler can be stored
// and invoked with a string message
class MessageHandler {
    struct IHandler {
        virtual ~IHandler() = default;
        virtual void handle(const std::string& msg) = 0;
        virtual std::string name() const = 0;
        virtual std::unique_ptr<IHandler> clone() const = 0;
    };

    template<typename H>
    struct HandlerImpl : IHandler {
        H handler;
        explicit HandlerImpl(H h) : handler(std::move(h)) {}

        void handle(const std::string& msg) override { handler.handle(msg); }
        std::string name() const override { return handler.name(); }
        std::unique_ptr<IHandler> clone() const override {
            return std::make_unique<HandlerImpl<H>>(handler);
        }
    };

    std::unique_ptr<IHandler> impl_;

public:
    template<typename H>
    explicit MessageHandler(H h)
        : impl_(std::make_unique<HandlerImpl<std::decay_t<H>>>(std::move(h))) {}

    MessageHandler(const MessageHandler& other)
        : impl_(other.impl_ ? other.impl_->clone() : nullptr) {}
    MessageHandler& operator=(const MessageHandler&) = delete;
    MessageHandler(MessageHandler&&) noexcept = default;
    MessageHandler& operator=(MessageHandler&&) = default;

    void handle(const std::string& msg) { impl_->handle(msg); }
    std::string name() const { return impl_->name(); }
};

// Different handler types — no common base class required
class RrcHandler {
public:
    void handle(const std::string& msg) {
        std::cout << "[RRC] " << msg << "\n";
    }
    std::string name() const { return "RRC"; }
};

class NgapHandler {
public:
    void handle(const std::string& msg) {
        std::cout << "[NGAP] " << msg << "\n";
    }
    std::string name() const { return "NGAP"; }
};

int main() {
    // Heterogeneous container — no common base class needed
    std::vector<MessageHandler> handlers;
    handlers.emplace_back(RrcHandler{});
    handlers.emplace_back(NgapHandler{});
    handlers.emplace_back([](const std::string& msg) {
        std::cout << "[LAMBDA] " << msg << "\n";
    });

    for (auto& h : handlers) {
        std::cout << "Dispatching to " << h.name() << ": ";
        h.handle("UE Setup Request");
    }
}

Telecom Relevance

Type erasure enables a protocol stack to register handlers for different message types without requiring a common base class. A timer manager can accept any callable, a metrics collector can store any type of metric source, and a message dispatcher can route to handlers of unrelated types — all with type safety and value semantics.

When to prefer type erasure over virtual dispatch: When you need value semantics (copyable objects), when types don't share a common base, or when you want to hide implementation details behind a clean interface. When performance is critical and types share a common base, virtual dispatch or CRTP may be more efficient.
