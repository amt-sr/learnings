Strategy Pattern

The Strategy pattern defines a family of algorithms, encapsulates each one in a separate class, and makes them interchangeable at runtime. It lets the algorithm vary independently from the objects that use it. In C++, strategies can be implemented via virtual functions (classic), std::function (runtime flexibility), or templates (compile-time selection, zero overhead).

Prerequisites: Factory Pattern, Observer Pattern, Virtual Functions & vtable, Lambda Expressions.

Classic Strategy (Virtual Functions)

#include <iostream>
#include <algorithm>
#include <vector>
#include <random>
#include <memory>

// Strategy interface
class SortStrategy {
public:
    virtual ~SortStrategy() = default;
    virtual void sort(std::vector<int>& data) = 0;
    virtual std::string name() const = 0;
};

// Concrete strategies
class QuickSort : public SortStrategy {
public:
    void sort(std::vector<int>& data) override {
        std::sort(data.begin(), data.end());
    }
    std::string name() const override { return "QuickSort (std::sort introsort)"; }
};

class StableSort : public SortStrategy {
public:
    void sort(std::vector<int>& data) override {
        std::stable_sort(data.begin(), data.end());
    }
    std::string name() const override { return "StableSort (merge sort)"; }
};

class PartialSort : public SortStrategy {
    std::size_t k_;
public:
    explicit PartialSort(std::size_t k = 10) : k_(k) {}

    void sort(std::vector<int>& data) override {
        if (data.size() > k_) {
            std::partial_sort(data.begin(), data.begin() + k_, data.end());
        } else {
            std::sort(data.begin(), data.end());
        }
    }
    std::string name() const override { return "PartialSort (top " + std::to_string(k_) + ")"; }
};

// Context — uses a strategy
class Sorter {
    std::unique_ptr<SortStrategy> strategy_;
public:
    void set_strategy(std::unique_ptr<SortStrategy> strategy) {
        strategy_ = std::move(strategy);
    }

    void sort(std::vector<int>& data) {
        if (!strategy_) throw std::runtime_error("No strategy set");
        std::cout << "Using: " << strategy_->name() << "\n";
        strategy_->sort(data);
    }
};

int main() {
    std::vector<int> data = {5, 2, 8, 1, 9, 3, 7, 4, 6, 0};

    Sorter sorter;

    sorter.set_strategy(std::make_unique<QuickSort>());
    sorter.sort(data);

    // Shuffle and try another strategy
    std::shuffle(data.begin(), data.end(), std::mt19937{42});
    sorter.set_strategy(std::make_unique<StableSort>());
    sorter.sort(data);

    sorter.set_strategy(std::make_unique<PartialSort>(5));
    sorter.sort(data);
}

std::function as Strategy

Using std::function eliminates the need for a strategy hierarchy. Lambdas, function pointers, and functors can all serve as strategies:

#include <iostream>
#include <functional>
#include <vector>
#include <algorithm>

class DataProcessor {
    std::function<void(std::vector<int>&)> sort_strategy_;
    std::function<bool(int, int)> filter_strategy_;

public:
    void set_sort(std::function<void(std::vector<int>&)> strategy) {
        sort_strategy_ = std::move(strategy);
    }

    void set_filter(std::function<bool(int, int)> strategy) {
        filter_strategy_ = std::move(strategy);
    }

    void process(std::vector<int>& data) {
        if (sort_strategy_) sort_strategy_(data);

        if (filter_strategy_) {
            data.erase(
                std::remove_if(data.begin(), data.end(),
                    [this](int v) { return !filter_strategy_(v, 0); }),
                data.end()
            );
        }
    }
};

int main() {
    std::vector<int> data = {15, 3, 8, 22, 1, 10, 5, 99};

    DataProcessor processor;

    // Strategy 1: sort descending
    processor.set_sort([](std::vector<int>& d) {
        std::sort(d.begin(), d.end(), std::greater<>());
    });

    // Strategy 2: filter even numbers
    processor.set_filter([](int v, int) { return v % 2 == 0; });

    processor.process(data);

    std::cout << "Result: ";
    for (int v : data) std::cout << v << " ";
    std::cout << "\n";  // 22 10 8
}

Template Strategy (Compile-Time, Zero Overhead)

#include <iostream>
#include <vector>
#include <algorithm>

// Strategy as a template policy (compile-time selection)
template<typename SortPolicy>
class SortedContainer {
    std::vector<int> data_;
public:
    void add(int value) { data_.push_back(value); }

    void sort() {
        // SortPolicy::sort is resolved at compile time
        SortPolicy::sort(data_);
    }

    void print() const {
        for (int v : data_) std::cout << v << " ";
        std::cout << "\n";
    }
};

// Policy classes
struct AscendingPolicy {
    static void sort(std::vector<int>& d) { std::sort(d.begin(), d.end()); }
};

struct DescendingPolicy {
    static void sort(std::vector<int>& d) {
        std::sort(d.begin(), d.end(), std::greater<>());
    }
};

struct PartialPolicy {
    static void sort(std::vector<int>& d) {
        std::partial_sort(d.begin(), d.begin() + std::min(d.size(), std::size_t{5}), d.end());
    }
};

int main() {
    SortedContainer<AscendingPolicy> asc;
    asc.add(5); asc.add(3); asc.add(8); asc.add(1);
    asc.sort();
    asc.print();  // 1 3 5 8

    SortedContainer<DescendingPolicy> desc;
    desc.add(5); desc.add(3); desc.add(8); desc.add(1);
    desc.sort();
    desc.print();  // 8 5 3 1
}

Strategy Pattern Comparison

ApproachDispatchOverheadFlexibility
Virtual functionsRuntime (vtable)One virtual callChange strategy at runtime
std::functionRuntime (type erasure)Type erasure + allocationAny callable, change at runtime
Template policyCompile-timeZero (inlined)Fixed at compile time

When to use which: Use template strategies when the algorithm is known at compile time (maximum performance). Use virtual functions when you need runtime switching between a fixed set of strategies. Use std::function when you need maximum flexibility (lambdas, function pointers, bound member functions).
