Socket Programming

Socket programming is the foundation of network communication in C++. The BSD socket API (POSIX) provides a unified interface for TCP, UDP, and Unix domain sockets. Understanding sockets is essential for building client-server applications, inter-process communication (IPC), and distributed systems.

Prerequisites: Shared Memory IPC, Signal Handling, Memory-Mapped Files, basic I/O and system calls.

TCP Echo Server

#include <iostream>
#include <string>
#include <cstring>
#include <unistd.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>

// RAII wrapper for socket file descriptor
class Socket {
    int fd_ = -1;
public:
    Socket() = default;
    explicit Socket(int fd) : fd_(fd) {}
    ~Socket() { if (fd_ != -1) ::close(fd_); }
    Socket(const Socket&) = delete;
    Socket& operator=(const Socket&) = delete;
    Socket(Socket&& other) noexcept : fd_(other.fd_) { other.fd_ = -1; }
    operator int() const { return fd_; }
    bool valid() const { return fd_ != -1; }
};

int main() {
    // 1. Create TCP socket
    Socket server_fd(::socket(AF_INET, SOCK_STREAM, 0));
    if (!server_fd.valid()) {
        std::cerr << "socket() failed: " << strerror(errno) << "\n";
        return 1;
    }

    // 2. Allow address reuse (avoid TIME_WAIT issues)
    int opt = 1;
    ::setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    // 3. Bind to port 8080
    sockaddr_in addr{};
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    addr.sin_port = htons(8080);

    if (::bind(server_fd, (sockaddr*)&addr, sizeof(addr)) < 0) {
        std::cerr << "bind() failed: " << strerror(errno) << "\n";
        return 1;
    }

    // 4. Listen for connections (backlog = 5)
    if (::listen(server_fd, 5) < 0) {
        std::cerr << "listen() failed: " << strerror(errno) << "\n";
        return 1;
    }

    std::cout << "Echo server listening on port 8080\n";

    // 5. Accept and handle connections
    while (true) {
        sockaddr_in client_addr{};
        socklen_t client_len = sizeof(client_addr);

        Socket client_fd(::accept(server_fd,
            (sockaddr*)&client_addr, &client_len));
        if (!client_fd.valid()) {
            std::cerr << "accept() failed\n";
            continue;
        }

        char client_ip[INET_ADDRSTRLEN];
        ::inet_ntop(AF_INET, &client_addr.sin_addr,
                     client_ip, sizeof(client_ip));
        std::cout << "Connection from " << client_ip
                  << ":" << ntohs(client_addr.sin_port) << "\n";

        // Echo loop: read and send back
        char buffer[1024];
        while (true) {
            ssize_t bytes_read = ::recv(client_fd, buffer, sizeof(buffer) - 1, 0);
            if (bytes_read <= 0) break;  // 0 = disconnected, -1 = error

            buffer[bytes_read] = '\0';
            ::send(client_fd, buffer, static_cast<size_t>(bytes_read), 0);
        }
        std::cout << "Client disconnected\n";
    }
}

TCP Client

#include <iostream>
#include <string>
#include <cstring>
#include <unistd.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>

int main() {
    int sockfd = ::socket(AF_INET, SOCK_STREAM, 0);
    if (sockfd < 0) {
        std::cerr << "socket() failed\n";
        return 1;
    }

    sockaddr_in server_addr{};
    server_addr.sin_family = AF_INET;
    server_addr.sin_port = htons(8080);
    ::inet_pton(AF_INET, "127.0.0.1", &server_addr.sin_addr);

    if (::connect(sockfd, (sockaddr*)&server_addr, sizeof(server_addr)) < 0) {
        std::cerr << "connect() failed\n";
        ::close(sockfd);
        return 1;
    }

    std::cout << "Connected to echo server\n";

    std::string line;
    while (std::getline(std::cin, line)) {
        if (line == "/quit") break;
        ::send(sockfd, line.data(), line.size(), 0);

        char buffer[1024];
        ssize_t n = ::recv(sockfd, buffer, sizeof(buffer) - 1, 0);
        if (n > 0) {
            buffer[n] = '\0';
            std::cout << "Echo: " << buffer << "\n";
        }
    }

    ::close(sockfd);
    return 0;
}

Non-Blocking I/O Basics

By default, sockets block on accept(), recv(), and send(). For handling multiple connections, use non-blocking I/O with O_NONBLOCK flag:

// Set socket to non-blocking
int flags = ::fcntl(sockfd, F_GETFL, 0);
::fcntl(sockfd, F_SETFL, flags | O_NONBLOCK);

// Now recv() returns -1 with errno == EAGAIN/EWOULDBLOCK
// when no data is available (instead of blocking)

I/O Multiplexing: select / poll / epoll

System CallComplexityMax FDsNotes
select()O(n) per callFD_SETSIZE (1024)Portable, oldest API
poll()O(n) per callNo limitNo FD_SETSIZE limit
epoll (Linux)O(1) per eventNo practical limitBest for many connections
kqueue (BSD/macOS)O(1) per eventNo practical limitmacOS equivalent of epoll

Best practice: Wrap socket file descriptors in RAII classes to prevent resource leaks. Always check return values from socket API calls. Handle SIGPIPE — see Signal Handling for details.

Thread safety: A single socket should not be used concurrently from multiple threads for read/write without synchronization. The typical pattern is one thread per connection, or a single event loop using epoll/kqueue.
