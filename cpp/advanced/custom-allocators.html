Custom Allocators

C++ allows you to replace the default operator new / malloc allocator used by STL containers with your own. Custom allocators are essential for performance-critical systems like telecom stacks where deterministic allocation, cache locality, and avoiding fragmentation are critical requirements. This topic builds on Lock-Free Programming patterns and leads into Placement New & Memory Pools.

The Allocator Protocol

An STL-compatible allocator must implement the following interface. Since C++11, allocators are stateless by default and do not need to be propagated on copy:

#include <memory>
#include <cstddef>
#include <cstdint>
#include <new>

// Minimum allocator interface (C++11+)
template<typename T>
class TrackingAllocator {
public:
    using value_type = T;
    using pointer = T*;
    using size_type = std::size_t;
    using difference_type = std::ptrdiff_t;

    // Rebind: allows containers to get allocator for different type
    template<typename U>
    struct rebind {
        using other = TrackingAllocator<U>;
    };

    TrackingAllocator() noexcept = default;

    template<typename U>
    TrackingAllocator(const TrackingAllocator<U>&) noexcept {}

    T* allocate(std::size_t n) {
        std::cout << "Allocating " << n * sizeof(T) << " bytes\n";
        return static_cast<T*>(::operator new(n * sizeof(T)));
    }

    void deallocate(T* p, std::size_t n) noexcept {
        std::cout << "Deallocating " << n * sizeof(T) << " bytes\n";
        ::operator delete(p);
    }

    // C++11: propagate_on_container_copy_assignment etc.
    using propagate_on_container_copy_assignment = std::true_type;
    using propagate_on_container_move_assignment = std::true_type;
    using propagate_on_container_swap = std::true_type;
    using is_always_equal = std::false_type;
};

// Equality operators (required by STL containers)
template<typename T, typename U>
bool operator==(const TrackingAllocator<T>&, const TrackingAllocator<U>&) noexcept {
    return true;
}

template<typename T, typename U>
bool operator!=(const TrackingAllocator<T>& a, const TrackingAllocator<U>& b) noexcept {
    return !(a == b);
}

Using Custom Allocators with STL Containers

#include <vector>
#include <map>
#include <string>

// All allocations for this vector go through TrackingAllocator
std::vector<int, TrackingAllocator<int>> tracked_vec;

// Map with tracking allocator
std::map<std::string, int, std::less<>,
         TrackingAllocator<std::pair<const std::string, int>>> tracked_map;

int main() {
    tracked_vec.push_back(1);  // Allocating 4 bytes
    tracked_vec.push_back(2);  // Allocating 8 bytes (reallocated)
    tracked_vec.reserve(10);   // Allocating 40 bytes
}

Pool Allocator

A pool allocator pre-allocates a fixed block of memory and distributes fixed-size chunks from it. Eliminates per-object allocation overhead and fragmentation:

template<typename T, std::size_t BlockSize = 4096>
class PoolAllocator {
    struct Block {
        alignas(std::max_align_t) char data[BlockSize];
        Block* next = nullptr;
    };

    struct FreeNode {
        FreeNode* next = nullptr;
    };

    Block* blocks_ = nullptr;
    FreeNode* free_list_ = nullptr;
    std::size_t objects_per_block_ = BlockSize / sizeof(T);

    void allocate_block() {
        Block* new_block = new Block();
        new_block->next = blocks_;
        blocks_ = new_block;

        // Add all chunks in the new block to the free list
        auto raw = reinterpret_cast<FreeNode*>(new_block->data);
        for (std::size_t i = 0; i < objects_per_block_; ++i) {
            auto node = reinterpret_cast<FreeNode*>(
                reinterpret_cast<char*>(raw) + i * sizeof(T));
            node->next = free_list_;
            free_list_ = node;
        }
    }

public:
    using value_type = T;
    using propagate_on_container_copy_assignment = std::true_type;
    using is_always_equal = std::false_type;

    template<typename U> struct rebind { using other = PoolAllocator<U, BlockSize>; };

    PoolAllocator() noexcept = default;

    template<typename U>
    PoolAllocator(const PoolAllocator<U, BlockSize>&) noexcept {}

    T* allocate(std::size_t n) {
        if (n != 1) {
            throw std::bad_alloc{};  // pool only supports single-object alloc
        }
        if (!free_list_) {
            allocate_block();
        }
        FreeNode* node = free_list_;
        free_list_ = node->next;
        return reinterpret_cast<T*>(node);
    }

    void deallocate(T* p, std::size_t n) noexcept {
        if (n != 1) return;
        auto node = reinterpret_cast<FreeNode*>(p);
        node->next = free_list_;
        free_list_ = node;
    }

    ~PoolAllocator() {
        while (blocks_) {
            Block* next = blocks_->next;
            delete blocks_;
            blocks_ = next;
        }
    }
};

// Equality: pool allocators are unequal (each has its own pool)
template<typename T1, std::size_t B1, typename T2, std::size_t B2>
bool operator==(const PoolAllocator<T1, B1>&, const PoolAllocator<T2, B2>&) { return false; }
template<typename T1, std::size_t B1, typename T2, std::size_t B2>
bool operator!=(const PoolAllocator<T1, B1>& a, const PoolAllocator<T2, B2>& b) { return !(a == b); }

// Usage: pool-allocated list of packets
struct Packet {
    std::uint8_t data[1500];
    std::size_t length = 0;
};

std::list<Packet, PoolAllocator<Packet, 65536>> packet_pool;

Arena Allocator (Stack-Based)

An arena allocator allocates from a pre-allocated buffer with a simple bump pointer. Deallocation is done in bulk (destroy the entire arena). Ideal for per-frame or per-transaction allocations:

class ArenaAllocator {
public:
    static constexpr std::size_t kDefaultSize = 1 << 20;  // 1 MB

    explicit ArenaAllocator(std::size_t size = kDefaultSize) {
        buffer_ = static_cast<char*>(std::aligned_alloc(
            alignof(std::max_align_t), size));
        capacity_ = size;
        offset_ = 0;
    }

    ~ArenaAllocator() { std::free(buffer_); }

    // Non-copyable, non-movable
    ArenaAllocator(const ArenaAllocator&) = delete;
    ArenaAllocator& operator=(const ArenaAllocator&) = delete;

    void* allocate(std::size_t size, std::size_t alignment = alignof(std::max_align_t)) {
        std::size_t aligned_offset = (offset_ + alignment - 1) & ~(alignment - 1);
        if (aligned_offset + size > capacity_) {
            throw std::bad_alloc{};
        }
        void* ptr = buffer_ + aligned_offset;
        offset_ = aligned_offset + size;
        return ptr;
    }

    // Reset: destroy all objects and reset offset
    // Callers must manually destruct objects first
    void reset() { offset_ = 0; }

    std::size_t used() const noexcept { return offset_; }
    std::size_t available() const noexcept { return capacity_ - offset_; }

    // STL-compatible wrapper
    template<typename T>
    class Allocator {
        ArenaAllocator* arena_;
    public:
        using value_type = T;
        template<typename U> struct rebind { using other = Allocator<U>; };

        explicit Allocator(ArenaAllocator* arena) noexcept : arena_(arena) {}
        template<typename U>
        Allocator(const Allocator<U>& other) noexcept : arena_(other.arena_) {}

        T* allocate(std::size_t n) {
            return static_cast<T*>(arena_->allocate(n * sizeof(T), alignof(T)));
        }
        void deallocate(T*, std::size_t) noexcept {
            // Arena deallocates everything at once — no individual free
        }
    };

private:
    char* buffer_ = nullptr;
    std::size_t capacity_ = 0;
    std::size_t offset_ = 0;
};

Telecom Relevance

Telecom stacks allocate millions of small objects (protocol headers, message buffers, UE contexts) per second. A pool allocator for fixed-size objects eliminates fragmentation. An arena allocator for per-packet processing provides zero-cost deallocation. Both are critical for meeting latency and throughput requirements in 5G systems.

Best practice: Profile your allocation patterns before choosing a strategy. Pool allocators win for uniform-size objects. Arena allocators win for per-frame/per-transaction bulk allocation. The default allocator is fine for everything else.
