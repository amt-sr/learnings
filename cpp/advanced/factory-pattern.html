Factory Pattern

The Factory pattern encapsulates object creation behind an interface, separating the "what to create" decision from the "how to use" code. It decouples client code from concrete classes, making it easy to introduce new types without modifying existing code. C++ offers several factory variants with different levels of abstraction.

Prerequisites: Singleton Pattern, Observer Pattern, Strategy Pattern, Virtual Functions & vtable, Abstract Classes.

Factory Variants

PatternStructureWhen to Use
Simple FactoryOne function/object creates different typesFew types, simple logic
Factory MethodVirtual method in base class, overridden by derivedEach subclass creates its own product
Abstract FactoryInterface for families of related objectsMultiple product families (e.g., UI themes)

Simple Factory

#include <iostream>
#include <memory>
#include <string>
#include <stdexcept>

// Product interface
class Message {
public:
    virtual ~Message() = default;
    virtual std::string encode() const = 0;
    virtual std::string type() const = 0;
};

// Concrete products
class RrcSetup : public Message {
public:
    std::string encode() const override { return "RRC_SETUP_REQ"; }
    std::string type() const override { return "RRC"; }
};

class NasAttach : public Message {
public:
    std::string encode() const override { return "NAS_ATTACH_REQ"; }
    std::string type() const override { return "NAS"; }
};

class NgapInitial : public Message {
public:
    std::string encode() const override { return "NGAP_INITIAL_UE"; }
    std::string type() const override { return "NGAP"; }
};

// Simple factory — one function decides what to create
class MessageFactory {
public:
    static std::unique_ptr<Message> create(const std::string& type) {
        if (type == "rrc") return std::make_unique<RrcSetup>();
        if (type == "nas") return std::make_unique<NasAttach>();
        if (type == "ngap") return std::make_unique<NgapInitial>();
        throw std::invalid_argument("Unknown message type: " + type);
    }
};

int main() {
    auto msg1 = MessageFactory::create("rrc");
    auto msg2 = MessageFactory::create("nas");

    std::cout << msg1->type() << ": " << msg1->encode() << "\n";
    std::cout << msg2->type() << ": " << msg2->encode() << "\n";
}

Factory Method

#include <iostream>
#include <memory>

// Product interface
class Serializer {
public:
    virtual ~Serializer() = default;
    virtual void serialize(const std::string& data) = 0;
    virtual std::string format() const = 0;
};

// Concrete products
class JsonSerializer : public Serializer {
public:
    void serialize(const std::string& data) override {
        std::cout << "{\"data\": \"" << data << "\"}\n";
    }
    std::string format() const override { return "JSON"; }
};

class XmlSerializer : public Serializer {
public:
    void serialize(const std::string& data) override {
        std::cout << "<data>" << data << "</data>\n";
    }
    std::string format() const override { return "XML"; }
};

class ProtobufSerializer : public Serializer {
public:
    void serialize(const std::string& data) override {
        std::cout << "protobuf: " << data.size() << " bytes\n";
    }
    std::string format() const override { return "Protobuf"; }
};

// Creator (base class) — defines the factory method
class MessageHandler {
public:
    virtual ~MessageHandler() = default;

    // The factory method — derived classes decide what to create
    virtual std::unique_ptr<Serializer> create_serializer() = 0;

    void process(const std::string& data) {
        auto serializer = create_serializer();
        std::cout << "Using " << serializer->format() << ":\n";
        serializer->serialize(data);
    }
};

// Concrete creators — each creates a specific serializer
class RestHandler : public MessageHandler {
public:
    std::unique_ptr<Serializer> create_serializer() override {
        return std::make_unique<JsonSerializer>();
    }
};

class SoapHandler : public MessageHandler {
public:
    std::unique_ptr<Serializer> create_serializer() override {
        return std::make_unique<XmlSerializer>();
    }
};

class GrpcHandler : public MessageHandler {
public:
    std::unique_ptr<Serializer> create_serializer() override {
        return std::make_unique<ProtobufSerializer>();
    }
};

int main() {
    RestHandler rest;
    SoapHandler soap;
    GrpcHandler grpc;

    rest.process("hello");    // JSON
    soap.process("hello");    // XML
    grpc.process("hello");    // Protobuf
}

Abstract Factory

#include <iostream>
#include <memory>

// Abstract products: two related families
class Encoder { public: virtual ~Encoder() = default; virtual void encode() = 0; };
class Decoder { public: virtual ~Decoder() = default; virtual void decode() = 0; };

// 4G LTE family
class LteEncoder : public Encoder { void encode() override { std::cout << "LTE encode\n"; } };
class LteDecoder : public Decoder { void decode() override { std::cout << "LTE decode\n"; } };

// 5G NR family
class NrEncoder : public Encoder { void encode() override { std::cout << "NR encode\n"; } };
class NrDecoder : public Decoder { void decode() override { std::cout << "NR decode\n"; } };

// Abstract factory — creates a family of related products
class ProtocolFactory {
public:
    virtual ~ProtocolFactory() = default;
    virtual std::unique_ptr<Encoder> create_encoder() = 0;
    virtual std::unique_ptr<Decoder> create_decoder() = 0;
};

// Concrete factories
class LteFactory : public ProtocolFactory {
    std::unique_ptr<Encoder> create_encoder() override { return std::make_unique<LteEncoder>(); }
    std::unique_ptr<Decoder> create_decoder() override { return std::make_unique<LteDecoder>(); }
};

class NrFactory : public ProtocolFactory {
    std::unique_ptr<Encoder> create_encoder() override { return std::make_unique<NrEncoder>(); }
    std::unique_ptr<Decoder> create_decoder() override { return std::make_unique<NrDecoder>(); }
};

void process(ProtocolFactory& factory) {
    auto encoder = factory.create_encoder();
    auto decoder = factory.create_decoder();
    encoder->encode();
    decoder->decode();
}

int main() {
    LteFactory lte;
    NrFactory nr;

    process(lte);  // LTE encode, LTE decode
    process(nr);   // NR encode, NR decode
}

Choosing a factory variant: Use Simple Factory when you have a small number of types and the creation logic is straightforward. Use Factory Method when subclasses should choose their own creation strategy. Use Abstract Factory when you need families of related objects (e.g., different protocol stacks that must be consistent).

Related topics: Singleton Pattern (factory as singleton), Observer Pattern (factory notifies on creation), Strategy Pattern (inject strategies via factory), Virtual Functions (polymorphic dispatch), Abstract Classes (product interfaces).
