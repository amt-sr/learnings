Fixed-Point Arithmetic

Fixed-point arithmetic represents real numbers using integers with an implicit fractional bit position. In telecom DSP, many embedded processors lack hardware floating-point units (FPU), or FPU usage introduces non-deterministic latency. Fixed-point provides deterministic, fast arithmetic for signal processing in PHY and filter implementations.

Q Format Notation

Q313231-1.0 to +0.9999999995~4.6e-10
Q1.141614-2.0 to +1.999941/16384 ~ 6e-5
Q3.121612-8.0 to +7.999761/4096 ~ 2.4e-4
Q17.143214-131072 to +131071.999~6e-5

Why fixed-point: Integer multiplication/addition takes 1-3 cycles on any processor. Floating-point takes 10-30 cycles on embedded cores, or requires a separate FPU. In PHY layer DSP (FFT, FIR filters), performing billions of operations per second requires the deterministic speed of integer arithmetic.

Fixed-Point Operations

  Q15 Multiplication:  a * b = (a * b) >> 15
  Q15 Addition:         a + b = a + b  (no shift)
  Q15 Shift Left:      a * 2 = a << 1
  Q15 Convert from float: Q15_val = (int16_t)(float_val * 32768)
  Q15 Convert to float: float_val = (float)Q15_val / 32768.0f

  Example:
  0.5 in Q15 = 16384  (0.5 * 32768)
  0.25 in Q15 = 8192  (0.25 * 32768)
  0.5 * 0.25 = (16384 * 8192) >> 15
              = 134217728 >> 15
              = 4096 = 0.125 in Q15  (correct!)

C++ Q15 Fixed-Point Class

#include <cstdint>
#include <stdexcept>
#include <type_traits>

// Q1.15 fixed-point (16-bit, range [-1, 1))
class Q15 {
public:
    // Construct from raw integer value
    constexpr explicit Q15(int16_t raw = 0) : val_(raw) {}

    // Construct from float (implicit conversion)
    static constexpr Q15 fromFloat(float f) {
        // Clamp to [-1.0, 1.0)
        f = f < -1.0f ? -1.0f : (f >= 1.0f ? 0.999969f : f);
        return Q15(static_cast<int16_t>(f * 32768.0f));
    }

    // Convert to float
    constexpr float toFloat() const {
        return static_cast<float>(val_) / 32768.0f;
    }

    // Raw access
    constexpr int16_t raw() const { return val_; }

    // Addition (no shift needed)
    constexpr Q15 operator+(Q15 other) const {
        return Q15(static_cast<int16_t>(
            static_cast<int32_t>(val_) + other.val_));
    }

    // Subtraction
    constexpr Q15 operator-(Q15 other) const {
        return Q15(static_cast<int16_t>(
            static_cast<int32_t>(val_) - other.val_));
    }

    // Multiplication with saturation
    Q15 operator*(Q15 other) const {
        int32_t result = (static_cast<int32_t>(val_)
                        * static_cast<int32_t>(other.val_)) >> 15;
        return Q15(saturate(result));
    }

    // Multiplication with rounding
    Q15 mulRound(Q15 other) const {
        int64_t prod = static_cast<int64_t>(val_)
                     * static_cast<int64_t>(other.val_);
        // Round: add 0.5 before shift
        int32_t result = static_cast<int32_t>(
            (prod + (1LL << 14)) >> 15);
        return Q15(saturate(result));
    }

    // MAC (multiply-accumulate) -- core DSP operation
    static Q15 mac(Q15 acc, Q15 a, Q15 b) {
        int64_t prod = static_cast<int64_t>(a.val_)
                     * static_cast<int64_t>(b.val_);
        int64_t sum = static_cast<int64_t>(acc.val_) << 15
                    + prod;
        int32_t result = static_cast<int32_t>(sum >> 15);
        return Q15(saturate(result));
    }

    // Left shift (multiply by power of 2)
    Q15 operator<<(int shift) const {
        int32_t shifted = static_cast<int32_t>(val_) << shift;
        return Q15(saturate(shifted));
    }

    // Comparison operators
    constexpr bool operator==(Q15 other) const { return val_ == other.val_; }
    constexpr bool operator<(Q15 other) const { return val_ < other.val_; }
    constexpr bool operator>(Q15 other) const { return val_ > other.val_; }

private:
    int16_t val_;

    // Saturation: clamp to int16_t range
    static int16_t saturate(int32_t val) {
        if (val > 32767) return 32767;
        if (val < -32768) return -32768;
        return static_cast<int16_t>(val);
    }
};

// Q15 FIR filter (common in telecom DSP)
template<uint8_t N>
class FirFilterQ15 {
public:
    FirFilterQ15(const Q15 (&coeffs)[N]) {
        for (uint8_t i = 0; i < N; ++i)
            coeffs_[i] = coeffs[i];
    }

    Q15 process(Q15 input) {
        // Shift delay line
        for (uint8_t i = N - 1; i > 0; --i)
            delay_[i] = delay_[i - 1];
        delay_[0] = input;

        // Multiply-accumulate
        Q15 acc(0);
        for (uint8_t i = 0; i < N; ++i)
            acc = Q15::mac(acc, coeffs_[i], delay_[i]);
        return acc;
    }

private:
    Q15 delay_[N]{};
    Q15 coeffs_[N];
};

Q15 Class in Action: DSP Example

#include <iostream>
#include <cmath>

// Using the Q15 class from above

int main() {
    // Convert real-world coefficients to Q15
    auto gain       = Q15::fromFloat(0.5f);
    auto offset     = Q15::fromFloat(0.1f);
    auto input      = Q15::fromFloat(0.75f);

    std::cout << "Input:   " << input.toFloat() << "\n";
    std::cout << "Gain:    " << gain.toFloat() << "\n";

    // Apply gain: output = input * gain
    auto amplified = input * gain;
    std::cout << "Amplified: " << amplified.toFloat() << "\n";

    // Add offset: output = amplified + offset
    auto result = amplified + offset;
    std::cout << "With offset: " << result.toFloat() << "\n";

    // FIR filter: low-pass 4-tap filter
    Q15 coeffs[4] = {
        Q15::fromFloat(0.1f),
        Q15::fromFloat(0.2f),
        Q15::fromFloat(0.4f),
        Q15::fromFloat(0.2f),
    };
    FirFilterQ15<4> lpf(coeffs);

    // Process a sample signal
    float samples[] = {1.0f, 0.8f, 0.3f, -0.2f, -0.5f, -0.3f, 0.1f, 0.6f};
    std::cout << "\nLow-pass filtered output:\n";
    for (float s : samples) {
        auto out = lpf.process(Q15::fromFloat(s));
        std::cout << "  " << out.toFloat() << "\n";
    }

    // MAC operation (core DSP primitive)
    auto acc = Q15(0);
    Q15 taps[] = { Q15::fromFloat(0.5f), Q15::fromFloat(0.3f),
                  Q15::fromFloat(0.2f) };
    Q15 samples_q15[] = { Q15::fromFloat(0.8f), Q15::fromFloat(0.6f),
                          Q15::fromFloat(0.4f) };
    for (int i = 0; i < 3; ++i) {
        acc = Q15::mac(acc, taps[i], samples_q15[i]);
    }
    std::cout << "\nDot product (MAC): " << acc.toFloat() << "\n";
    // Expected: 0.5*0.8 + 0.3*0.6 + 0.2*0.4 = 0.4 + 0.18 + 0.08 = 0.66

    return 0;
}

DSP note: The MAC (multiply-accumulate) operation is the fundamental building block of FIR filters, FFTs, and correlation. Modern DSP cores have dedicated MAC instructions that execute in a single cycle. The Q15::mac() function uses 64-bit intermediate results to prevent overflow during accumulation.

Overflow and Saturation

Overflow danger: Adding two Q15 values can overflow 16 bits. Always promote to 32 bits before adding: (int32_t)a + (int32_t)b. After multiplication, the 32-bit intermediate result must be shifted right by 15 and then saturated back to 16 bits. Overflow without saturation causes catastrophic signal distortion.
