CCDU Concept

The CCDU (Centralized, Cloud, Distributed, Unit) framework describes the deployment topologies for 5G RAN components. It defines where CU, DU, and RU are physically placed -- whether centralized in a data center, distributed at edge sites, or integrated in a single unit. Understanding CCDU is essential for designing C++ software that can be deployed across different topologies.

C++ Placement Abstraction

#include <string>
#include <memory>
#include <optional>
#include <unordered_map>

// Deployment location types
enum class Location : uint8_t {
    DATA_CENTER,     // CU centralized in DC
    EDGE_CLOUD,      // CU at regional edge
    CELL_SITE,       // DU/RU at cell site
    INTEGRATED       // All-in-one at cell site
};

// Transport network type between components
enum class TransportType : uint8_t {
    DIRECT_EMBEDDED,   // Shared memory / PCIe
    FIBER_FRONTHAUL,   // Optical fiber (CPRI/eCPRI)
    PACKET_MPLS,       // MPLS network
    PACKET_IP,         // Standard IP network
    PACKET_ROE,        // Radio over Ethernet
};

// Latency budget per link type
struct LinkBudget {
    TransportType type;
    uint32_t      one_way_latency_us;  // Target one-way latency
    uint32_t      jitter_us;           // Allowed jitter
    uint64_t      bandwidth_bps;       // Available bandwidth
};

class DeploymentTopology {
public:
    void addCuLocation(Location loc, const std::string& site_id) {
        cu_location_ = loc;
        cu_site_ = site_id;
    }

    void addDuLocation(Location loc, const std::string& site_id) {
        du_locations_[site_id] = loc;
    }

    void setTransport(const std::string& site_id,
                      TransportType type,
                      const LinkBudget& budget) {
        transport_[site_id] = {type, budget};
    }

    // Check if DU is co-located with RU (no fronthaul needed)
    bool isDuRuColocated(const std::string& site_id) const {
        // If DU is at cell site, it is co-located with RU
        auto it = du_locations_.find(site_id);
        return it != du_locations_.end() &&
               it->second == Location::CELL_SITE;
    }

    // Determine if F1 interface uses IP transport or shared memory
    bool isF1OverIp(const std::string& site_id) const {
        auto it = transport_.find(site_id);
        if (it == transport_.end()) return false;
        return it->second.type != TransportType::DIRECT_EMBEDDED;
    }

    Location getCuLocation() const { return cu_location_; }

private:
    Location cu_location_;
    std::string cu_site_;
    std::unordered_map<std::string, Location> du_locations_;
    std::unordered_map<std::string,
                        std::pair<TransportType, LinkBudget>> transport_;
};

C++ Adaptive Transport Layer

// Abstract transport that adapts to deployment topology
class IF1Transport {
public:
    virtual ~IF1Transport() = default;

    // Send F1AP message (control or user plane)
    virtual int sendF1Message(const uint8_t* data, size_t len) = 0;

    // Receive F1AP message
    virtual int receiveF1Message(uint8_t* buffer, size_t max_len,
                                 size_t* received_len) = 0;

    // Get transport statistics
    virtual uint64_t getTxBytes() const = 0;
    virtual uint64_t getRxBytes() const = 0;
    virtual uint32_t getLatencyUs() const = 0;
};

// Shared memory transport for co-located CU/DU
class SharedMemF1Transport : public IF1Transport {
public:
    explicit SharedMemF1Transport(size_t ring_size = 65536)
        : ring_buffer_(ring_size) {}

    int sendF1Message(const uint8_t* data, size_t len) override {
        // Zero-copy write to shared memory ring buffer
        return ring_buffer_.write(data, len);
    }

    int receiveF1Message(uint8_t* buffer, size_t max_len,
                         size_t* received_len) override {
        return ring_buffer_.read(buffer, max_len, received_len);
    }

    uint32_t getLatencyUs() const override { return 1; }  // ~1 us

private:
    // See circular-buffers.html for implementation
    class SharedMemoryRing { /* ... */ };
    SharedMemoryRing ring_buffer_;
};

// UDP/SCTP transport for distributed CU/DU over IP
class IpF1Transport : public IF1Transport {
public:
    IpF1Transport(const std::string& local_addr, uint16_t local_port,
                  const std::string& remote_addr, uint16_t remote_port)
        : local_addr_(local_addr), local_port_(local_port),
          remote_addr_(remote_addr), remote_port_(remote_port) {}

    int sendF1Message(const uint8_t* data, size_t len) override {
        // Send over SCTP (F1-C) or UDP (F1-U)
        return sendto(sockfd_, data, len, 0,
                     (struct sockaddr*)&remote_addr_,
                     sizeof(remote_addr_));
    }

    int receiveF1Message(uint8_t* buffer, size_t max_len,
                         size_t* received_len) override {
        ssize_t n = recvfrom(sockfd_, buffer, max_len, 0, nullptr, nullptr);
        if (n > 0) *received_len = n;
        return n >= 0 ? 0 : -1;
    }

    uint32_t getLatencyUs() const override { return avg_latency_us_; }

private:
    std::string local_addr_;
    uint16_t   local_port_;
    std::string remote_addr_;
    uint16_t   remote_port_;
    int        sockfd_{-1};
    uint32_t   avg_latency_us_{5000};  // ~5 ms
};

// Transport factory based on topology
std::unique_ptr<IF1Transport> createF1Transport(
    const DeploymentTopology& topo, const std::string& site_id)
{
    if (!topo.isF1OverIp(site_id)) {
        // CU and DU co-located: use shared memory
        return std::make_unique<SharedMemF1Transport>();
    } else {
        // Distributed: use IP transport
        return std::make_unique<IpF1Transport>(
            "0.0.0.0", 38000,           // local
            "10.0.1.100", 38000         // remote DU/CU
        );
    }
}

Scaling Considerations

Centralized CU Benefits: A single CU can serve 100+ DUs, enabling coordinated multi-point (CoMP), centralized scheduling, and reduced inter-site handover latency. C++ software must handle concurrent UE contexts from multiple DUs efficiently.

  • CU Scaling: A single CU-CP can manage up to 10,000 UEs; CU-UP throughput limited by PDCP processing (typically 10-20 Gbps per instance).
  • DU Scaling: Each DU handles a single cell site with typically 100-500 UEs.
  • RU Scaling: One RU per sector; a cell site typically has 3 sectors (3 RUs per DU).
