RLC Layer

The RLC (Radio Link Control) layer provides segmentation, reassembly, and error correction services. It operates in three modes: Transparent Mode (TM), Unacknowledged Mode (UM), and Acknowledged Mode (AM). RLC sits between PDCP (above) and MAC (below) in the protocol stack.

RLC Mode Comparison

FeatureTMUMAM
HeaderNone2-3 bytes (fixed)2-3 bytes (fixed)
SegmentationNoYesYes
ARQ RetransmissionNoNoYes (RLC status PDU)
ReassemblyNoYes (in-order)Yes (in-order)
Use CaseBCCH, PCCHVoIP, videoInternet data, signaling (SRB)
SN SizeN/A6 or 12 bits12 or 18 bits
Typical BearersSRB0, BCCHDRB for real-timeSRB1/2, DRB for data

RLC PDU Structure

  RLC AM Data PDU (16-bit SN):
  +-------+-----+-------+-------+-------+---+
  | D/C=1 | P   | SN(12) | SO(16)|  Data |   |
  | 1 bit |1bit | 12 bits| 16bits|       |   |
  +-------+-----+-------+-------+-------+---+
  D/C = Data/Control (1=Data PDU)
  P   = Poll bit (request STATUS from peer)
  SN  = Sequence Number (12 or 18 bits)
  SO  = Segment Offset

  RLC UM Data PDU (12-bit SN):
  +-------+-----+-------+-------+-------+
  | D/C=1 | FI  | SN(12)| SO(16)|  Data |
  | 1 bit |2bit | 12bits| 16bits|       |
  +-------+-----+-------+-------+-------+
  FI  = Framing Info (first/last byte segmentation status)

  RLC Status PDU (AM only):
  +-------+--------+--------+------+
  | D/C=0 | CPT=0  | ACK_SN | NACK |
  | 1 bit |  3 bits| 12bits | bits |
  +-------+--------+--------+------+

C++ RLC Entity Implementation

#include <queue>
#include <vector>
#include <unordered_map>
#include <memory>
#include <cstdint>

// RLC operating mode
enum class RlcMode : uint8_t {
    TRANSPARENT,
    UNACKNOWLEDGED,
    ACKNOWLEDGED
};

// RLC SDU received from PDCP (upper layer)
struct RlcSdu {
    uint32_t    channel_id;
    uint32_t    sdu_size;
    std::vector<uint8_t> data;
};

// RLC PDU to send to MAC (lower layer)
struct RlcPdu {
    uint32_t    channel_id;
    bool        is_data;        // true = data PDU, false = control PDU
    uint16_t    byte_size;      // Total PDU size including header
    std::vector<uint8_t> data;
};

// RLC segment buffer entry (for AM retransmission)
struct RlcAmSegment {
    uint32_t    sn;             // Sequence number
    uint16_t    so;             // Segment offset
    uint16_t    segment_size;
    bool        acked;
    uint8_t     retx_count;
    std::vector<uint8_t> data;
};

// Abstract RLC entity base
class RlcEntity {
public:
    virtual ~RlcEntity() = default;

    virtual void handleSduFromUpper(const RlcSdu& sdu) = 0;
    virtual void handlePduFromLower(const uint8_t* pdu, size_t len) = 0;

    // MAC requests data for transmission
    virtual size_t getPduSize(uint32_t grant_bytes) = 0;
    virtual bool buildPdu(uint8_t* buffer, uint32_t max_len,
                          uint32_t& pdu_len) = 0;
};

// RLC AM Entity (most complex, with ARQ)
class RlcAmEntity : public RlcEntity {
public:
    explicit RlcAmEntity(uint32_t channel_id, uint16_t sn_modulus = 4096)
        : channel_id_(channel_id), sn_modulus_(sn_modulus) {}

    void handleSduFromUpper(const RlcSdu& sdu) override {
        // Buffer SDU for segmentation
        tx_buffer_.push(sdu);
    }

    void handlePduFromLower(const uint8_t* pdu, size_t len) override {
        // Parse RLC PDU header
        bool is_data = (pdu[0] & 0x80) != 0;
        if (!is_data) {
            handleStatusPdu(pdu, len);
        } else {
            handleDataPdu(pdu, len);
        }
    }

    size_t getPduSize(uint32_t grant_bytes) override {
        // Calculate how much data can fit in the grant
        if (pending_retx_.empty() && tx_buffer_.empty())
            return 0;
        uint32_t header_size = 4;  // AM header
        return std::min(grant_bytes, header_size + getMaxPayload(grant_bytes - header_size));
    }

    bool buildPdu(uint8_t* buffer, uint32_t max_len,
                  uint32_t& pdu_len) override {
        // Priority: 1) retransmissions, 2) new segments
        if (!pending_retx_.empty()) {
            return buildRetxPdu(buffer, max_len, pdu_len);
        }
        if (!tx_buffer_.empty()) {
            return buildNewSegment(buffer, max_len, pdu_len);
        }
        return false;
    }

    // Process received status PDU (ACK/NACK from peer)
    void handleStatusPdu(const uint8_t* pdu, size_t len) {
        // Parse ACK_SN -- mark all PDU with SN < ACK_SN as acked
        uint16_t ack_sn = parseAckSn(pdu, len);
        for (auto& seg : tx_segments_) {
            if (seg.sn < ack_sn) {
                seg.acked = true;
            }
        }
        // Handle NACK entries -- retransmit specific segments
        parseAndHandleNacks(pdu, len);
    }

    // Process received data PDU
    void handleDataPdu(const uint8_t* pdu, size_t len) {
        // Parse header: SN, SO
        uint16_t sn = parseSn(pdu);
        uint16_t so = parseSo(pdu);

        // Place in reassembly buffer
        auto& entry = rx_buffer_[sn];
        entry.segments.push_back({so, len - getHeaderSize(pdu)});

        // Check if complete SDU can be reassembled
        if (isComplete(sn)) {
            reassembleAndDeliver(sn);
        }

        // Send status report if needed (missing segments)
        if (hasGaps(sn)) {
            sendStatusReport(sn);
        }
    }

    uint32_t nextTxSn() const { return tx_next_; }

private:
    uint32_t channel_id_;
    uint16_t sn_modulus_;
    uint32_t tx_next_{0};          // Next SN to transmit
    uint32_t tx_ack_{0};           // Highest ACKed SN
    uint32_t rx_next_{0};          // Next expected SN (receive)

    std::queue<RlcSdu>               tx_buffer_;       // SDUs awaiting segmentation
    std::vector<RlcAmSegment>       tx_segments_;     // Segments sent (for retx)
    std::deque<RlcAmSegment>        pending_retx_;    // Pending retransmissions

    struct RxReassemblyEntry {
        std::vector<std::pair<uint16_t, uint16_t>> segments; // (offset, length)
    };
    std::unordered_map<uint32_t, RxReassemblyEntry> rx_buffer_;

    bool buildNewSegment(uint8_t* buf, uint32_t max_len, uint32_t& pdu_len);
    bool buildRetxPdu(uint8_t* buf, uint32_t max_len, uint32_t& pdu_len);
    uint16_t parseAckSn(const uint8_t* pdu, size_t len);
    uint16_t parseSn(const uint8_t* pdu);
    uint16_t parseSo(const uint8_t* pdu);
    bool isComplete(uint32_t sn);
    bool hasGaps(uint32_t sn);
    void reassembleAndDeliver(uint32_t sn);
    void sendStatusReport(uint32_t nack_sn);
    void parseAndHandleNacks(const uint8_t* pdu, size_t len);
    uint32_t getMaxPayload(uint32_t available);
};

RLC UM and TM Entities

#include <queue>
#include <vector>
#include <cstdint>

// RLC UM Entity (no retransmission, used for VoIP/video)
class RlcUmEntity : public RlcEntity {
public:
    explicit RlcUmEntity(uint32_t channel_id,
                         uint16_t sn_modulus = 4096)
        : channel_id_(channel_id), sn_modulus_(sn_modulus) {}

    void handleSduFromUpper(const RlcSdu& sdu) override {
        tx_buffer_.push(sdu);
    }

    void handlePduFromLower(const uint8_t* pdu, size_t len) override {
        // Parse UM PDU header
        uint16_t sn = (pdu[0] & 0x03) << 8 | pdu[1]; // 10-bit SN
        uint8_t  fi = (pdu[0] >> 2) & 0x03;

        size_t header_size = 2;
        const uint8_t* payload = pdu + header_size;
        size_t payload_len = len - header_size;

        // Store in reassembly buffer (keyed by SN)
        auto& entry = rx_reassembly_[sn];
        // FI indicates if first/last bytes of SDU are present
        if (fi == 0x00 || fi == 0x02) {
            // First segment of SDU
            entry.segments.clear();
            entry.expected_size = 0;
        }
        entry.segments.push_back({payload, payload_len});
        entry.expected_size += payload_len;

        // Deliver if FI indicates last segment
        if (fi == 0x00 || fi == 0x01) {
            deliverToPdcp(sn);
        }
    }

    size_t getPduSize(uint32_t grant_bytes) override {
        if (tx_buffer_.empty()) return 0;
        uint32_t header = 2;  // UM header (2 bytes)
        return std::min(grant_bytes, header +
            tx_buffer_.front().data.size());
    }

    bool buildPdu(uint8_t* buffer, uint32_t max_len,
                  uint32_t& pdu_len) override {
        if (tx_buffer_.empty()) return false;

        auto& sdu = tx_buffer_.front();
        uint32_t header = 2;
        uint32_t payload = std::min(max_len - header,
                                   static_cast<uint32_t>(sdu.data.size()));

        // Write UM header: D/C=1, FI=00, SN
        buffer[0] = 0x80 | (tx_sn_ >> 8) & 0x03;
        buffer[1] = tx_sn_ & 0xFF;
        std::memcpy(buffer + header, sdu.data.data(), payload);
        pdu_len = header + payload;

        // Advance state
        sdu.data.erase(sdu.data.begin(),
                       sdu.data.begin() + payload);
        if (sdu.data.empty()) {
            tx_buffer_.pop();
        }
        tx_sn_ = (tx_sn_ + 1) % sn_modulus_;
        return true;
    }

    uint32_t nextTxSn() const { return tx_sn_; }

private:
    uint32_t channel_id_;
    uint16_t sn_modulus_;
    uint32_t tx_sn_{0};
    uint32_t rx_sn_{0};

    std::queue<RlcSdu> tx_buffer_;

    struct RxEntry {
        std::vector<std::pair<const uint8_t*, size_t>> segments;
        size_t expected_size;
    };
    std::unordered_map<uint32_t, RxEntry> rx_reassembly_;

    void deliverToPdcp(uint32_t sn) {
        // Reassemble and pass up to PDCP layer
        rx_reassembly_.erase(sn);
    }
};

// RLC TM Entity (simplest, no header, no segmentation)
class RlcTmEntity : public RlcEntity {
public:
    explicit RlcTmEntity(uint32_t channel_id)
        : channel_id_(channel_id) {}

    void handleSduFromUpper(const RlcSdu& sdu) override {
        // No segmentation — pass through as-is
        pending_pdus_.push_back(sdu);
    }

    void handlePduFromLower(const uint8_t* pdu, size_t len) override {
        // No header to parse — deliver directly
        RlcSdu sdu;
        sdu.channel_id = channel_id_;
        sdu.data.assign(pdu, pdu + len);
        // Deliver to PDCP (upper layer callback)
    }

    size_t getPduSize(uint32_t grant_bytes) override {
        if (pending_pdus_.empty()) return 0;
        return std::min(grant_bytes,
            static_cast<uint32_t>(pending_pdus_.front().data.size()));
    }

    bool buildPdu(uint8_t* buffer, uint32_t max_len,
                  uint32_t& pdu_len) override {
        if (pending_pdus_.empty()) return false;
        auto& sdu = pending_pdus_.front();
        uint32_t len = std::min(max_len,
            static_cast<uint32_t>(sdu.data.size()));
        std::memcpy(buffer, sdu.data.data(), len);
        pdu_len = len;
        pending_pdus_.erase(pending_pdus_.begin());
        return true;
    }

private:
    uint32_t channel_id_;
    std::vector<RlcSdu> pending_pdus_;
};

RLC Buffer Management

Buffer Tips: RLC entities maintain transmit and receive buffers that must handle bursty traffic patterns. Use memory pools to pre-allocate buffer space and avoid dynamic allocation in the real-time path. The reassembly buffer needs a timer for discarding incomplete SDUs (t-Reassembly timer).
