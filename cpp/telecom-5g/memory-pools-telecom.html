Memory Pools in Telecom

In telecom data-path processing, calling malloc/new per packet is unacceptable. Heap allocation introduces non-deterministic latency (1-100 us), memory fragmentation, and cache thrashing. Memory pools solve this by pre-allocating fixed-size blocks that can be acquired and released in O(1) time.

Why Memory Pools?

Aspectmalloc/newMemory Pool
Allocation time1-100 us (non-deterministic)10-50 ns (constant)
FragmentationYes (over time)None (fixed-size blocks)
Cache behaviorPoor (scattered)Good (contiguous region)
Thread safetyGlobal lock in glibcPer-thread pool (no lock)
DeallocationComplex (coalescing)O(1) return to pool

Data path rule: In production RAN systems, the packet processing path must never call malloc, free, new, or delete. All buffers are pre-allocated from pools at system startup. Violating this rule causes tail-latency spikes and potential deadline misses.

Pool Strategies

  • Fixed-size pool: All blocks same size (simplest, most common). Used for packet buffers, mbufs.
  • Ring buffer pool: Index-based free list using a circular buffer. Lock-free with atomics.
  • Per-thread pool: Each processing core has its own pool. No synchronization needed.
  • Slab allocator: Multiple pools for different sizes. One pool per size class.

C++ Simple Memory Pool

#include <vector>
#include <atomic>
#include <cstdint>
#include <new>

// Fixed-size memory pool with lock-free free list
template<typename T, uint32_t POOL_SIZE = 4096>
class MemoryPool {
public:
    MemoryPool() {
        // Pre-allocate all blocks contiguously
        blocks_.resize(POOL_SIZE);
        // Initialize free list: all blocks are free
        for (uint32_t i = 0; i < POOL_SIZE; ++i) {
            free_list_[i] = i;
        }
        free_count_.store(POOL_SIZE, std::memory_order_relaxed);
    }

    // Allocate one object from the pool
    T* allocate() {
        uint32_t idx = free_count_.fetch_sub(1,
                        std::memory_order_acq_rel) - 1;
        if (idx >= POOL_SIZE) {
            // Pool exhausted
            free_count_.fetch_add(1, std::memory_order_relaxed);
            return nullptr;
        }
        return &blocks_[free_list_[idx]];
    }

    // Return object to pool
    void deallocate(T* ptr) {
        if (!ptr) return;
        uint32_t block_idx = ptr - &blocks_[0];
        uint32_t slot = free_count_.fetch_add(1,
                        std::memory_order_acq_rel);
        if (slot < POOL_SIZE) {
            free_list_[slot] = block_idx;
        }
    }

    // Get number of available blocks
    [[nodiscard]] uint32_t available() const {
        return free_count_.load(std::memory_order_relaxed);
    }

    // Pre-construct all objects (optional)
    void constructAll() {
        for (auto& block : blocks_) {
            new (&block) T();
        }
    }

private:
    alignas(64) std::vector<T> blocks_;       // Storage
    alignas(64) uint32_t free_list_[POOL_SIZE]; // Free indices
    alignas(64) std::atomic<uint32_t> free_count_;
};

// Example usage: packet buffer pool
struct PacketBuffer {
    static constexpr size_t MAX_SIZE = 2048;
    uint8_t data[MAX_SIZE];
    uint16_t length;
    uint32_t timestamp_us;
    uint8_t  lcid;  // Logical channel ID

    void reset() {
        length = 0;
        timestamp_us = 0;
        lcid = 0;
    }
};

// Per-core packet pool
class PerCorePacketPool {
public:
    static constexpr int MAX_CORES = 16;

    void init(int num_cores) {
        for (int i = 0; i < num_cores && i < MAX_CORES; ++i) {
            pools_[i].constructAll();
        }
        num_cores_ = num_cores;
    }

    PacketBuffer* allocate(int core_id) {
        if (core_id < 0 || core_id >= num_cores_) return nullptr;
        return pools_[core_id].allocate();
    }

    void deallocate(int core_id, PacketBuffer* buf) {
        if (core_id < 0 || core_id >= num_cores_) return;
        pools_[core_id].deallocate(buf);
    }

private:
    MemoryPool<PacketBuffer, 8192> pools_[MAX_CORES];
    int num_cores_{0};
};

Multi-Size Slab Allocator

Telecom systems need buffers of different sizes for different layers. A slab allocator manages multiple fixed-size pools:

#include <cstdint>
#include <array>
#include <atomic>
#include <new>
#include <iostream>

// Slab allocator: multiple pools for different buffer sizes
class SlabAllocator {
public:
    static constexpr int NUM_SIZES = 4;

    struct SizeClass {
        uint32_t block_size;
        uint32_t pool_count;     // Number of blocks in this pool
    };

    // Telecom buffer sizes
    static constexpr std::array<SizeClass, NUM_SIZES> SIZE_CLASSES = {{
        {  256, 4096 },  // Small: MAC headers, control PDU
        { 2048, 2048 },  // Medium: RLC/PDCP SDUs
        { 8192, 1024 },  // Large: Transport blocks
        { 65536, 128 },  // Jumbo: Full MAC PDUs, reassembly
    }};

    SlabAllocator() {
        for (int s = 0; s < NUM_SIZES; ++s) {
            pools_[s] = std::make_unique<uint8_t[]>(
                SIZE_CLASSES[s].block_size * SIZE_CLASSES[s].pool_count);
            // Initialize free list (stack-based, per size class)
            free_stacks_[s] = std::make_unique<uintptr_t[]>(
                SIZE_CLASSES[s].pool_count);
            for (uint32_t i = 0; i < SIZE_CLASSES[s].pool_count; ++i) {
                free_stacks_[s][i] = i;  // Index into pool
            }
            free_counts_[s].store(SIZE_CLASSES[s].pool_count,
                                  std::memory_order_relaxed);
        }
    }

    // Allocate buffer of at least 'size' bytes
    void* allocate(uint32_t size) {
        int size_class = findSizeClass(size);
        if (size_class < 0) return nullptr;

        uint32_t idx = free_counts_[size_class].fetch_sub(
            1, std::memory_order_acq_rel) - 1;
        if (idx >= SIZE_CLASSES[size_class].pool_count) {
            free_counts_[size_class].fetch_add(
                1, std::memory_order_relaxed);
            return nullptr;  // Pool exhausted
        }

        uint32_t block_idx = free_stacks_[size_class][idx];
        return &pools_[size_class][block_idx *
               SIZE_CLASSES[size_class].block_size];
    }

    void deallocate(void* ptr, uint32_t size) {
        if (!ptr) return;
        int size_class = findSizeClass(size);
        if (size_class < 0) return;

        // Calculate block index from pointer
        uintptr_t base = reinterpret_cast<uintptr_t>(pools_[size_class].get());
        uintptr_t p = reinterpret_cast<uintptr_t>(ptr);
        uint32_t block_idx = static_cast<uint32_t>(
            (p - base) / SIZE_CLASSES[size_class].block_size);

        uint32_t slot = free_counts_[size_class].fetch_add(
            1, std::memory_order_acq_rel);
        if (slot < SIZE_CLASSES[size_class].pool_count) {
            free_stacks_[size_class][slot] = block_idx;
        }
    }

    // Stats per size class
    void printStats() {
        const char* names[] = {"Small(256)", "Medium(2048)",
                               "Large(8192)", "Jumbo(65536)"};
        for (int s = 0; s < NUM_SIZES; ++s) {
            uint32_t used = SIZE_CLASSES[s].pool_count -
                free_counts_[s].load(std::memory_order_relaxed);
            std::cout << names[s] << ": "
                      << used << "/" << SIZE_CLASSES[s].pool_count
                      << " used\n";
        }
    }

private:
    int findSizeClass(uint32_t size) {
        for (int s = 0; s < NUM_SIZES; ++s) {
            if (size <= SIZE_CLASSES[s].block_size) return s;
        }
        return -1;
    }

    std::array<std::unique_ptr<uint8_t[]>, NUM_SIZES> pools_;
    std::array<std::unique_ptr<uintptr_t[]>, NUM_SIZES> free_stacks_;
    std::array<std::atomic<uint32_t>, NUM_SIZES> free_counts_{};
};

int main() {
    SlabAllocator alloc;

    // Allocate buffers for different protocol layers
    void* mac_buf    = alloc.allocate(128);   // -> Small pool
    void* rlc_buf    = alloc.allocate(1500);   // -> Medium pool
    void* tb_buf     = alloc.allocate(5000);   // -> Large pool
    void* reasm_buf  = alloc.allocate(50000);  // -> Jumbo pool

    alloc.printStats();
    // Small(256): 1/4096 used
    // Medium(2048): 1/2048 used
    // Large(8192): 1/1024 used
    // Jumbo(65536): 1/128 used

    // Return buffers (size must match allocation size class)
    alloc.deallocate(mac_buf, 256);
    alloc.deallocate(rlc_buf, 2048);
    alloc.deallocate(tb_buf, 8192);
    alloc.deallocate(reasm_buf, 65536);

    std::cout << "\nAfter deallocation:\n";
    alloc.printStats();
    return 0;
}

Pre-Allocation Strategies

  At system startup:
  1. Calculate max buffers needed per layer:
     - PHY: max_TBs * max_segments per TTI
     - MAC: max_LCIDs * max_buffer_size
     - RLC: max_bearers * max_sdus_per_bearer
     - PDCP: max_pdus_for_handover

  2. Allocate from hugepages (NUMA-aware):
     void* pool_mem = mmap(NULL, total_size,
         PROT_READ | PROT_WRITE,
         MAP_PRIVATE | MAP_ANONYMOUS | MAP_HUGETLB,
         -1, 0);

  3. Partition memory into pools:
     - Small pool: 256 bytes (MAC headers, control)
     - Medium pool: 2048 bytes (RLC/PDCP SDUs)
     - Large pool: 16384 bytes (transport blocks)

  4. Per-core partitioning avoids contention:
     Core 0: pool_mem[0..N-1]
     Core 1: pool_mem[N..2N-1]
     ...

NUMA tip: Allocate pool memory on the same NUMA node as the processing core. Use numa_alloc_onnode() or mmap with MPOL_BIND. Cross-NUMA pool access adds 40-80ns per operation, defeating the purpose of using pools.
