PHY Layer

The Physical Layer (Layer 1) in 5G NR is responsible for coding, modulation, and radio transmission. It uses OFDM (Orthogonal Frequency Division Multiplexing) with CP-OFDM in the downlink and DFT-s-OFDM in the uplink. The PHY layer is the most computationally intensive part of the gNB and typically requires DSP/ASIC acceleration.

5G NR Resource Grid

  Time Domain:
  +----------+----------+----------+----------+---------+
  | Slot 0   | Slot 1   | Slot 2   | ...      | Slot N  |
  +----------+----------+----------+----------+---------+
  | 1 slot = 14 OFDM symbols (normal CP)            |
  | 1 ms / 2 / 4 / ... slots depending on SCS         |

  Frequency Domain (1 slot):
  +------+------+------+------+------+---+------+
  | PRB0 | PRB1 | PRB2 | ...  | PRBn |   | PRB99 |
  +------+------+------+------+------+---+------+
  | 12 subcarriers per PRB                          |
  | BW = N_PRBs * 12 * SCS                          |

  Resource Element (RE):
  +------+------+------+------+------+------+------+
  | SC 0 | SC 1 | SC 2 | ...  | SC10 | SC11 |      |  Symbol 0
  +------+------+------+------+------+------+------+
  | SC 0 | SC 1 | SC 2 | ...  | SC10 | SC11 |      |  Symbol 1
  +------+------+------+------+------+------+------+
  | ...                                                |
  +------+------+------+------+------+------+------+
  | SC 0 | SC 1 | SC 2 | ...  | SC10 | SC11 |      |  Symbol 13
  +------+------+------+------+------+------+------+

C++ OFDM Resource Grid Simulation

#include <vector>
#include <complex>
#include <cstddef>
#include <cmath>
#include <iomanip>

// Complex sample type for IQ data
using Complex = std::complex<double>;

// NR Resource Grid
class NrResourceGrid {
public:
    // n_prbs: number of PRBs, n_symbols: symbols per slot (14 for normal CP)
    // n_antennas: number of antenna ports
    NrResourceGrid(uint32_t n_prbs, uint32_t n_symbols = 14,
                   uint32_t n_antennas = 1)
        : n_prbs_(n_prbs), n_symbols_(n_symbols), n_antennas_(n_antennas)
    {
        // 12 subcarriers per PRB
        grid_.resize(n_prbs * 12 * n_symbols * n_antennas, Complex(0.0));
    }

    // Access RE: (symbol, prb, subcarrier, antenna)
    Complex& re(uint32_t symbol, uint32_t prb,
                uint32_t subcarrier, uint32_t antenna = 0) {
        size_t idx = (symbol * n_prbs_ * 12 + prb * 12 + subcarrier)
                     * n_antennas_ + antenna;
        return grid_[idx];
    }

    const Complex& re(uint32_t symbol, uint32_t prb,
                      uint32_t subcarrier, uint32_t antenna = 0) const {
        size_t idx = (symbol * n_prbs_ * 12 + prb * 12 + subcarrier)
                     * n_antennas_ + antenna;
        return grid_[idx];
    }

    [[nodiscard]] uint32_t prbs() const { return n_prbs_; }
    [[nodiscard]] uint32_t symbols() const { return n_symbols_; }
    [[nodiscard]] size_t size() const { return grid_.size(); }

    void clear() {
        std::fill(grid_.begin(), grid_.end(), Complex(0.0));
    }

private:
    uint32_t n_prbs_;
    uint32_t n_symbols_;
    uint32_t n_antennas_;
    std::vector<Complex> grid_;
};

// OFDM Modulator: converts frequency-domain grid to time-domain
class OfdmModulator {
public:
    explicit OfdmModulator(uint32_t n_prbs, uint32_t fft_size)
        : n_prbs_(n_prbs), fft_size_(fft_size)
    {
        twiddle_factors_.resize(fft_size);
        for (uint32_t k = 0; k < fft_size; ++k) {
            double angle = -2.0 * M_PI * k / fft_size;
            twiddle_factors_[k] = Complex(std::cos(angle), std::sin(angle));
        }
    }

    // Simple radix-2 IFFT (for simulation; real DSP uses FFTW or custom)
    void ifft(std::vector<Complex>& data) {
        const size_t N = data.size();
        if (N == 0 || (N & (N - 1)) != 0) return;

        // Bit-reversal permutation
        size_t j = 0;
        for (size_t i = 1; i < N; ++i) {
            size_t bit = N >> 1;
            while (j & bit) {
                j ^= bit;
                bit >> 1;
            }
            j ^= bit;
            if (i < j) std::swap(data[i], data[j]);
        }

        // Cooley-Tukey FFT
        for (size_t len = 2; len <= N; len <<= 1) {
            double angle = -2.0 * M_PI / len;
            Complex wlen(std::cos(angle), std::sin(angle));
            for (size_t i = 0; i < N; i += len) {
                Complex w(1.0);
                for (size_t k = 0; k < len / 2; ++k) {
                    Complex u = data[i + k];
                    Complex v = data[i + k + len/2] * w;
                    data[i + k] = u + v;
                    data[i + k + len/2] = u - v;
                    w *= wlen;
                }
            }
        }

        // Scale for IFFT
        for (auto& x : data) x /= static_cast<double>(N);
    }

    // Map resource grid to OFDM symbols
    void mapToOfdmSymbol(const NrResourceGrid& grid,
                          uint32_t symbol_idx,
                          std::vector<Complex>& time_domain) {
        uint32_t fft_size = fft_size_;
        time_domain.resize(fft_size, Complex(0.0));

        uint32_t dc_pos = fft_size / 2;  // DC subcarrier position
        uint32_t n_sc = n_prbs_ * 12;     // Total active subcarriers
        uint32_t start_sc = dc_pos - n_sc / 2;

        // Map active subcarriers
        for (uint32_t sc = 0; sc < n_sc; ++sc) {
            uint32_t prb = sc / 12;
            uint32_t k = sc % 12;
            time_domain[start_sc + sc] = grid.re(symbol_idx, prb, k);
        }

        // Apply IFFT
        ifft(time_domain);
    }

private:
    uint32_t n_prbs_;
    uint32_t fft_size_;
    std::vector<Complex> twiddle_factors_;
};
