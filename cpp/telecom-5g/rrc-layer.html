RRC Layer

RRC (Radio Resource Control) is the control plane protocol that manages UE connection states, radio bearers, measurements, mobility, and security configuration. RRC messages are ASN.1 encoded and carried over SRB0/SRB1/SRB2 using RLC AM or TM mode.

RRC State Machine (5G NR)

                  RRC_IDLE
                 |        |
                 | RRC Setup / Resume
                 v        |
              RRC_INACTIVE |
             |        ^    |
             | Resume |     |
             v        |    |
           RRC_CONNECTED --------+
             |  Release        |
             | RRC Release      |
             v                  |
          RRC_INACTIVE  ------+
             | Resume          |
             +--------->--------+
                    (Paging -> Resume)

  Key differences from LTE:
  - RRC_INACTIVE is new in 5G (like LTE connected-but-suspended)
  - UE keeps UE context in CN and gNB while inactive
  - Resume is faster than full setup

C++ RRC State Machine Implementation

#include <string>
#include <unordered_map>
#include <functional>
#include <cstdint>

// RRC states (5G NR)
enum class RrcState : uint8_t {
    IDLE,
    INACTIVE,
    CONNECTED
};

// RRC message types (simplified)
enum class RrcMessageType : uint8_t {
    RRC_SETUP_REQUEST,
    RRC_SETUP,
    RRC_SETUP_COMPLETE,
    RRC_RECONFIGURATION,
    RRC_RECONFIGURATION_COMPLETE,
    RRC_RELEASE,
    RRC_RESUME_REQUEST,
    RRC_RESUME,
    RRC_RESUME_COMPLETE,
    RRC_REESTABLISHMENT_REQUEST,
    RRC_REESTABLISHMENT,
    SYSTEM_INFORMATION,
    PAGING,
    MEASUREMENT_REPORT,
    RRC_REJECT
};

// UE context maintained in gNB
struct UeRrcContext {
    uint32_t           ue_id;           // gNB UE ID
    uint16_t           rnti;            // C-RNTI or I-RNTI
    RrcState           state;
    uint32_t           cell_id;         // Serving cell

    // Security
    bool               security_activated;
    uint8_t            kgnb[32];        // gNB security key
    uint8_t            krrenc[16];      // RRC ciphering key
    uint8_t            krrint[16];      // RRC integrity key

    // Bearers
    uint8_t            srbs[4];         // SRB IDs configured (SRB0-3)
    std::vector<uint8_t> drbs;         // DRB IDs configured

    // Capabilities
    uint8_t            release_version;  // UE 5G release

    // Timers
    uint32_t           t310_expiry;     // Radio link failure timer
    uint32_t           t311_expiry;     // Reestablishment timer
};

// RRC state machine for one UE
class RrcStateMachine {
public:
    explicit RrcStateMachine(uint32_t ue_id)
        : ctx_({ue_id, 0, RrcState::IDLE, 0, false, {}, {}, {},
               {}, {}, 15, 0, 0}) {}

    // Handle incoming RRC message
    void handleMessage(RrcMessageType msg_type,
                       const uint8_t* msg, size_t len) {
        switch (ctx_.state) {
            case RrcState::IDLE:
                handleIdle(msg_type, msg, len);
                break;
            case RrcState::INACTIVE:
                handleInactive(msg_type, msg, len);
                break;
            case RrcState::CONNECTED:
                handleConnected(msg_type, msg, len);
                break;
        }
    }

    RrcState getState() const { return ctx_.state; }
    const UeRrcContext& getContext() const { return ctx_; }

private:
    UeRrcContext ctx_;

    void transition(RrcState new_state) {
        RrcState old = ctx_.state;
        ctx_.state = new_state;
        onStateTransition(old, new_state);
    }

    // === IDLE state handlers ===
    void handleIdle(RrcMessageType msg_type,
                    const uint8_t* msg, size_t len) {
        switch (msg_type) {
            case RrcMessageType::RRC_SETUP_REQUEST:
                // Parse setup request, check PLMN/TA
                onRrcSetupRequest(msg, len);
                // Send RRC SETUP with SRB1 config
                transition(RrcState::CONNECTED);
                sendRrcSetup();
                break;

            case RrcMessageType::RRC_RESUME_REQUEST:
                // UE resuming from INACTIVE
                onRrcResumeRequest(msg, len);
                transition(RrcState::CONNECTED);
                sendRrcResume();
                break;

            case RrcMessageType::PAGING:
                // Not applicable (gNB sends paging, doesn't receive)
                break;

            default:
                // Unexpected message in IDLE
                rejectConnection();
                break;
        }
    }

    // === INACTIVE state handlers ===
    void handleInactive(RrcMessageType msg_type,
                        const uint8_t* msg, size_t len) {
        switch (msg_type) {
            case RrcMessageType::RRC_RESUME_REQUEST:
                transition(RrcState::CONNECTED);
                sendRrcResume();
                break;

            case RrcMessageType::RRC_REESTABLISHMENT_REQUEST:
                // Re-establish after radio link failure
                transition(RrcState::CONNECTED);
                sendRrcReestablishment();
                break;

            default:
                break;
        }
    }

    // === CONNECTED state handlers ===
    void handleConnected(RrcMessageType msg_type,
                         const uint8_t* msg, size_t len) {
        switch (msg_type) {
            case RrcMessageType::RRC_SETUP_COMPLETE:
                onRrcSetupComplete(msg, len);
                // Security activation, DRB setup via E1
                break;

            case RrcMessageType::RRC_RECONFIGURATION_COMPLETE:
                onRrcReconfigComplete(msg, len);
                break;

            case RrcMessageType::RRC_RECONFIGURATION:
                // UE-initiated reconfig (rare)
                break;

            case RrcMessageType::MEASUREMENT_REPORT:
                onMeasurementReport(msg, len);
                // May trigger handover
                break;

            case RrcMessageType::RRC_RELEASE:
                // Release to INACTIVE (suspend) or IDLE
                if (isSuspend(msg, len)) {
                    transition(RrcState::INACTIVE);
                    ctx_.rnti = allocateIrnti();
                } else {
                    transition(RrcState::IDLE);
                }
                break;

            default:
                break;
        }
    }

    // Callbacks (implemented by gNB)
    void onRrcSetupRequest(const uint8_t* msg, size_t len);
    void onRrcSetupComplete(const uint8_t* msg, size_t len);
    void onRrcResumeRequest(const uint8_t* msg, size_t len);
    void onRrcReconfigComplete(const uint8_t* msg, size_t len);
    void onMeasurementReport(const uint8_t* msg, size_t len);
    void onStateTransition(RrcState old, RrcState new_state);

    void sendRrcSetup();
    void sendRrcResume();
    void sendRrcReestablishment();
    void rejectConnection();
    bool isSuspend(const uint8_t* msg, size_t len);
    uint16_t allocateIrnti();
};

RRC Measurement Configuration

// Measurement object for inter-frequency/inter-RAT
struct MeasObject {
    uint32_t    carrier_freq;      // ARFCN of the measured carrier
    uint8_t     object_type;        // SSb, CSI-RS, or inter-RAT
    uint32_t    ssb_periodicity;    // SSB measurement period
    uint32_t    ssb_offset;         // SSB offset in dB
    bool        abs_threshold;      // Absolute threshold or not
};

// Report configuration for measurement reporting
struct ReportConfig {
    uint32_t    report_interval_ms; // Time between reports
    uint8_t     report_type;        // Periodic, event A1-A6, B1-B2
    int8_t      threshold_rsrp;     // RSRP threshold (dBm)
    int8_t      hysteresis_db;      // Hysteresis (dB)
    uint32_t    time_to_trigger_ms; // Time to trigger
};

// Measurement ID linking object + report config
struct MeasId {
    uint32_t meas_id;
    uint32_t object_id;
    uint32_t report_config_id;
};
