PDCP Layer

PDCP (Packet Data Convergence Protocol) sits between RRC/SDAP and RLC. Its primary functions are IP header compression (ROHC), ciphering (encryption), integrity protection, sequence number management, and duplicate packet detection. PDCP is the first protocol layer below the application/transport layer and is essential for security and efficiency.

PDCP in the Protocol Stack

  +-----------+
  | SDAP      |  QoS flow mapping (5G only)
  +-----+-----+
        |      SDAP SDU = PDCP SDU
  +-----+-----+
  | PDCP      |  Header compression, ciphering, integrity
  +-----+-----+
        |      PDCP PDU = RLC SDU
  +-----+-----+
  | RLC       |  Segmentation, ARQ
  +-----+-----+
  | MAC       |
  +-----+-----+
  | PHY       |
  +-----------+

PDCP PDU Structure

  PDCP Data PDU (12-bit SN, with integrity):
  +------+--------+------+--------+-----+----------+
  | D/C  |  SN(12)| R   |  Next  |  H |  Payload |
  | 1bit | 12bits | 1bit | Header | 1b | (ciphered)|
  +------+--------+------+--------+-----+----------+
  D/C   = Data (0) / Control (1) PDU
  SN    = Sequence Number (12 or 18 bits)
  R     = Reserved
  Next  = Depending on PDCP config:
          - H: Header compression (ROHC)
          - I: Integrity (MAC-I for SRB)
          - P: Piggybacked STATUS report
  H     = Header (1 = ROHC compressed header present)

  PDCP Control PDU:
  +------+--------+--------+
  | D/C=1|  CPT   |  ROHC  |
  | 1bit |  3bits | Feedback|
  +------+--------+--------+
  CPT = Control PDU Type:
        000 = PDCP Status Report
        001 = ROHC feedback
        010 = EHC feedback (5G)
        011 = UDC feedback (5G)

C++ PDCP Entity

#include <vector>
#include <deque>
#include <memory>
#include <cstdint>

// Ciphering algorithms
enum class CipheringAlg : uint8_t {
    NONE       = 0,
    NEA0       = 1,  // NULL (no ciphering)
    NEA1       = 2,  // AES-128 (SNOW 3G-based)
    NEA2       = 3,  // SNOW 3G
    NEA3       = 4   // ZUC (Chinese standard)
};

// Integrity algorithms
enum class IntegrityAlg : uint8_t {
    NIA0 = 0,
    NIA1 = 1,  // AES-128 CMAC
    NIA2 = 2,  // SNOW 3G
    NIA3 = 3   // ZUC
};

// PDCP configuration
struct PdcpConfig {
    uint8_t      drb_id;
    uint8_t      sn_size;           // 12 or 18 bits
    bool         is_srb;            // Signaling Radio Bearer
    bool         enable_ciphering;
    bool         enable_integrity;   // Only for SRBs
    CipheringAlg cipher_alg;
    IntegrityAlg integrity_alg;
    bool         enable_rohc;        // Header compression
    uint16_t     t_reordering;       // Reordering timer (ms)
    uint32_t     discard_timer;      // Discard timer (ms)
    uint32_t     max_sn;             // 4096 for 12-bit, 262144 for 18-bit
};

// PDCP COUNT value (used for ciphering/integrity)
struct PdcpCount {
    uint32_t sn;           // Sequence Number
    uint16_t hfn;          // Hyper Frame Number
};

// PDCP SDU (from SDAP or RRC)
struct PdcpSdu {
    uint32_t             channel_id;
    PdcpCount            count;
    std::vector<uint8_t> data;
    uint32_t             timestamp_us;
};

// PDCP entity (TX and RX combined)
class PdcpEntity {
public:
    explicit PdcpEntity(uint8_t drb_id) : drb_id_(drb_id) {}

    void configure(const PdcpConfig& cfg) {
        config_ = cfg;
        next_tx_sn_ = 0;
        next_rx_sn_ = 0;
        tx_hfn_ = 0;
        rx_hfn_ = 0;
    }

    // TX: receive SDU from upper layer, process, send to RLC
    void handleSduFromUpper(const uint8_t* data, size_t len) {
        PdcpSdu sdu;
        sdu.channel_id = drb_id_;
        sdu.count.sn = next_tx_sn_;
        sdu.count.hfn = tx_hfn_;
        sdu.data.assign(data, data + len);

        // Build PDCP PDU
        uint8_t pdu[8192];
        uint32_t pdu_len = 0;

        // Add PDCP header
        pdu_len += addHeader(pdu, sdu.count);

        // Apply header compression (ROHC) if enabled
        if (config_.enable_rohc && !config_.is_srb) {
            pdu_len += compressHeader(pdu + pdu_len, sdu.data);
        } else {
            memcpy(pdu + pdu_len, sdu.data.data(), sdu.data.size());
            pdu_len += sdu.data.size();
        }

        // Cipher the payload
        if (config_.enable_ciphering && config_.cipher_alg != CipheringAlg::NEA0) {
            cipher(pdu + pdu_len - sdu.data.size(),
                   sdu.data.size(), sdu.count);
        }

        // Add integrity (MAC-I) for SRBs
        if (config_.is_srb && config_.enable_integrity) {
            uint32_t mac_i = calculateMacI(pdu, pdu_len, sdu.count);
            pdu_len += appendMacI(pdu + pdu_len, mac_i);
        }

        // Store for retransmission (handover)
        storePdu(sdu.count.sn, pdu, pdu_len);

        // Send to RLC
        sendToRlc(pdu, pdu_len);

        // Advance SN
        next_tx_sn_++;
        if (next_tx_sn_ >= config_.max_sn) {
            next_tx_sn_ = 0;
            tx_hfn_++;
        }
    }

    // RX: receive PDU from RLC, process, deliver to upper layer
    void handlePduFromLower(const uint8_t* pdu, size_t len) {
        // Parse PDCP header
        PdcpCount count{};
        uint32_t header_len = parseHeader(pdu, len, count);

        // Decipher
        if (config_.enable_ciphering && config_.cipher_alg != CipheringAlg::NEA0) {
            decipher(pdu + header_len, len - header_len, count);
        }

        // Verify integrity (for SRBs)
        if (config_.is_srb && config_.enable_integrity) {
            if (!verifyIntegrity(pdu, len, count)) {
                return;  // Discard integrity-failed PDU
            }
        }

        // Header decompression
        uint8_t sdu_buffer[8192];
        uint32_t sdu_len;
        if (config_.enable_rohc && !config_.is_srb) {
            sdu_len = decompressHeader(sdu_buffer, pdu + header_len,
                                       len - header_len);
        } else {
            sdu_len = len - header_len;
            memcpy(sdu_buffer, pdu + header_len, sdu_len);
        }

        // Duplicate detection (reordering)
        if (count.sn < next_rx_sn_) {
            // Duplicate -- discard
            return;
        }

        // Deliver to upper layer
        deliverToUpper(sdu_buffer, sdu_len);

        // Update RX state
        next_rx_sn_ = count.sn + 1;
        if (next_rx_sn_ >= config_.max_sn) {
            next_rx_sn_ = 0;
            rx_hfn_++;
        }
    }

    // Generate PDCP STATUS report (for handover)
    void generateStatusReport(std::vector<uint8_t>& report) {
        // Report: bitmap of missing PDUs since last ACK
        report.push_back(0xC0 | ((next_rx_sn_ >> 8) & 0x0F));
        report.push_back(next_rx_sn_ & 0xFF);
    }

private:
    uint8_t     drb_id_;
    PdcpConfig  config_{};
    uint32_t    next_tx_sn_{0};
    uint32_t    next_rx_sn_{0};
    uint32_t    tx_hfn_{0};
    uint32_t    rx_hfn_{0};

    // Store transmitted PDUs for handover data forwarding
    std::unordered_map<uint32_t, std::vector<uint8_t>> tx_pdus_;

    uint32_t addHeader(uint8_t* buf, const PdcpCount& count);
    uint32_t parseHeader(const uint8_t* pdu, size_t len, PdcpCount& count);
    void cipher(uint8_t* data, size_t len, const PdcpCount& count);
    void decipher(uint8_t* data, size_t len, const PdcpCount& count);
    uint32_t calculateMacI(const uint8_t* pdu, size_t len,
                          const PdcpCount& count);
    bool verifyIntegrity(const uint8_t* pdu, size_t len,
                         const PdcpCount& count);
    uint32_t appendMacI(uint8_t* buf, uint32_t mac_i);
    size_t compressHeader(uint8_t* out, const std::vector<uint8_t>& in);
    size_t decompressHeader(uint8_t* out, const uint8_t* in, size_t len);
    void storePdu(uint32_t sn, const uint8_t* pdu, size_t len);
    void sendToRlc(const uint8_t* pdu, uint32_t len);
    void deliverToUpper(const uint8_t* sdu, uint32_t len);
};

PDCP Count Value and Ciphering

Security Context: PDCP ciphering uses the COUNT value (HFN + SN) as input to the ciphering algorithm (AES, SNOW 3G, or ZUC). HFN increments when SN wraps around. For 12-bit SN: HFN increments every 4096 PDUs. For 18-bit SN: every 262144 PDUs. This is critical for handover -- the target gNB must receive the correct COUNT value.
