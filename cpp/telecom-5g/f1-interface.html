F1 Interface

The F1 interface connects the Central Unit (CU) and the Distributed Unit (DU) in the 5G NG-RAN. It consists of F1-C (control plane, F1AP protocol) and F1-U (user plane, GTP-U). F1 is defined in 3GPP TS 38.470 and is the primary interface for CU-DU coordination in the Option 2 split.

F1 Interface Overview

  +-----------+        +-----------+
  |    CU     |        |    DU     |
  |           |        |           |
  |  CU-CP    |-- F1-C ->|  RRC mgmt  |   F1AP over SCTP
  |  (RRC)    |<- F1-C --|  Cell mgmt |   Port: 38472
  |           |        |           |
  |  CU-UP    |-- F1-U ->|  MAC/RLC   |   GTP-U over UDP
  |  (PDCP)   |<- F1-U --|  PHY       |   Port: 2152
  |           |        |           |
  +-----------+        +-----------+

  Transport: IP-based (SCTP for F1-C, UDP for F1-U)

C++ F1AP Message Handler

#include <cstdint>
#include <string>
#include <unordered_map>
#include <functional>
#include <vector>

// F1AP Procedure Codes (3GPP TS 38.473)
enum class F1apProcedure : uint8_t {
    F1_SETUP               = 0,
    UE_CONTEXT_SETUP       = 1,
    UE_CONTEXT_RELEASE     = 2,
    UE_CONTEXT_MODIFY      = 3,
    BEARER_CONTEXT_SETUP   = 4,
    BEARER_CONTEXT_MODIFY  = 5,
    BEARER_CONTEXT_RELEASE = 6,
    CELL_CONFIG_UPDATE      = 7,
    F1_REMOVAL            = 8,
    RESET                 = 9
};

// F1AP message direction
enum class F1apDirection : uint8_t {
    INITIATING,   // Request
    SUCCESSFUL,   // Response (success)
    UNSUCCESSFUL  // Response (failure)
};

// F1AP UE Context Setup Request (simplified)
struct F1apUeContextSetupRequest {
    uint32_t gnb_cu_ue_f1ap_id;
    uint32_t gnb_du_ue_f1ap_id;
    uint8_t  rrc_container[4096];
    uint32_t rrc_container_length;

    // Cell information
    uint32_t serv_cell_id;          // Serving Cell ID (NR Cell ID)
    uint32_t pci;                   // Physical Cell ID

    // SRB and DRB to setup
    std::vector<uint8_t> srb_ids;   // SRB IDs to add/modify
    std::vector<uint8_t> drb_ids;   // DRB IDs to add/modify

    // Candidate PCI info (for CA)
    std::vector<uint32_t> candidate_pcis;
};

// F1AP UE Context Setup Response (simplified)
struct F1apUeContextSetupResponse {
    uint32_t gnb_cu_ue_f1ap_id;
    uint32_t gnb_du_ue_f1ap_id;
    bool     success;

    // Transport layer info for each DRB
    struct DrbInfo {
        uint8_t  drb_id;
        uint32_t gtp_teid;      // GTP TEID for F1-U
        std::string transport_addr;  // DU IP for F1-U
    };
    std::vector<DrbInfo> drbs_info;
};

// F1AP message dispatcher using message handler pattern
class F1apMessageDispatcher {
public:
    using Handler = std::function<void(const uint8_t* data, size_t len)>;

    void registerHandler(F1apProcedure proc, F1apDirection dir,
                         Handler handler) {
        handlers_[{static_cast<uint8_t>(proc),
                    static_cast<uint8_t>(dir)}] = std::move(handler);
    }

    bool dispatch(F1apProcedure proc, F1apDirection dir,
                  const uint8_t* data, size_t len) {
        auto key = std::make_pair(static_cast<uint8_t>(proc),
                                   static_cast<uint8_t>(dir));
        auto it = handlers_.find(key);
        if (it != handlers_.end()) {
            it->second(data, len);
            return true;
        }
        return false;
    }

private:
    // Key: (procedure_code, direction)
    std::unordered_map<std::pair<uint8_t, uint8_t>, Handler> handlers_;
};

// F1 interface endpoint (CU side)
class F1CuEndpoint {
public:
    F1CuEndpoint(const std::string& du_addr, uint16_t f1c_port)
        : du_addr_(du_addr), f1c_port_(f1c_port) {}

    bool connect() {
        // Establish SCTP association to DU F1-C endpoint
        // Send F1 SETUP REQUEST
        return sendF1SetupRequest();
    }

    void setupUe(uint32_t cu_ue_id, uint32_t du_ue_id,
                 const uint8_t* rrc_container, size_t rrc_len) {
        F1apUeContextSetupRequest req{};
        req.gnb_cu_ue_f1ap_id = cu_ue_id;
        req.gnb_du_ue_f1ap_id = du_ue_id;
        memcpy(req.rrc_container, rrc_container, rrc_len);
        req.rrc_container_length = rrc_len;

        // Encode F1AP message (ASN.1 PER)
        // Send over SCTP
        uint8_t buffer[8192];
        size_t encoded_len = encodeF1ap(req, buffer, sizeof(buffer));
        sendOverSctp(buffer, encoded_len);
    }

    void registerHandlers(F1apMessageDispatcher& disp) {
        // Register handlers for responses from DU
        disp.registerHandler(F1apProcedure::UE_CONTEXT_SETUP,
                            F1apDirection::SUCCESSFUL,
            [this](const uint8_t* data, size_t len) {
                handleUeContextSetupResponse(data, len);
            });
    }

private:
    std::string du_addr_;
    uint16_t    f1c_port_;
    int         sctp_fd_{-1};

    bool sendF1SetupRequest();
    size_t encodeF1ap(const F1apUeContextSetupRequest& req,
                      uint8_t* buf, size_t max_len);
    void handleUeContextSetupResponse(const uint8_t* data, size_t len);
    void sendOverSctp(const uint8_t* data, size_t len);
};

F1-U User Plane (GTP-U)

// F1-U uses GTP-U over UDP
#pragma pack(push, 1)
struct GtpUHeader {
    uint8_t  flags;          // Version (3 bits), PT (1), E (1), S (1), PN (1)
    uint8_t  message_type;   // 0xFF = GTP-U data
    uint16_t length;         // Payload length
    uint32_t teid;           // Tunnel Endpoint Identifier
};
#pragma pack(pop)

class F1UEndpoint {
public:
    // Setup F1-U tunnel with TEID learned from F1-C setup
    void addTunnel(uint32_t teid, const std::string& du_addr,
                   uint16_t port = 2152) {
        tunnels_[teid] = {du_addr, port};
    }

    // Send PDCP PDU to DU via F1-U
    int sendDownlink(uint32_t teid, const uint8_t* pdu, size_t len) {
        uint8_t buf[4096];
        auto& hdr = *reinterpret_cast<GtpUHeader*>(buf);
        hdr.flags = 0x30;        // Version 1, no extensions
        hdr.message_type = 0xFF; // GTP-U
        hdr.length = static_cast<uint16_t>(len);
        hdr.teid = teid;

        memcpy(buf + sizeof(GtpUHeader), pdu, len);
        return sendUdp(buf, sizeof(GtpUHeader) + len, teid);
    }

private:
    struct Tunnel {
        std::string addr;
        uint16_t    port;
    };
    std::unordered_map<uint32_t, Tunnel> tunnels_;
    int sendUdp(const uint8_t* data, size_t len, uint32_t teid);
};

Note: F1AP messages are encoded using ASN.1 Packed Encoding Rules (PER). See ASN.1 Encoding for the encoding/decoding details used to serialize/deserialize F1AP messages.
