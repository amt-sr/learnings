E1 Interface

The E1 interface connects the CU-CP (Central Unit Control Plane) and CU-UP (Central Unit User Plane) within the 5G gNB. Defined in 3GPP TS 38.460, E1 enables the separation of control and user plane processing in the CU, allowing independent scaling and deployment. The protocol is E1AP, carried over SCTP.

E1 Interface Architecture

                  +------------------+
                  |     CU-CP        |
                  |  (RRC / NGAP)    |
                  |  F1-C Management |
                  +--------+---------+
                           |
                    E1-C (E1AP)
                    SCTP Port: 38462
                           |
                  +--------+---------+
                  |     CU-UP        |
                  |  (SDAP / PDCP)   |
                  |  F1-U Management |
                  +--------+---------+
                           |
                    F1-U (GTP-U)
                           |
                  +--------+---------+
                  |       DU          |
                  |  (RLC / MAC / PHY)|
                  +------------------+

C++ E1AP Bearer Context Management

#include <memory>
#include <vector>
#include <unordered_map>
#include <cstdint>
#include <string>

// E1 Bearer Type
enum class E1BearerType : uint8_t {
    DRB,             // Data Radio Bearer
    MRB,             // Multicast Radio Bearer (MBS)
    SRB_OVER_E1      // Signaling over E1 (rare)
};

// PDCP Configuration for a DRB
struct PdcpConfig {
    uint8_t  drb_id;
    uint8_t  pdcp_sn_size;      // 12 or 18 bits
    uint8_t  rlcmode;           // AM, UM, TM
    uint16_t t_reordering;      // Reordering timer (ms)
    bool     header_compression; // ROHC enabled
    bool     ciphering;
    bool     integrity;
    uint8_t  ciphering_algorithm;  // AES, SNOW, ZUC
    uint8_t  integrity_algorithm;
};

// E1 Bearer Context Setup Request
struct E1BearerContextSetupRequest {
    uint32_t gnb_cu_cp_ue_e1ap_id;
    uint32_t gnb_cu_up_ue_e1ap_id;

    // Bearer to setup
    E1BearerType bearer_type;
    PdcpConfig   pdcp_config;

    // SDAP configuration
    uint8_t  qfi;             // QoS Flow Identifier
    uint8_t  qos_mapping_rule; // Reflective/non-reflective

    // NG-U tunnel info (from CU-CP, learned from AMF)
    uint32_t ngu_teid;        // UPF TEID
    std::string upf_addr;     // UPF transport address

    // F1-U tunnel (CU-UP assigns for F1-U to DU)
    uint32_t f1u_teid;        // CU-UP side TEID for F1-U
    std::string cu_up_addr;   // CU-UP transport address
};

// DRB context managed by CU-UP
class DrbContext {
public:
    explicit DrbContext(uint8_t drb_id) : drb_id_(drb_id) {}

    uint8_t getDrbId() const { return drb_id_; }
    uint32_t getTeidUl() const { return teid_ul_; }
    uint32_t getTeidDl() const { return teid_dl_; }
    bool isActive() const { return active_; }

    void activate(const PdcpConfig& cfg,
                  uint32_t ngu_teid, const std::string& upf_addr) {
        pdcp_config_ = cfg;
        ngu_teid_ = ngu_teid;
        upf_addr_ = upf_addr;
        active_ = true;
    }

    void updatePdcp(const PdcpConfig& new_cfg) {
        pdcp_config_ = new_cfg;
    }

    void deactivate() { active_ = false; }

private:
    uint8_t    drb_id_;
    bool       active_{false};
    PdcpConfig pdcp_config_;
    uint32_t   teid_ul_{0};
    uint32_t   teid_dl_{0};
    uint32_t   ngu_teid_{0};
    std::string upf_addr_;
};

// CU-UP E1 interface handler
class CuUpE1Handler {
public:
    // Handle E1 Bearer Context Setup Request from CU-CP
    void onBearerContextSetup(const E1BearerContextSetupRequest& req) {
        auto& drb = getOrCreateDrb(req.pdcp_config.drb_id);
        drb.activate(req.pdcp_config, req.ngu_teid, req.upf_addr);

        // Setup PDCP entity
        setupPdcpEntity(drb);

        // Create F1-U tunnel to DU
        uint32_t f1u_teid = allocateTeid();
        drb.setTeidDl(f1u_teid);

        // Send E1 Bearer Context Setup Response
        sendBearerContextResponse(req, f1u_teid);
    }

    void onBearerContextModify(uint8_t drb_id,
                                const PdcpConfig& new_config) {
        auto it = drb_contexts_.find(drb_id);
        if (it != drb_contexts_.end()) {
            it->second.updatePdcp(new_config);
            reconfigurePdcp(it->second);
        }
    }

    void onBearerContextRelease(uint8_t drb_id) {
        auto it = drb_contexts_.find(drb_id);
        if (it != drb_contexts_.end()) {
            releasePdcpEntity(it->second);
            it->second.deactivate();
        }
    }

private:
    std::unordered_map<uint8_t, DrbContext> drb_contexts_;

    DrbContext& getOrCreateDrb(uint8_t drb_id) {
        auto it = drb_contexts_.find(drb_id);
        if (it != drb_contexts_.end()) {
            return it->second;
        }
        auto [iter, _] = drb_contexts_.emplace(
            drb_id, DrbContext(drb_id));
        return iter->second;
    }

    uint32_t allocateTeid() {
        static uint32_t next_teid = 1;
        return next_teid++;
    }

    void setupPdcpEntity(DrbContext& drb);
    void reconfigurePdcp(DrbContext& drb);
    void releasePdcpEntity(DrbContext& drb);
    void sendBearerContextResponse(
        const E1BearerContextSetupRequest& req,
        uint32_t f1u_teid);
};

E1 Data Forwarding for Handover

// Data forwarding configuration for seamless handover
struct E1DataForwardingConfig {
    uint32_t source_gnb_cu_up_ue_e1ap_id;
    uint32_t target_gnb_cu_up_ue_e1ap_id;

    struct ForwardingBearer {
        uint8_t  drb_id;
        uint32_t target_teid;       // TEID at target CU-UP
        std::string target_addr;    // Target CU-UP address
        uint32_t count_dl;          // PDCP SN count (DL)
        uint32_t count_ul;          // PDCP SN count (UL)
    };
    std::vector<ForwardingBearer> bearers;
};

Scalability: A single CU-CP can connect to multiple CU-UPs for scaling user plane throughput. Each CU-UP can handle 10-20 Gbps. The E1 interface must support hundreds of concurrent bearer contexts. Use memory pools for DRB context allocation.
