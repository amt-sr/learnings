Bit-Level Operations

Bitwise operations are fundamental in telecom protocol parsing, DSP, and hardware interfacing. Every protocol header in the radio stack -- MAC, RLC, PDCP, RRC -- requires extracting and inserting fields at specific bit positions within bytes.

Bitwise Operators

OperatorSymbolUse Case
AND&Clear bits, extract fields, mask
OR|Set bits, combine fields
XOR^Toggle bits, CRC computation
NOT~Complement, mask generation
Left Shift<<Multiply by 2^n, insert fields
Right Shift>>Divide by 2^n, extract fields

Common Bit Manipulation Idioms

// Set bit N (0-indexed from LSB)
uint32_t setBit(uint32_t val, int n) {
    return val | (1U << n);
}

// Clear bit N
uint32_t clearBit(uint32_t val, int n) {
    return val & ~(1U << n);
}

// Toggle bit N
uint32_t toggleBit(uint32_t val, int n) {
    return val ^ (1U << n);
}

// Check if bit N is set
bool checkBit(uint32_t val, int n) {
    return (val & (1U << n)) != 0;
}

// Extract bits [high:low] (inclusive)
uint32_t extractBits(uint32_t val, int high, int low) {
    uint32_t mask = (1U << (high - low + 1)) - 1;
    return (val >> low) & mask;
}

// Insert value into bits [high:low]
uint32_t insertBits(uint32_t val, uint32_t field, int high, int low) {
    uint32_t mask = (1U << (high - low + 1)) - 1;
    val &= ~(mask << low);       // Clear target bits
    val |= ((field & mask) << low); // Insert field
    return val;
}

Bit Fields in Structs

// MAC subheader (LCID + F + L fields)
struct MacSubheader {
    uint8_t R     : 1;  // Reserved
    uint8_t F     : 2;  // Format (00=1 byte, 01=2 bytes, etc.)
    uint8_t LCID  : 5;  // Logical Channel ID (max 32)
} __attribute__((packed));

// DCI format 1_0 (simplified, bit-packed)
struct DciFormat10 {
    uint16_t format_id   : 2;  // DCI format identifier
    uint16_t frequency    : 1;  // Frequency domain flag
    uint16_t time         : 1;  // Time domain flag
    uint16_t vrb_flag     : 1;  // VRB to PRB mapping
    uint16_t mcs          : 5;  // Modulation and coding scheme
    uint16_t ndi          : 1;  // New data indicator
    uint16_t rv           : 2;  // Redundancy version
    uint16_t harq_pid     : 4;  // HARQ process ID
} __attribute__((packed));

C++ Protocol Field Extraction

#include <cstdint>
#include <cstring>
#include <iostream>

// Extract fields from a 32-bit MAC Control Element word
class MacCeParser {
public:
    struct BsrFields {
        uint8_t  lcg;           // Logical Channel Group (0-3)
        uint8_t  buffer_size;   // Buffer size index (0-63)
        uint8_t  short_bsr;     // Short BSR flag
    };

    // Parse MAC CE from a 32-bit word:
    // Bits [31:30] = LCG (2 bits)
    // Bits [29:24] = Buffer Size (6 bits)
    // Bit  [23]    = Short BSR flag
    // Bits [22:0]  = Reserved/padding
    static BsrFields parseBsr(uint32_t word) {
        BsrFields result{};
        result.lcg = static_cast<uint8_t>((word >> 30) & 0x3);
        result.buffer_size = static_cast<uint8_t>(
            (word >> 24) & 0x3F);
        result.short_bsr = static_cast<uint8_t>(
            (word >> 23) & 0x1);
        return result;
    }

    // Build MAC CE word from fields
    static uint32_t buildBsr(const BsrFields& bsr) {
        uint32_t word = 0;
        word |= (static_cast<uint32_t>(bsr.lcg) & 0x3) << 30;
        word |= (static_cast<uint32_t>(bsr.buffer_size) & 0x3F) << 24;
        word |= (static_cast<uint32_t>(bsr.short_bsr) & 0x1) << 23;
        return word;
    }
};

// Parse RLC header from byte stream
class RlcHeaderParser {
public:
    struct RlcAmHeader {
        uint16_t sn;           // Sequence Number (10 or 18 bits)
        uint8_t  so;            // Segment Offset (15 bits)
        bool     dc;            // Data/Control
        bool     p;             // Poll bit
        uint8_t  fi;            // Framing Info (2 bits)
    };

    // Parse 16-bit RLC AM data PDU header
    // Byte 0: D/C(1) | RF(1) | P(1) | FI(2) | E(1) | SN(2)
    // Byte 1: SN(8)
    static RlcAmHeader parseAmHeader(const uint8_t* data) {
        RlcAmHeader hdr{};
        hdr.dc  = (data[0] & 0x80) != 0;
        hdr.p   = (data[0] & 0x20) != 0;
        hdr.fi  = (data[0] >> 3) & 0x03;
        hdr.sn  = ((data[0] & 0x03) << 8) | data[1];
        return hdr;
    }
};

Packing/Unpacking Protocol Fields

Tip: When parsing telecom protocols, always use manual bit shifting rather than C++ bit fields. Bit fields have implementation-defined ordering (MSB vs LSB) and may not match the protocol specification. Explicit >> and & operations are portable and verifiable against the 3GPP spec.
