X2 / NG-X Interface

The X2 interface connects LTE eNodeBs to each other, while the Xn interface (also called NG-X) connects 5G gNBs. Both support handover, load management, and inter-cell coordination. In NSA (EN-DC) mode, the X2 interface also connects the 5G gNB (SgNB) to the LTE master eNodeB (MeNB).

C++ XnAP Handover State Machine

#include <string>
#include <unordered_map>
#include <functional>
#include <memory>
#include <cstdint>

// Xn Handover states
enum class XnHoState : uint8_t {
    IDLE,                      // No handover in progress
    HO_REQUEST_SENT,           // Xn HO REQUEST sent to target gNB
    HO_REQUEST_ACK_RECEIVED,  // Xn HO REQUEST ACK received
    HO_EXECUTED,               // UE executing handover
    PATH_SWITCH_PENDING,       // Waiting for path switch
    COMPLETED,                 // Handover complete
    FAILED                     // Handover failed
};

// Xn Handover Procedure messages
enum class XnHoMessage : uint8_t {
    HANDOVER_REQUEST,
    HANDOVER_REQUEST_ACK,
    HANDOVER_CMD,
    SN_STATUS_TRANSFER,
    UE_CONTEXT_RELEASE,
    HANDOVER_PREPARATION_FAILURE,
    HANDOVER_CANCEL
};

// Xn Handover Request (simplified)
struct XnHandoverRequest {
    // Source gNB info
    uint32_t source_gnb_id;
    uint32_t source_cell_id;
    uint32_t source_pci;

    // UE context
    uint32_t ue_id;
    uint32_t old_amf_ue_ngap_id;

    // Target info
    uint32_t target_cell_id;

    // Bearers to handover
    struct BearerContext {
        uint8_t  drb_id;
        uint32_t teid_dl;        // GTP-U TEID (source side)
        uint32_t teid_ul;
        uint32_t pdu_sn_dl;      // PDCP TX NEXT
        uint32_t pdu_sn_ul;      // PDCP RX NEXT
        uint32_t pdcp_count;     // PDCP COUNT value
    };
    std::vector<BearerContext> bearers;

    // UE capability and measurement info
    uint8_t  ue_cap_info[256];
};

// Xn Handover state machine
class XnHandoverMachine {
public:
    void initiate(uint32_t ue_id, uint32_t target_cell_id) {
        ue_id_ = ue_id;
        target_cell_ = target_cell_id;
        transition(XnHoState::HO_REQUEST_SENT);
        sendXnHoRequest();
    }

    void onHoRequestAck(const uint8_t* msg, size_t len) {
        if (state_ != XnHoState::HO_REQUEST_SENT) {
            // Unexpected message
            return;
        }
        // Parse HO REQUEST ACK -- contains target config
        parseHoRequestAck(msg, len);
        transition(XnHoState::HO_REQUEST_ACK_RECEIVED);
        sendHoCommandToUe();
    }

    void onSnStatusTransfer(const uint8_t* msg, size_t len) {
        // Forward PDCP SN status from source to target
        // Used for PDCP reordering after handover
    }

    void onHoComplete() {
        transition(XnHoState::PATH_SWITCH_PENDING);
        initiatePathSwitch();
    }

    void onPathSwitchAck() {
        transition(XnHoState::COMPLETED);
        releaseSourceResources();
    }

    XnHoState getState() const { return state_; }

private:
    uint32_t    ue_id_;
    uint32_t    target_cell_;
    XnHoState   state_ = XnHoState::IDLE;

    void transition(XnHoState new_state) {
        XnHoState old = state_;
        state_ = new_state;
        onStateChange(old, new_state);
    }

    void onStateChange(XnHoState old, XnHoState new_state) {
        // Logging, metrics, cleanup
    }

    void sendXnHoRequest();
    void parseHoRequestAck(const uint8_t* msg, size_t len);
    void sendHoCommandToUe();
    void initiatePathSwitch();
    void releaseSourceResources();
};

Xn Dual Connectivity (NR-DC / EN-DC)

// Dual connectivity configuration
enum class DualConnectivityMode : uint8_t {
    NONE,
    EN_DC,         // LTE + NR (NSA)
    NE_DC,         // NR + LTE
    NR_DC,         // NR + NR (CA across gNBs)
    NR_CA          // NR Carrier Aggregation (same gNB)
};

struct DualConnectivityConfig {
    DualConnectivityMode mode;
    uint32_t mg_nodeb_id;      // Master NodeB ID
    uint32_t sg_nodeb_id;      // Secondary NodeB ID
    uint32_t m_cell_id;        // Master cell
    uint32_t s_cell_id;        // Secondary cell

    // Split bearer config (SCG bearer)
    struct SplitBearer {
        uint8_t  drb_id;
        uint32_t m_teid;       // MCG leg TEID
        uint32_t s_teid;       // SCG leg TEID
        uint32_t ul_split_ratio; // % to MCG vs SCG
    };
    std::vector<SplitBearer> split_bearers;
};
