5G RAN Overview

The 5G Radio Access Network (5G RAN), also known as NG-RAN (Next-Generation RAN), is the radio network architecture defined by 3GPP for 5G New Radio (NR). It replaces the LTE E-UTRAN and introduces a fundamentally restructured architecture with a flexible functional split between central and distributed units.

Key Components

  • gNB (Next-Generation NodeB): The 5G base station, responsible for radio resource management, connection mobility control, routing of user data, and measurement configuration.
  • ng-eNB: LTE eNodeB upgraded with NG interface to connect to the 5G Core Network.
  • NG Interface: Connects NG-RAN to the 5G Core -- NG-C (control plane to AMF) and NG-U (user plane to UPF).
  • Xn Interface: Connects gNBs to each other for handover and data forwarding.
  • NR Uu: The air interface between UE and gNB.

SA vs NSA C++ Tip: The control plane routing logic differs significantly. In SA, gNB connects directly to AMF via NG-C. In NSA (EN-DC), the gNB acts as a secondary node -- see X2/NG-X Interface for dual connectivity handling.

C++ Data Structures for gNB

#include <string>
#include <vector>
#include <memory>
#include <cstdint>

// gNB deployment mode enumeration
enum class DeploymentMode : uint8_t {
    STANDALONE,      // SA - connected to 5GC
    NON_STANDALONE,  // NSA - connected via LTE eNB to EPC
    EN_DC,           // E-UTRA NR Dual Connectivity
    NE_DC,           // NR E-UTRA Dual Connectivity
    NR_DC            // NR-NR Dual Connectivity
};

// gNB identifier types
struct GnbId {
    uint32_t plmn_id;       // Public Land Mobile Network ID
    uint32_t gnb_id;        // gNB unique identifier (up to 32 bits)
    uint8_t  gnb_id_length; // Number of bits in gnb_id (22-32)
};

// NR Cell configuration
struct NrCellConfig {
    uint32_t           cell_id;
    uint32_t           pci;           // Physical Cell ID (0-1007)
    uint32_t           arfcn;         // NR-ARFCN
    uint32_t           bandwidth;     // Bandwidth in PRBs
    uint32_t           scs_khz;       // Subcarrier spacing: 15, 30, 60, 120, 240
    uint8_t            duplex_mode;   // 0: TDD, 1: FDD
    uint32_t           ssb_period;    // SSB periodicity in ms
};

// gNB base class representing a 5G base station
class GNodeB {
public:
    GNodeB(const GnbId& id, DeploymentMode mode)
        : id_(id), mode_(mode), is_active_(false) {}

    virtual ~GNodeB() = default;

    // Core interface management
    virtual bool connectToCore() = 0;
    virtual bool establishNgInterface() = 0;
    virtual bool handleNgSetup() = 0;

    // Cell management
    void addCell(const NrCellConfig& cell) {
        cells_.push_back(cell);
    }

    [[nodiscard]] size_t cellCount() const { return cells_.size(); }
    [[nodiscard]] DeploymentMode getMode() const { return mode_; }

protected:
    GnbId              id_;
    DeploymentMode     mode_;
    bool               is_active_;
    std::vector<NrCellConfig> cells_;
};

// Standalone gNB implementation
class SaGNodeB : public GNodeB {
public:
    explicit SaGNodeB(const GnbId& id)
        : GNodeB(id, DeploymentMode::STANDALONE) {}

    bool connectToCore() override {
        // SA: Connect directly to 5GC AMF via NG-C
        // Establish SCTP association on NG-C (N2 interface)
        return establishSctpConnection(ng_c_amf_addr_);
    }

    bool establishNgInterface() override {
        // Setup NG-C (control) and NG-U (user) interfaces
        return setupNgControl() && setupNgUserPlane();
    }

    bool handleNgSetup() override {
        // Send NG SETUP REQUEST to AMF
        // Contains: gNB ID, PLMN, supported TA list, etc.
        return sendNgSetupRequest();
    }

private:
    std::string ng_c_amf_addr_;  // AMF address for control plane
    std::string ng_u_upf_addr_;   // UPF address for user plane

    bool establishSctpConnection(const std::string& addr);
    bool setupNgControl();
    bool setupNgUserPlane();
    bool sendNgSetupRequest();
};

// NSA gNB (as secondary node in EN-DC)
class NsaGNodeB : public GNodeB {
public:
    explicit NsaGNodeB(const GnbId& id)
        : GNodeB(id, DeploymentMode::EN_DC), master_eNB_(nullptr) {}

    void setMasterENodeB(void* eNB) { master_eNB_ = eNB; }

    bool connectToCore() override {
        // NSA: X2 connection to master eNodeB (not directly to core)
        return establishX2Connection();
    }

    bool establishNgInterface() override {
        // NSA uses X2-U to master eNB, which routes to S-GW
        return true;  // X2 established during SgNB addition
    }

    bool handleNgSetup() override {
        // NSA: SgNB Addition procedure instead of NG Setup
        return performSgNBAddition();
    }

private:
    void* master_eNB_;  // Pointer to master LTE eNodeB

    bool establishX2Connection();
    bool performSgNBAddition();
};

C++ NR Bandwidth Configuration

#include <map>
#include <stdexcept>

struct NrBandInfo {
    uint32_t band_number;
    uint32_t lower_freq_mhz;
    uint32_t upper_freq_mhz;
    bool     is_mmwave;       // FR2 = millimeter wave
};

// Bandwidth configuration based on SCS and max PRBs
class NrBandwidthConfig {
public:
    // 3GPP TS 38.104 Table 5.3.2-1
    static constexpr int MAX_PRBS[] = {
        // SCS: 15kHz, 30kHz, 60kHz, 120kHz, 240kHz
        264,    // 15 kHz
        273,    // 30 kHz
        270,    // 60 kHz
        243,    // 120 kHz
        99      // 240 kHz
    };

    static uint32_t getBandwidthHz(uint8_t scs_khz, uint32_t n_prbs) {
        // BW = N_PRBs * 12 * SCS
        // Each PRB has 12 subcarriers
        return n_prbs * 12 * scs_khz * 1000;
    }

    static uint32_t getMaxPrbs(uint8_t scs_khz) {
        int idx = scsToIndex(scs_khz);
        return MAX_PRBS[idx];
    }

    static uint32_t getSlotDurationUs(uint8_t scs_khz) {
        // Slot = 14 OFDM symbols * (1/SCS)
        // Normal cyclic prefix: slot = 1ms for 15kHz
        return 1000000 / (scs_khz / 15);
    }

    static uint32_t getSymbolsPerSlot() {
        return 14;  // Normal CP
    }

private:
    static int scsToIndex(uint8_t scs_khz) {
        switch (scs_khz) {
            case 15:  return 0;
            case 30:  return 1;
            case 60:  return 2;
            case 120: return 3;
            case 240: return 4;
            default:  throw std::invalid_argument("Invalid SCS");
        }
    }
};
