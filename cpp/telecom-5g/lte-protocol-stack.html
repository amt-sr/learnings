LTE Protocol Stack

The LTE protocol stack defines how user and control data traverse the radio interface (Uu). It consists of two planes: the User Plane (data) and the Control Plane (signaling). Each layer has specific responsibilities in processing, from physical modulation to application-level protocol handling.

Control Plane Stack

  Control Plane (Signaling):
  +-----------+
  | RRC       |  Radio Resource Control
  +-----------+
  | PDCP      |  Ciphering, integrity protection
  +-----------+
  | RLC       |  Segmentation, ARQ (AM mode)
  +-----------+
  | MAC       |  Logical channels, HARQ, scheduling
  +-----------+
  | PHY       |  OFDMA, channel coding, modulation
  +-----------+
    (carried over SRB0/SRB1/SRB2)
    RRC uses RLC AM; Initial RRC (SRB0) uses RLC TM

User Plane Stack

  User Plane (Data):
  +-----------+
  | IP / SDP  |  Application / Transport (TCP/UDP)
  +-----------+
  | PDCP      |  ROHC header compression, ciphering
  +-----------+
  | RLC       |  Segmentation, ARQ
  +-----------+
  | MAC       |  Multiplexing, HARQ retransmission
  +-----------+
  | PHY       |  OFDMA transmission
  +-----------+
    (carried over DRB)

LTE vs 5G Protocol Stack

  LTE:
  IP -> PDCP -> RLC -> MAC -> PHY
  RRC -> PDCP -> RLC -> MAC -> PHY

  5G NR (SDAP added):
  IP -> SDAP -> PDCP -> RLC -> MAC -> PHY
  RRC -> PDCP -> RLC -> MAC -> PHY

  SDAP is new in 5G. It provides QoS flow -> DRB mapping.
  In LTE, QoS mapping happens at the eNodeB without a
  dedicated protocol layer.

SDAP addition: 5G introduces SDAP between IP and PDCP. This is the only new protocol layer added to the radio stack. It enables fine-grained QoS flow management and reflective QoS mapping, which was not possible in LTE.

C++ Layer Struct Hierarchy

#include <memory>
#include <functional>
#include <cstdint>
#include <vector>

// Generic layer interface
class IProtocolLayer {
public:
    virtual ~IProtocolLayer() = default;

    // Upper layer sends data down
    virtual void sendDataDown(uint8_t channel_id,
                               const uint8_t* data, size_t len) = 0;

    // Lower layer sends data up
    virtual void receiveDataUp(uint8_t channel_id,
                                const uint8_t* data, size_t len) = 0;

    // Configuration callback (from RRC)
    virtual void configure(const void* cfg) = 0;
};

// PHY layer (lowest)
class PhyLayer : public IProtocolLayer {
public:
    void sendDataDown(uint8_t, const uint8_t* data, size_t len) override {
        // Modulate, encode, transmit over air
        transmitOverAir(data, len);
    }

    void receiveDataUp(uint8_t, const uint8_t* data, size_t len) override {
        // Demodulate, decode, pass to MAC
        if (upper_layer_) {
            upper_layer_->receiveDataUp(0, data, len);
        }
    }

    void setUpperLayer(IProtocolLayer* upper) { upper_layer_ = upper; }
private:
    IProtocolLayer* upper_layer_{nullptr};
    void transmitOverAir(const uint8_t*, size_t) {}
};

// MAC layer
class MacLayer : public IProtocolLayer {
public:
    void sendDataDown(uint8_t lcid, const uint8_t* data, size_t len) override {
        // Multiplex logical channels, add MAC header
        // Perform HARQ, schedule resources
        if (lower_layer_) {
            uint8_t pdu[4096];
            size_t pdu_len = addMacHeader(pdu, lcid, data, len);
            lower_layer_->sendDataDown(lcid, pdu, pdu_len);
        }
    }

    void receiveDataUp(uint8_t, const uint8_t* pdu, size_t len) override {
        // Parse MAC header, demultiplex to logical channels
        uint8_t lcid; const uint8_t* payload; size_t payload_len;
        parseMacHeader(pdu, len, lcid, payload, payload_len);
        if (upper_layer_) {
            upper_layer_->receiveDataUp(lcid, payload, payload_len);
        }
    }

    void setLowerLayer(IProtocolLayer* lower) { lower_layer_ = lower; }
    void setUpperLayer(IProtocolLayer* upper) { upper_layer_ = upper; }

private:
    IProtocolLayer* lower_layer_{nullptr};
    IProtocolLayer* upper_layer_{nullptr};
    size_t addMacHeader(uint8_t*, uint8_t, const uint8_t*, size_t);
    void parseMacHeader(const uint8_t*, size_t, uint8_t&,
                         const uint8_t*&, size_t&);
};

// RLC layer
class RlcLayer : public IProtocolLayer {
public:
    // Same pattern: sendDataDown -> add RLC header, segment
    // receiveDataUp -> reassemble, remove RLC header
    void setLowerLayer(IProtocolLayer* lower) { lower_layer_ = lower; }
    void setUpperLayer(IProtocolLayer* upper) { upper_layer_ = upper; }
    void sendDataDown(uint8_t ch, const uint8_t* data, size_t len) override;
    void receiveDataUp(uint8_t ch, const uint8_t* data, size_t len) override;
    void configure(const void* cfg) override {}
private:
    IProtocolLayer* lower_layer_{nullptr};
    IProtocolLayer* upper_layer_{nullptr};
};

// Build the stack (composition)
class ProtocolStack {
public:
    void init() {
        phy_.setUpperLayer(&mac_);
        mac_.setLowerLayer(&phy_);
        mac_.setUpperLayer(&rlc_);
        rlc_.setLowerLayer(&mac_);
    }

    PhyLayer& phy() { return phy_; }
    MacLayer& mac() { return mac_; }
    RlcLayer& rlc() { return rlc_; }

private:
    PhyLayer  phy_;
    MacLayer  mac_;
    RlcLayer  rlc_;
    // PDCP and SDAP would follow the same pattern above RLC
};
