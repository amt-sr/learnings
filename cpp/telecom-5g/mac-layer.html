MAC Layer

The MAC (Medium Access Control) layer in 5G NR handles scheduling, HARQ retransmission, logical channel multiplexing, and random access. It is the most critical layer for radio resource management and operates within strict timing constraints (sub-millisecond for TTI processing).

MAC PDU Format

  MAC PDU:
  +------+---+---+------+---+---+---+---+------+---+---+------+
  | SubH | SDU1 | SubH | SDU2 | SubH | CE | Pad |    |     |     |
  +------+-------+------+-------+------+-----+-----+----+-----+-----+
  |  R   | R/F  | L  |  MAC SDU (RLC PDU)                    |
  |  R   | R/F  | L  |  MAC CE  (BSR, PHR, C-RNTI...)       |
  |  R   | R/F  | L  |  Padding                               |

  R  = Reserved bit (1)
  F  = Format bit: 0=1-byte length, 1=2-byte length
  L  = Length of MAC SDU or MAC CE

C++ MAC Scheduler

#include <vector>
#include <queue>
#include <algorithm>
#include <memory>
#include <cstdint>
#include <cmath>

// UE context for scheduler
struct UeContext {
    uint32_t ue_id;
    uint16_t rnti;
    uint8_t  cqi;             // Channel Quality Indicator (0-15)
    uint32_t buffer_dl;      // DL buffer size (bytes)
    uint32_t buffer_ul;      // UL buffer status (bytes)
    uint8_t  priority;       // QoS priority
    uint32_t mcs_table_idx;  // MCS table index
    uint32_t avg_throughput; // Past throughput (for PF scheduling)
    uint32_t prb_usage;       // PRBs allocated last TTI
    bool     is_retx;         // HARQ retransmission pending
    uint8_t  harq_pid;        // HARQ process ID for retx
};

// Scheduling decision for one UE
struct SchedDecision {
    uint32_t ue_id;
    uint16_t rnti;
    uint16_t start_rb;       // Start PRB index
    uint16_t n_rbs;          // Number of PRBs allocated
    uint8_t  start_symbol;    // Start OFDM symbol
    uint8_t  n_symbols;       // Number of symbols
    uint8_t  mcs;             // MCS index
    uint8_t  harq_id;         // HARQ process ID
    uint8_t  ndi;             // New Data Indicator
    uint8_t  rv;              // Redundancy Version
    uint32_t tb_size;         // Transport block size
};

// Proportional Fair scheduler (most common commercial scheduler)
class PfScheduler {
public:
    void addUe(const UeContext& ue) {
        ues_[ue.ue_id] = ue;
    }

    void removeUe(uint32_t ue_id) {
        ues_.erase(ue_id);
    }

    // Schedule DL for one TTI
    void scheduleDl(uint32_t available_prbs, uint32_t tti_duration_us,
                    std::vector<SchedDecision>& decisions) {
        decisions.clear();
        uint32_t remaining_prbs = available_prbs;

        // Sort UEs by priority metric (proportional fair)
        std::vector<UeContext*> sorted_ues;
        for (auto& [id, ue] : ues_) {
            if (ue.buffer_dl > 0 || ue.is_retx) {
                sorted_ues.push_back(&ue);
            }
        }

        // Calculate PF metric: CQI-based rate / average throughput
        std::sort(sorted_ues.begin(), sorted_ues.end(),
            [](const UeContext* a, const UeContext* b) {
                double metric_a = estimateRate(a->cqi) /
                                  std::max(1.0, (double)a->avg_throughput);
                double metric_b = estimateRate(b->cqi) /
                                  std::max(1.0, (double)b->avg_throughput);
                return metric_a > metric_b;
            });

        for (auto* ue : sorted_ues) {
            if (remaining_prbs == 0) break;

            // Estimate PRBs needed
            uint16_t n_rbs = estimateRequiredPrbs(
                ue->buffer_dl, ue->cqi, remaining_prbs);

            uint8_t mcs = cqiToMcs(ue->cqi);
            uint32_t tb_size = getTbs(mcs, n_rbs);

            SchedDecision dec{};
            dec.ue_id = ue->ue_id;
            dec.rnti = ue->rnti;
            dec.start_rb = available_prbs - remaining_prbs;
            dec.n_rbs = n_rbs;
            dec.start_symbol = 0;
            dec.n_symbols = (ue->is_retx) ? 14 : 12;
            dec.mcs = mcs;
            dec.harq_id = selectHarqProcess(ue);
            dec.ndi = ue->is_retx ? 0 : 1;
            dec.rv = ue->is_retx ? getRv(ue->harq_pid) : 0;
            dec.tb_size = tb_size;

            decisions.push_back(dec);
            remaining_prbs -= n_rbs;
        }
    }

private:
    std::unordered_map<uint32_t, UeContext> ues_;

    // CQI to estimated rate (bits per PRB per symbol)
    static double estimateRate(uint8_t cqi) {
        // Approximate mapping
        // CQI 1-15 maps to spectral efficiency 0.15-5.55
        return 0.15 * pow(1.48, cqi);
    }

    static uint8_t cqiToMsc(uint8_t cqi) {
        // Simplified CQI-to-MCS mapping (Table 5.2.2.1-2)
        return std::min(cqi + 4, (uint8_t)28);
    }

    static uint16_t estimateRequiredPrbs(uint32_t buffer_bytes,
                                          uint8_t cqi,
                                          uint32_t max_prbs) {
        double rate = estimateRate(cqi) * 12;  // bits per PRB
        uint32_t prbs_needed = (buffer_bytes * 8) / rate + 1;
        return std::min(static_cast<uint16_t>(prbs_needed),
                       static_cast<uint16_t>(max_prbs));
    }

    static uint8_t cqiToMcs(uint8_t cqi) {
        return std::min(cqi + 4, (uint8_t)28);
    }

    static uint32_t getTbs(uint8_t mcs, uint16_t n_prbs) {
        // Simplified TBS calculation
        // Real implementation uses TS 38.214 Table 5.1.3.1-1
        return n_prbs * 12 * (mcs > 16 ? 6 : 4);
    }

    static uint8_t selectHarqProcess(const UeContext* ue) {
        static uint8_t round_robin = 0;
        return (round_robin++ % 8);
    }

    static uint8_t getRv(uint8_t harq_pid) {
        static const uint8_t rv_seq[] = {0, 2, 3, 1};
        static uint8_t rv_idx = 0;
        return rv_seq[rv_idx++ % 4];
    }
};

HARQ Process Management

// HARQ entity: manages 8 parallel HARQ processes
class HarqEntity {
public:
    static constexpr uint8_t NUM_PROCESSES = 8;

    enum class HarqState : uint8_t {
        IDLE,
        WAITING_ACK,
        RETRANSMITTING,
        NACK_RECEIVED
    };

    struct HarqProcess {
        HarqState state = HarqState::IDLE;
        uint8_t   rv = 0;            // Redundancy version
        uint8_t   retx_count = 0;   // Retransmission count
        uint32_t  tb_size = 0;
        uint16_t  n_rbs = 0;
        uint8_t   mcs = 0;
    };

    HarqProcess& getProcess(uint8_t pid) { return processes_[pid]; }

    // Find next available process for new transmission
    std::optional<uint8_t> findFreeProcess() const {
        for (uint8_t i = 0; i < NUM_PROCESSES; ++i) {
            if (processes_[i].state == HarqState::IDLE) return i;
        }
        return std::nullopt;
    }

    // Process ACK/NACK from UE
    void onAckNack(uint8_t pid, bool ack) {
        if (pid >= NUM_PROCESSES) return;
        if (ack) {
            processes_[pid].state = HarqState::IDLE;
            processes_[pid].retx_count = 0;
        } else {
            processes_[pid].retx_count++;
            if (processes_[pid].retx_count <= max_retx_) {
                processes_[pid].rv = getNextRv(processes_[pid].retx_count);
                processes_[pid].state = HarqState::RETRANSMITTING;
            } else {
                // Max retransmissions reached, discard
                processes_[pid].state = HarqState::IDLE;
            }
        }
    }

private:
    std::array<HarqProcess, NUM_PROCESSES> processes_{};
    static constexpr uint8_t max_retx_ = 3;

    static uint8_t getNextRv(uint8_t retx_count) {
        static const uint8_t rv_seq[] = {0, 2, 3, 1};
        return rv_seq[retx_count % 4];
    }
};
