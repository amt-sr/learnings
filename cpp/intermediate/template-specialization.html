Template Specialization

Template specialization allows you to provide alternative implementations for specific types or patterns of template arguments. There are two forms: full (explicit) specialization which provides a complete replacement for a specific type, and partial specialization which provides an alternative for a subset of types.

Full Specialization

Full specialization replaces the entire primary template for one specific set of template arguments.

#include <iostream>
#include <string>
#include <cstring>

// Primary template — generic for all types
template<typename T>
class Serializer {
public:
    static std::string serialize(const T& value) {
        return std::to_string(value);
    }
};

// Full specialization for std::string
template<>
class Serializer<std::string> {
public:
    static std::string serialize(const std::string& value) {
        return "\"" + value + "\"";  // Add quotes around strings
    }
};

// Full specialization for bool
template<>
class Serializer<bool> {
public:
    static std::string serialize(bool value) {
        return value ? "true" : "false";  // Textual true/false
    }
};

void full_specialization_demo() {
    std::cout << Serializer<int>::serialize(42) << "\n";          // "42"
    std::cout << Serializer<double>::serialize(3.14) << "\n";      // "3.140000"
    std::cout << Serializer<std::string>::serialize("hello") << "\n"; // "\"hello\""
    std::cout << Serializer<bool>::serialize(true) << "\n";       // "true"
}

Function Template Specialization

Note: Function template specialization has limitations. You cannot partially specialize function templates (only class templates). For functions, prefer overloading instead of specialization.

// Primary function template
template<typename T>
T get_default();

// Full specializations
template<>
int get_default<int>() { return 0; }

template<>
double get_default<double>() { return 0.0; }

template<>
std::string get_default<std::string>() { return ""; }

// Better approach: use overloading instead of specialization for functions
void print_value(const std::string& s) { std::cout << "string: " << s << "\n"; }
void print_value(int n) { std::cout << "int: " << n << "\n"; }
void print_value(double d) { std::cout << "double: " << d << "\n"; }

Partial Specialization

Partial specialization is only available for class templates. It allows you to specialize for a category of types rather than a single specific type.

#include <iostream>
#include <type_traits>

// Primary template
template<typename T>
class TypeInfo {
public:
    static std::string name() { return "unknown"; }
    static bool is_numeric() { return false; }
};

// Partial specialization for all pointer types
template<typename T>
class TypeInfo<T*> {
public:
    static std::string name() { return TypeInfo<T>::name() + "*"; }
    static bool is_numeric() { return false; }
};

// Partial specialization for all integral types (C++11)
template<typename T>
class TypeInfo<std::enable_if_t<std::is_integral_v<T>, T>> {
public:
    static std::string name() { return "integral"; }
    static bool is_numeric() { return true; }
};

// Partial specialization for all floating-point types
template<typename T>
class TypeInfo<std::enable_if_t<std::is_floating_point_v<T>, T>> {
public:
    static std::string name() { return "floating_point"; }
    static bool is_numeric() { return true; }
};

// Partial specialization for all array types
template<typename T, size_t N>
class TypeInfo<T[N]> {
public:
    static std::string name() {
        return TypeInfo<T>::name() + "[" + std::to_string(N) + "]";
    }
    static bool is_numeric() { return TypeInfo<T>::is_numeric(); }
};

void partial_specialization_demo() {
    std::cout << TypeInfo<int>::name() << "\n";           // "integral"
    std::cout << TypeInfo<double>::name() << "\n";        // "floating_point"
    std::cout << TypeInfo<std::string>::name() << "\n";  // "unknown"
    std::cout << TypeInfo<int*>::name() << "\n";          // "integral*"
    std::cout << TypeInfo<int[5]>::name() << "\n";       // "integral[5]"
}

Practical Example: Custom Type Traits

#include <iostream>
#include <string>
#include <complex>

// Primary template: not serializable by default
template<typename T>
struct is_serializable : std::false_type {};

// Partial specialization: all arithmetic types are serializable
template<typename T>
struct is_serializable<std::enable_if_t<std::is_arithmetic_v<T>, T>>
    : std::true_type {};

// Full specialization: std::string is serializable
template<>
struct is_serializable<std::string> : std::true_type {};

// Conditional serializer based on type trait
template<typename T>
std::string smart_serialize(const T& value) {
    if constexpr (is_serializable<T>::value) {
        return std::to_string(value);
    } else {
        return "[non-serializable: " + std::string(typeid(T).name()) + "]";
    }
}

void trait_specialization_demo() {
    std::cout << smart_serialize(42) << "\n";           // "42"
    std::cout << smart_serialize(3.14) << "\n";         // "3.14..."
    std::cout << smart_serialize(std::string("hi")) << "\n";  // "hi..."
    // std::cout << smart_serialize(std::complex<double>(1,2)) << "\n";
    // "[non-serializable: ...]"
}

Specialization for Container Types

#include <vector>
#include <map>
#include <iostream>

// Primary template
template<typename T>
struct container_info {
    static void print_info() {
        std::cout << "Not a container\n";
    }
};

// Partial specialization for std::vector
template<typename T, typename Alloc>
struct container_info<std::vector<T, Alloc>> {
    static void print_info() {
        std::cout << "std::vector of " << typeid(T).name() << "\n";
    }
};

// Partial specialization for std::map
template<typename K, typename V, typename Comp, typename Alloc>
struct container_info<std::map<K, V, Comp, Alloc>> {
    static void print_info() {
        std::cout << "std::map<" << typeid(K).name() << ", "
                  << typeid(V).name() << ">\n";
    }
};

void container_specialization_demo() {
    container_info<int>::print_info();                                    // Not a container
    container_info<std::vector<int>>::print_info();                     // std::vector of int
    container_info<std::map<std::string, int>>::print_info();           // std::map<...>
}

Choosing specialization vs overloading: For classes, use specialization. For functions, use overloading (it interacts better with SFINAE). Partial specialization is essential for type traits and is used by the standard library extensively.

Template specialization is a stepping stone to variadic templates and SFINAE, where you will learn more advanced template metaprogramming techniques.
