Thread-Safe Data Structures

When multiple threads access the same data concurrently, you need synchronization to prevent data races and undefined behavior. This page covers three approaches: thread-safe queue (mutex + condition variable), concurrent map (read-write lock), and lock-free basics using atomics.

Thread-Safe Queue (Mutex + Condition Variable)

#include <mutex>
#include <condition_variable>
#include <queue>
#include <optional>

template<typename T>
class ThreadSafeQueue {
    mutable std::mutex mutex_;
    std::condition_variable not_empty_;
    std::queue<T> queue_;
    bool shutdown_ = false;
public:
    ThreadSafeQueue() = default;

    void push(T value) {
        {
            std::lock_guard<std::mutex> lock(mutex_);
            queue_.push(std::move(value));
        }
        not_empty_.notify_one();
    }

    // Blocking pop — waits until an element is available
    T pop() {
        std::unique_lock<std::mutex> lock(mutex_);
        not_empty_.wait(lock, [this] {
            return !queue_.empty() || shutdown_;
        });
        if (shutdown_ && queue_.empty())
            throw std::runtime_error("Queue shut down");
        T value = std::move(queue_.front());
        queue_.pop();
        return value;
    }

    // Non-blocking pop using optional
    std::optional<T> try_pop() {
        std::lock_guard<std::mutex> lock(mutex_);
        if (queue_.empty()) return std::nullopt;
        T value = std::move(queue_.front());
        queue_.pop();
        return value;
    }

    // Timed pop — waits up to timeout_ms
    std::optional<T> pop_for(int timeout_ms) {
        std::unique_lock<std::mutex> lock(mutex_);
        if (!not_empty_.wait_for(lock,
            std::chrono::milliseconds(timeout_ms),
            [this] { return !queue_.empty() || shutdown_; })) {
            return std::nullopt;  // timeout
        }
        if (queue_.empty()) return std::nullopt;
        T value = std::move(queue_.front());
        queue_.pop();
        return value;
    }

    void shutdown() {
        {
            std::lock_guard<std::mutex> lock(mutex_);
            shutdown_ = true;
        }
        not_empty_.notify_all();
    }

    bool empty() const {
        std::lock_guard<std::mutex> lock(mutex_);
        return queue_.empty();
    }

    size_t size() const {
        std::lock_guard<std::mutex> lock(mutex_);
        return queue_.size();
    }
};

Concurrent Map (Read-Write Lock)

A std::shared_mutex (C++14, upgraded to std::shared_timed_mutex) allows multiple concurrent readers but exclusive access for writers.

#include <shared_mutex>
#include <unordered_map>
#include <string>
#include <optional>

template<typename Key, typename Value>
class ConcurrentMap {
    mutable std::shared_mutex rw_mutex_;
    std::unordered_map<Key, Value> map_;
public:
    // Read operations: shared (multiple readers allowed)
    std::optional<Value> get(const Key& key) const {
        std::shared_lock lock(rw_mutex_);  // shared lock
        auto it = map_.find(key);
        if (it != map_.end()) return it->second;
        return std::nullopt;
    }

    bool contains(const Key& key) const {
        std::shared_lock lock(rw_mutex_);
        return map_.count(key) > 0;
    }

    size_t size() const {
        std::shared_lock lock(rw_mutex_);
        return map_.size();
    }

    // Write operations: exclusive (only one writer)
    void put(const Key& key, const Value& value) {
        std::unique_lock lock(rw_mutex_);  // exclusive lock
        map_[key] = value;
    }

    bool remove(const Key& key) {
        std::unique_lock lock(rw_mutex_);
        return map_.erase(key) > 0;
    }

    bool update(const Key& key, const Value& value) {
        std::unique_lock lock(rw_mutex_);
        auto it = map_.find(key);
        if (it == map_.end()) return false;
        it->second = value;
        return true;
    }
};

void concurrent_map_demo() {
    ConcurrentMap<std::string, int> db;

    // Writer thread
    std::thread writer([&]() {
        db.put("Alice", 30);
        db.put("Bob", 25);
        db.put("Charlie", 35);
    });

    // Reader threads (can run concurrently with each other)
    std::thread reader1([&]() {
        if (auto val = db.get("Alice"))
            std::cout << "Alice: " << *val << "\n";
    });
    std::thread reader2([&]() {
        if (auto val = db.get("Bob"))
            std::cout << "Bob: " << *val << "\n";
    });

    writer.join();
    reader1.join();
    reader2.join();
}

Lock-Free Counter (Atomics)

For simple data structures, std::atomic provides lock-free synchronization using hardware atomic instructions. No mutex is needed.

#include <atomic>
#include <iostream>
#include <thread>
#include <vector>

class LockFreeCounter {
    std::atomic<int> count_{0};
public:
    void increment() {
        // fetch_add is an atomic RMW operation
        count_.fetch_add(1, std::memory_order_relaxed);
    }

    void decrement() {
        count_.fetch_sub(1, std::memory_order_relaxed);
    }

    int get() const {
        return count_.load(std::memory_order_relaxed);
    }

    int exchange(int new_val) {
        return count_.exchange(new_val);
    }

    bool compare_and_swap(int expected, int desired) {
        return count_.compare_exchange_strong(expected, desired);
    }
};

void lock_free_counter_demo() {
    LockFreeCounter counter;
    std::vector<std::thread> threads;

    for (int i = 0; i < 10; ++i) {
        threads.emplace_back([&counter]() {
            for (int j = 0; j < 10000; ++j) {
                counter.increment();
            }
        });
    }

    for (auto& t : threads) t.join();
    std::cout << "Lock-free counter: " << counter.get() << "\n";
    // Output: 100000
}

Comparison: Approaches to Thread Safety

ApproachProsConsBest For
Mutex + condvarGeneral-purpose, blocking waitKernel syscall overheadGeneral data structures, waiting
Shared mutexMultiple readers, single writerWriter starvation possibleRead-heavy maps/dictionaries
Atomics (lock-free)No kernel overhead, no deadlockLimited to simple operationsCounters, flags, simple state
Thread-local storageNo synchronization neededNo sharing between threadsPer-thread caches, RNG state

Granularity matters: Fine-grained locking (one lock per bucket/node) allows more concurrency but adds complexity and deadlock risk. Coarse-grained locking (one lock for the whole structure) is simpler but limits throughput. Choose based on your access pattern.

Thread-Local Storage

#include <thread>
#include <iostream>

// Each thread gets its own copy
thread_local int thread_id = -1;

void thread_local_demo() {
    thread_id = 42;
    std::cout << "Thread " << std::this_thread::get_id()
              << " has thread_id = " << thread_id << "\n";
}
