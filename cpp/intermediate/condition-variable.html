Condition Variable

std::condition_variable allows threads to wait for a condition to become true, and other threads to signal that the condition may have changed. It is used with a std::unique_lock<std::mutex> to safely check and modify shared state. Condition variables eliminate the need for busy-waiting (spin loops).

wait / notify Pattern

#include <mutex>
#include <condition_variable>
#include <iostream>

std::mutex mtx;
std::condition_variable cv;
bool ready = false;

void worker() {
    std::unique_lock<std::mutex> lock(mtx);
    cv.wait(lock, []{ return ready; });  // wait until ready == true
    std::cout << "Worker proceeding\n";
}

void signaler() {
    std::this_thread::sleep_for(std::chrono::milliseconds(100));
    {
        std::lock_guard<std::mutex> lock(mtx);
        ready = true;
    }
    cv.notify_one();  // wake one waiting thread
}

void basic_wait_notify() {
    std::thread t1(worker);
    std::thread t2(signaler);
    t1.join();
    t2.join();
    // Output: Worker proceeding
}

Always use wait with a predicate: cv.wait(lock, predicate) is preferred over cv.wait(lock) because it handles spurious wakeups -- the OS may wake a thread without any notification. The predicate ensures the thread only proceeds when the condition is actually true.

Producer-Consumer Pattern

#include <mutex>
#include <condition_variable>
#include <queue>
#include <iostream>
#include <string>

template<typename T>
class ThreadSafeQueue {
    std::queue<T> queue_;
    mutable std::mutex mtx_;
    std::condition_variable not_empty_;
    std::condition_variable not_full_;
    size_t max_size_;
public:
    explicit ThreadSafeQueue(size_t max_size = 100) : max_size_(max_size) {}

    void push(T value) {
        std::unique_lock<std::mutex> lock(mtx_);
        not_full_.wait(lock, [this] { return queue_.size() < max_size_; });
        queue_.push(std::move(value));
        not_empty_.notify_one();
    }

    T pop() {
        std::unique_lock<std::mutex> lock(mtx_);
        not_empty_.wait(lock, [this] { return !queue_.empty(); });
        T value = std::move(queue_.front());
        queue_.pop();
        not_full_.notify_one();
        return value;
    }

    bool try_pop(T& value, int timeout_ms) {
        std::unique_lock<std::mutex> lock(mtx_);
        if (!not_empty_.wait_for(lock,
            std::chrono::milliseconds(timeout_ms),
            [this] { return !queue_.empty(); })) {
            return false;
        }
        value = std::move(queue_.front());
        queue_.pop();
        not_full_.notify_one();
        return true;
    }

    bool empty() const {
        std::lock_guard<std::mutex> lock(mtx_);
        return queue_.empty();
    }
};

void producer_consumer_demo() {
    ThreadSafeQueue<int> q(5);

    // Producer thread
    std::thread producer([&]() {
        for (int i = 1; i <= 10; ++i) {
            q.push(i);
            std::cout << "Produced: " << i << "\n";
        }
    });

    // Consumer thread
    std::thread consumer([&]() {
        for (int i = 0; i < 10; ++i) {
            std::this_thread::sleep_for(std::chrono::milliseconds(50));
            int value = q.pop();
            std::cout << "Consumed: " << value << "\n";
        }
    });

    producer.join();
    consumer.join();
}

notify_one vs notify_all

FunctionBehaviorUse Case
notify_one()Wakes one waiting threadSingle producer, single consumer
notify_all()Wakes all waiting threadsMultiple consumers, one-shot events
#include <mutex>
#include <condition_variable>
#include <iostream>
#include <vector>

std::mutex barrier_mtx;
std::condition_variable barrier_cv;

void barrier_demo() {
    const int num_threads = 4;
    std::vector<std::thread> threads;
    int count = 0;

    for (int i = 0; i < num_threads; ++i) {
        threads.emplace_back([i, &count]() {
            std::cout << "Thread " << i << " doing work\n";

            std::unique_lock<std::mutex> lock(barrier_mtx);
            ++count;
            if (count == num_threads) {
                // Last thread to arrive: wake everyone
                barrier_cv.notify_all();
            } else {
                // Wait for all threads to arrive
                barrier_cv.wait(lock, [&count] {
                    return count == num_threads;
                });
            }
            std::cout << "Thread " << i << " passed barrier\n";
        });
    }

    for (auto& t : threads) t.join();
}

wait_for and wait_until

#include <mutex>
#include <condition_variable>
#include <iostream>

void timeout_demo() {
    std::mutex mtx;
    std::condition_variable cv;
    bool data_ready = false;

    std::thread consumer([&]() {
        std::unique_lock<std::mutex> lock(mtx);
        // Wait for up to 500ms
        bool result = cv.wait_for(lock, std::chrono::milliseconds(500),
            [&data_ready] { return data_ready; });

        if (result) {
            std::cout << "Data received\n";
        } else {
            std::cout << "Timeout: no data\n";
        }
    });

    // No producer -- consumer times out
    consumer.join();
    // Output: Timeout: no data
}

Common mistakes: Forgetting to hold the mutex when modifying the shared state, forgetting to notify after state change, using wait() without a predicate (spurious wakeup vulnerability), and unlocking the mutex before calling notify (this is actually fine -- but the state must be modified while holding the lock).

Condition variables require mutexes and are used by threads. See async/futures for a higher-level alternative. Practice with thread pool and producer-consumer pattern.
