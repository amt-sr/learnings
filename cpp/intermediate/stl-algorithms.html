STL Algorithms

The <algorithm> and <numeric> headers provide over 100 generic algorithms that operate on iterator ranges. They separate the "what" (what operation to perform) from the "how" (what container to use), enabling code reuse and correctness. Modern C++ code should use algorithms over hand-written loops wherever possible.

Sorting and Searching

#include <algorithm>
#include <numeric>
#include <vector>
#include <iostream>

void sorting_searching() {
    std::vector<int> v = {5, 2, 8, 1, 9, 3};

    // sort: O(n log n) — requires random-access iterators
    std::sort(v.begin(), v.end());  // {1, 2, 3, 5, 8, 9}

    // Sort with custom comparator (descending)
    std::sort(v.begin(), v.end(), std::greater<>());

    // Partial sort: only sort the first N elements
    std::partial_sort(v.begin(), v.begin() + 3, v.end());
    // v = {9, 8, 5, ...} (first 3 are the largest, rest unsorted)

    // Binary search (requires sorted range)
    std::sort(v.begin(), v.end());
    bool found = std::binary_search(v.begin(), v.end(), 5);  // true

    // lower_bound / upper_bound
    auto lo = std::lower_bound(v.begin(), v.end(), 5);  // first >= 5
    auto hi = std::upper_bound(v.begin(), v.end(), 5);  // first > 5

    // find: linear search — O(n)
    auto it = std::find(v.begin(), v.end(), 8);
    if (it != v.end()) {
        std::cout << "Found 8 at index " << (it - v.begin()) << "\n";
    }

    // find_if: with predicate
    auto it2 = std::find_if(v.begin(), v.end(),
        [](int x) { return x > 5; });
}

Counting and Accumulating

#include <algorithm>
#include <numeric>
#include <vector>
#include <iostream>

void counting() {
    std::vector<int> v = {1, 2, 3, 2, 4, 2, 5};

    // count: occurrences of a value
    int twos = std::count(v.begin(), v.end(), 2);
    std::cout << "Count of 2: " << twos << "\n";  // 3

    // count_if: with predicate
    int evens = std::count_if(v.begin(), v.end(),
        [](int x) { return x % 2 == 0; });
    std::cout << "Even count: " << evens << "\n";  // 4

    // accumulate: sum with initial value (from <numeric>)
    int sum = std::accumulate(v.begin(), v.end(), 0);
    std::cout << "Sum: " << sum << "\n";  // 19

    // accumulate with custom operation
    int product = std::accumulate(v.begin(), v.end(), 1,
        [](int acc, int x) { return acc * x; });
    std::cout << "Product: " << product << "\n";  // 240
}

Transform and For Each

#include <algorithm>
#include <vector>
#include <string>
#include <iostream>

void transform_for_each() {
    std::vector<int> v = {1, 2, 3, 4, 5};

    // transform: apply function to each element, store result
    std::vector<int> result(v.size());
    std::transform(v.begin(), v.end(), result.begin(),
        [](int x) { return x * x; });
    // result = {1, 4, 9, 16, 25}

    // transform in-place
    std::transform(v.begin(), v.end(), v.begin(),
        [](int x) { return x * 10; });
    // v = {10, 20, 30, 40, 50}

    // for_each: perform action on each element
    std::for_each(v.begin(), v.end(),
        [](int x) { std::cout << x << " "; });
    // Output: 10 20 30 40 50

    // Transform strings
    std::vector<std::string> words = {"hello", "world"};
    std::vector<std::string> upper;
    std::transform(words.begin(), words.end(), std::back_inserter(upper),
        [](std::string s) {
            std::transform(s.begin(), s.end(), s.begin(), ::toupper);
            return s;
        });
}

Removing and Unique

The erase-remove idiom is the standard way to remove elements from a container. The std::remove family moves non-removed elements to the front and returns the new end.

#include <algorithm>
#include <vector>
#include <iostream>

void removing_elements() {
    std::vector<int> v = {1, 2, 3, 4, 5, 3, 7, 3, 9};

    // Erase-remove idiom: remove all 3s
    v.erase(
        std::remove(v.begin(), v.end(), 3),
        v.end());
    // v = {1, 2, 4, 5, 7, 9}

    // Remove with predicate
    v.erase(
        std::remove_if(v.begin(), v.end(),
            [](int x) { return x < 4; }),
        v.end());
    // v = {4, 5, 7, 9}

    // unique: remove consecutive duplicates
    std::vector<int> dup = {1, 1, 2, 2, 2, 3, 4, 4};
    dup.erase(std::unique(dup.begin(), dup.end()), dup.end());
    // dup = {1, 2, 3, 4}

    // Note: unique only removes CONSECUTIVE duplicates
    // For full deduplication, sort first:
    std::vector<int> data = {3, 1, 4, 1, 5, 9, 2, 6, 5};
    std::sort(data.begin(), data.end());
    data.erase(std::unique(data.begin(), data.end()), data.end());
    // data = {1, 2, 3, 4, 5, 6, 9}
}

Lambdas with Algorithms

#include <algorithm>
#include <vector>
#include <iostream>

void lambda_algorithms() {
    std::vector<int> data = {5, 12, 7, 3, 18, 9, 1, 15};

    // any_of, all_of, none_of
    bool has_large = std::any_of(data.begin(), data.end(),
        [](int x) { return x > 20; });  // false
    bool all_positive = std::all_of(data.begin(), data.end(),
        [](int x) { return x > 0; });  // true
    bool none_negative = std::none_of(data.begin(), data.end(),
        [](int x) { return x < 0; });  // true

    // min_element / max_element
    auto [min_it, max_it] = std::minmax_element(data.begin(), data.end());
    std::cout << "Min: " << *min_it << ", Max: " << *max_it << "\n";

    // mismatch: find first differing elements
    std::vector<int> a = {1, 2, 3, 5};
    std::vector<int> b = {1, 2, 4, 5};
    auto [ai, bi] = std::mismatch(a.begin(), a.end(), b.begin());
    std::cout << "First diff: " << *ai << " vs " << *bi << "\n";
    // Output: First diff: 3 vs 4

    // Clamp (C++17)
    for (int x : {5, 15, 25}) {
        std::cout << std::clamp(x, 0, 20) << " ";
    }
    // Output: 5 15 20
}

Prefer algorithms over raw loops: Algorithms express intent (sort, find, count), are well-tested, and may use SIMD optimizations. Use range-based for loops for simple iteration; use algorithms for anything involving searching, transforming, or computing.

STL algorithms work with lambda expressions for predicates and transformations. They require iterators to specify ranges. In C++20, ranges provide a compositional pipeline interface on top of these algorithms.
