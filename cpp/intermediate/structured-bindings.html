Structured Bindings

Structured bindings (C++17) let you decompose a compound value into its individual components using a single declaration. They provide a cleaner alternative to std::tie and manual member access, dramatically improving readability when working with pairs and tuples, maps, and custom types.

Syntax

auto [name1, name2, ...] = expression;
auto& [name1, name2, ...] = expression;
const auto& [name1, name2, ...] = expression;

With std::pair

#include <utility>
#include <iostream>
#include <string>

void pair_demo() {
    // Return multiple values using a pair
    std::pair<std::string, int> get_user() {
        return {"Alice", 30};
    }

    // Without structured bindings
    auto result = get_user();
    std::cout << result.first << " is " << result.second << "\n";

    // With structured bindings — much cleaner
    auto [name, age] = get_user();
    std::cout << name << " is " << age << "\n";

    // Works with references
    std::pair<int, int> point = {10, 20};
    auto& [x, y] = point;
    x = 15;  // modifies point.first
    y = 25;  // modifies point.second
    std::cout << "point: (" << point.first << ", " << point.second << ")\n";
}

With std::tuple

#include <tuple>
#include <iostream>
#include <string>

std::tuple<std::string, int, double> parse_record(const std::string& line) {
    // Imagine parsing "Alice,30,3.14"
    return {"Alice", 30, 3.14};
}

void tuple_demo() {
    // Before C++17: std::tie (verbose, requires default-constructible variables)
    std::string name;
    int age;
    double score;
    std::tie(name, age, score) = parse_record("Alice,30,3.14");

    // C++17 structured bindings — clean and direct
    auto [n, a, s] = parse_record("Bob,25,2.71");
    std::cout << n << ", age " << a << ", score " << s << "\n";
}

With std::map — Iterating Key-Value Pairs

#include <map>
#include <unordered_map>
#include <iostream>
#include <string>

void map_demo() {
    // This is the most common use case for structured bindings
    std::map<std::string, int> scores = {
        {"Alice", 95},
        {"Bob", 87},
        {"Carol", 92},
        {"Dave", 78}
    };

    // Before C++17
    for (const auto& pair : scores) {
        std::cout << pair.first << ": " << pair.second << "\n";
    }

    // C++17 structured bindings
    for (const auto& [name, score] : scores) {
        std::cout << name << ": " << score << "\n";
    }

    // With unordered_map
    std::unordered_map<std::string, std::string> capitals = {
        {"France", "Paris"}, {"Germany", "Berlin"}, {"Italy", "Rome"}
    };
    for (const auto& [country, capital] : capitals) {
        std::cout << "Capital of " << country << " is " << capital << "\n";
    }
}

With Custom Types

Structured bindings work with any aggregate type (struct/class with all public members) or any type with std::tuple_size, std::tuple_element, and get<I>() specialized for it.

#include <iostream>
#include <string>

// Method 1: Aggregate — public members, no constructors needed
struct Point3D {
    double x;
    double y;
    double z;
};

void aggregate_demo() {
    Point3D p{1.0, 2.0, 3.0};

    auto [x, y, z] = p;  // copies each member
    std::cout << "(" << x << ", " << y << ", " << z << ")\n";

    auto&& [rx, ry, rz] = Point3D{4.0, 5.0, 6.0};  // rvalue reference
    std::cout << "(" << rx << ", " << ry << ", " << rz << ")\n";
}

// Method 2: Non-aggregate with tuple-like interface
struct Employee {
    std::string name;
    int id;
    double salary;

    // Provide tuple-like interface
    template<size_t I>
    auto& get() {
        if constexpr (I == 0) return name;
        else if constexpr (I == 1) return id;
        else if constexpr (I == 2) return salary;
    }
};

// Specialize std::tuple_size and std::tuple_element for Employee
namespace std {
    template<> struct tuple_size<Employee> : integral_constant<size_t, 3> {};
    template<> struct tuple_element<0, Employee> { using type = std::string; };
    template<> struct tuple_element<1, Employee> { using type = int; };
    template<> struct tuple_element<2, Employee> { using type = double; };
}

Important: Structured bindings create aliases, not new variables. All the names refer to parts of the same underlying object. You cannot move individual components out of a structured binding to a const reference — the binding itself is a copy in that case.

Practical Examples

Returning Multiple Values with structured_error

#include <string>
#include <optional>
#include <iostream>

struct Result {
    bool success;
    std::string data;
    std::string error;
};

Result fetch_data(int id) {
    if (id > 0)
        return {true, "Data for " + std::to_string(id), ""};
    return {false, "", "Invalid ID"};
}

void result_demo() {
    // Clean destructuring
    auto [ok, data, err] = fetch_data(42);
    if (ok)
        std::cout << data << "\n";
    else
        std::cout << "Error: " << err << "\n";
}

Insert into map with structured bindings

#include <map>
#include <iostream>
#include <string>

void map_insert_demo() {
    std::map<std::string, int> word_count;

    // insert() returns a pair<iterator, bool>
    auto [it, inserted] = word_count.insert({"hello", 1});
    if (!inserted) {
        it->second++;  // word already existed
    }

    auto [it2, inserted2] = word_count.insert({"world", 1});
    std::cout << "inserted 'hello': " << inserted << "\n";  // true
    std::cout << "inserted 'world': " << inserted2 << "\n";  // true
}

Ignore Some Components

#include <tuple>
#include <string>

// You can use [[maybe_unused]] or just name unused variables with _
auto process() {
    return std::make_tuple(std::string("result"), 42, 3.14);
}

void ignore_demo() {
    // Only care about the string, ignore the rest
    auto [result, /*count*/, /*ratio*/] = process();

    // Common convention: use underscore-like names
    auto [res, count_unused, ratio_unused] = process();
}
