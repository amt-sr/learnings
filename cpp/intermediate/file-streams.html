File Streams (fstream)

C++ file I/O is handled by three stream classes defined in <fstream>: std::ifstream for reading, std::ofstream for writing, and std::fstream for both. These classes inherit from std::ios and integrate seamlessly with the std::string and STL ecosystem.

Stream Types

ClassHeaderPurposeDefault Mode
std::ifstream<fstream>Read from filein
std::ofstream<fstream>Write to fileout
std::fstream<fstream>Read and writein | out

Opening and Closing Files

#include <fstream>
#include <iostream>

void open_close_demo() {
    // Method 1: Constructor opens the file
    std::ifstream reader("data.txt");

    // Method 2: Open later with open()
    std::ofstream writer;
    writer.open("output.txt");

    // Always check if the file opened successfully
    if (!reader.is_open()) {
        std::cerr << "Error: cannot open data.txt\n";
        return;
    }

    // Close explicitly (optional — destructor closes automatically)
    reader.close();
    writer.close();
}

Common pitfall: Forgetting to check is_open() leads to silent failures. Always validate the file opened before reading or writing. The destructor will close the file, so explicit close() is optional but good for clarity.

Reading from Files

#include <fstream>
#include <string>
#include <vector>
#include <iostream>

// Method 1: Line by line (most common)
void read_line_by_line(const std::string& path) {
    std::ifstream file(path);
    if (!file) return;

    std::string line;
    while (std::getline(file, line)) {
        std::cout << line << "\n";
    }
}

// Method 2: Word by word (whitespace-delimited)
void read_words(const std::string& path) {
    std::ifstream file(path);
    std::string word;
    while (file >> word) {  // operator>> skips whitespace
        std::cout << "[" << word << "] ";
    }
    std::cout << "\n";
}

// Method 3: Read entire file into a string
std::string read_entire_file(const std::string& path) {
    std::ifstream file(path, std::ios::binary | std::ios::ate);
    if (!file) return "";

    auto size = file.tellg();
    file.seekg(0);

    std::string content(size, '\0');
    file.read(content.data(), size);
    return content;
}

// Method 4: Read lines into a vector
std::vector<std::string> read_lines(const std::string& path) {
    std::vector<std::string> lines;
    std::ifstream file(path);
    std::string line;
    while (std::getline(file, line)) {
        lines.push_back(line);
    }
    return lines;
}

Writing to Files

#include <fstream>

void writing_demo() {
    // Overwrite mode (default — truncates existing file)
    std::ofstream out1("log.txt");
    out1 << "First entry\n";

    // Append mode
    std::ofstream out2("log.txt", std::ios::app);
    out2 << "Second entry (appended)\n";

    // Write multiple lines with formatting
    std::ofstream data("records.csv");
    data << "name,age,score\n";
    data << "Alice,30,95.5\n";
    data << "Bob,25,87.2\n";

    // flush() forces buffer to disk (automatic on close/destruct)
    data.flush();
}

Best practice: Use std::ios::app for log files to avoid overwriting previous entries. Use std::ios::ate to open and seek to the end (for reading) without truncating.

FlagMeaning

Binary Mode

Binary mode prevents \n from being translated to platform-specific line endings. Use it for non-text data like structs, images, or protocol buffers.

#include <fstream>
#include <iostream>

struct Record {
    int id;
    double value;
    char name[32];
};

void binary_io_demo() {
    // Write binary
    Record r{42, 3.14, "sensor_data"};
    std::ofstream out("data.bin", std::ios::binary);
    out.write(reinterpret_cast<const char*>(&r), sizeof(r));

    // Read binary
    Record loaded{};
    std::ifstream in("data.bin", std::ios::binary);
    in.read(reinterpret_cast<char*>(&loaded), sizeof(loaded));

    std::cout << "id=" << loaded.id
              << " value=" << loaded.value
              << " name=" << loaded.name << "\n";
}

Portability warning: Binary I/O with reinterpret_cast depends on struct layout, endianness, and alignment. For portable serialization, use libraries like protobuf, flatbuffers, or write field-by-field.

File Parsing Example: Reading a Config

#include <fstream>
#include <string>
#include <unordered_map>
#include <iostream>

// Parses files like: key = value  (ignores # comments and blank lines)
std::unordered_map<std::string, std::string> parse_config(const std::string& path) {
    std::unordered_map<std::string, std::string> config;
    std::ifstream file(path);
    std::string line;

    while (std::getline(file, line)) {
        // Skip empty lines and comments
        if (line.empty() || line[0] == '#') continue;

        // Find the delimiter
        auto pos = line.find('=');
        if (pos == std::string::npos) continue;

        std::string key = line.substr(0, pos);
        std::string val = line.substr(pos + 1);

        // Trim whitespace (simplified)
        auto trim = [](std::string& s) {
            size_t start = s.find_first_not_of(" \t");
            size_t end   = s.find_last_not_of(" \t");
            if (start != std::string::npos)
                s = s.substr(start, end - start + 1);
        };
        trim(key);
        trim(val);

        config[key] = val;
    }
    return config;
}

void config_demo() {
    auto cfg = parse_config("settings.cfg");
    std::cout << "host = " << cfg["host"] << "\n";
    std::cout << "port = " << cfg["port"] << "\n";
}

File Position and State

#include <fstream>

void position_demo() {
    std::fstream file("data.txt", std::ios::in | std::ios::out);

    // Get current position
    auto pos = file.tellg();

    // Seek to beginning
    file.seekg(0, std::ios::beg);

    // Seek to end
    file.seekg(0, std::ios::end);

    // Seek relative to current position
    file.seekg(-10, std::ios::cur);

    // Check stream state
    if (file.eof())  std::cout << "End of file\n";
    if (file.fail()) std::cout << "Read/write error\n";
    if (file.bad())  std::cout << "Fatal I/O error\n";
}

File streams work naturally with std::string operations and are often combined with regular expressions for structured data parsing. For an overview of STL containers that store loaded data, see STL Overview.
