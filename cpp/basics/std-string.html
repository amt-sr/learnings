1.5.4 std::string

std::string is the standard C++ class for text manipulation. It manages its own memory, handles resizing automatically, and provides a rich set of member functions for searching, modifying, and comparing strings. Unlike C-style character arrays, std::string is safe, convenient, and integrates seamlessly with the STL.

Construction and Initialization

#include <iostream>
#include <string>

int main() {
    // Default: empty string
    std::string s1;

    // From C-style string literal
    std::string s2 = "hello";

    // From another string (copy)
    std::string s3 = s2;

    // Repeated characters
    std::string s4(5, 'x');  // "xxxxx"

    // From a substring
    std::string s5 = "hello world";
    std::string s6(s5, 6);           // "world" (from position 6)
    std::string s7(s5, 0, 5);        // "hello" (5 chars from position 0)

    // Move constructor (C++11)
    std::string s8 = std::move(s7);  // s7 is now empty

    std::cout << "s2: " << s2 << std::endl;   // hello
    std::cout << "s4: " << s4 << std::endl;   // xxxxx
    std::cout << "s6: " << s6 << std::endl;   // world

    return 0;
}

Concatenation

#include <iostream>
#include <string>

int main() {
    std::string first = "Hello";
    std::string second = "World";

    // Using + operator
    std::string combined = first + ", " + second + "!";
    std::cout << combined << std::endl;  // Hello, World!

    // Using +=
    first += " C++";
    std::cout << first << std::endl;  // Hello C++

    // Mixing string and C-style literals works with +
    std::string msg = std::string("Count: ") + std::to_string(42);
    std::cout << msg << std::endl;  // Count: 42

    // Building strings in a loop
    std::string result;
    for (int i = 0; i < 5; ++i) {
        result += std::to_string(i) + " ";
    }
    std::cout << result << std::endl;  // 0 1 2 3 4

    return 0;
}

Note: You cannot concatenate two C-string literals with + (e.g., "hello" + "world" is pointer arithmetic, not concatenation). At least one operand must be a std::string. std::string("hello") + " world" works fine.

Substring and Find

#include <iostream>
#include <string>

int main() {
    std::string text = "Hello, World!";

    // substr(pos, len): extract substring
    std::string greeting = text.substr(0, 5);    // "Hello"
    std::string world = text.substr(7);           // "World!" (from pos 7 to end)
    std::cout << greeting << std::endl;           // Hello
    std::cout << world << std::endl;               // World!

    // find(): search for substring (returns position, or std::string::npos)
    std::string sentence = "the cat sat on the mat";

    size_t pos = sentence.find("cat");
    if (pos != std::string::npos) {
        std::cout << "Found 'cat' at: " << pos << std::endl;  // 4
    }

    // rfind(): search from the end
    size_t rpos = sentence.rfind("the");
    std::cout << "Last 'the' at: " << rpos << std::endl;  // 16

    // find_first_of / find_last_of: find any character from a set
    size_t digit = sentence.find_first_of("0123456789");
    if (digit == std::string::npos) {
        std::cout << "No digits found" << std::endl;
    }

    return 0;
}

String to Number and Back

#include <iostream>
#include <string>
#include <cmath>

int main() {
    // String to number
    int i = std::stoi("42");
    double d = std::stod("3.14159");
    long l = std::stol("1234567890");

    std::cout << "stoi: " << i << std::endl;       // 42
    std::cout << "stod: " << d << std::endl;       // 3.14159
    std::cout << "stol: " << l << std::endl;       // 1234567890

    // With base specification
    int hex = std::stoi("0xFF", nullptr, 16);  // 255
    int bin = std::stoi("1010", nullptr, 2);  // 10
    std::cout << "hex: " << hex << ", bin: " << bin << std::endl;

    // Number to string
    std::string s1 = std::to_string(42);         // "42"
    std::string s2 = std::to_string(3.14159);   // "3.141590"
    std::cout << "to_string(int):   " << s1 << std::endl;
    std::cout << "to_string(double): " << s2 << std::endl;

    // Handling parse errors
    try {
        int bad = std::stoi("not a number");
    } catch (const std::invalid_argument& e) {
        std::cout << "Parse error: " << e.what() << std::endl;
    }

    return 0;
}

Never use: strcpy, strcat, gets, sprintf — these C functions are unsafe and have no bounds checking. Buffer overflows in these functions are a major source of security vulnerabilities. Always prefer std::string and its member functions.

String Modification

#include <iostream>
#include <string>
#include <algorithm>

int main() {
    std::string s = "Hello, World!";

    // Replace
    s.replace(7, 5, "C++");  // "Hello, C++!"
    std::cout << s << std::endl;

    // Erase
    s.erase(5, 1);  // Remove comma: "Hello C++!"
    std::cout << s << std::endl;

    // Insert
    s.insert(5, ",");  // "Hello, C++!"
    std::cout << s << std::endl;

    // Transform to upper/lower case
    std::string upper = s;
    std::transform(upper.begin(), upper.end(), upper.begin(), ::toupper);
    std::cout << upper << std::endl;  // HELLO, C++!

    return 0;
}

Reading Strings from Input

#include <iostream>
#include <string>

int main() {
    // cin >> : reads one word (stops at whitespace)
    std::string word;
    std::cout << "Enter a word: ";
    std::cin >> word;
    std::cout << "You entered: " << word << std::endl;

    // getline: reads entire line (including spaces)
    std::cin.ignore();  // Clear leftover newline
    std::string line;
    std::cout << "Enter a line: ";
    std::getline(std::cin, line);
    std::cout << "You entered: " << line << std::endl;

    return 0;
}

Tip: After using cin >>, always call cin.ignore() before getline() to consume the leftover newline character. Without this, getline() will read an empty string immediately.
