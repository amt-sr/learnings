1.4.2 Class Templates

Class templates let you define a class that works with any data type. The entire STL (Standard Template Library) is built on class templates — std::vector<int>, std::map<string, int>, and std::pair<double, string> are all class template instantiations.

Template Class Syntax

A class template is declared with template<typename T> before the class definition. Member functions defined outside the class need their own template<typename T> prefix.

#include <iostream>

// A generic Pair class that holds two values of any type
template <typename T>
class Pair {
private:
    T first_;
    T second_;

public:
    Pair(T first, T second) : first_{first}, second_{second} {}

    T get_first() const { return first_; }
    T get_second() const { return second_; }

    T sum() const { return first_ + second_; }

    void print() const {
        std::cout << "(" << first_ << ", " << second_ << ")" << std::endl;
    }
};

int main() {
    Pair<int> nums(10, 20);
    nums.print();           // (10, 20)
    std::cout << "Sum: " << nums.sum() << std::endl;  // 30

    Pair<double> decimals(3.14, 2.71);
    decimals.print();       // (3.14, 2.71)
    std::cout << "Sum: " << decimals.sum() << std::endl;  // 5.85

    return 0;
}

Generic Stack Container

Here is a practical example: a stack data structure that works with any element type.

#include <iostream>
#include <stdexcept>

template <typename T>
class Stack {
private:
    static const int MAX_SIZE = 100;
    T data_[MAX_SIZE];
    int top_;

public:
    Stack() : top_{-1} {}

    void push(const T& value) {
        if (top_ >= MAX_SIZE - 1) {
            throw std::overflow_error("Stack overflow");
        }
        data_[++top_] = value;
    }

    T pop() {
        if (top_ < 0) {
            throw std::underflow_error("Stack underflow");
        }
        return data_[top_--];
    }

    T& top() {
        if (top_ < 0) {
            throw std::underflow_error("Stack is empty");
        }
        return data_[top_];
    }

    bool empty() const { return top_ < 0; }
    int size() const { return top_ + 1; }
};

int main() {
    Stack<int> int_stack;
    int_stack.push(10);
    int_stack.push(20);
    int_stack.push(30);

    std::cout << "Size: " << int_stack.size() << std::endl;  // 3
    std::cout << "Top: " << int_stack.top() << std::endl;    // 30
    std::cout << "Pop: " << int_stack.pop() << std::endl;      // 30
    std::cout << "Pop: " << int_stack.pop() << std::endl;      // 20

    // Same stack with strings
    Stack<std::string> str_stack;
    str_stack.push("hello");
    str_stack.push("world");
    std::cout << str_stack.pop() << std::endl;  // world

    return 0;
}

Note: This example uses a fixed-size array for simplicity. In production, std::vector would be used internally for dynamic resizing. See std::vector for the standard library's dynamic array container.

Multiple Type Parameters

Class templates can have multiple type parameters, just like function templates.

#include <iostream>
#include <string>

// A Pair with two different types
template <typename K, typename V>
class KeyValuePair {
private:
    K key_;
    V value_;

public:
    KeyValuePair(K key, V value) : key_{key}, value_{value} {}

    K get_key() const { return key_; }
    V get_value() const { return value_; }

    void print() const {
        std::cout << key_ << " => " << value_ << std::endl;
    }
};

int main() {
    KeyValuePair<std::string, int> age{"Alice", 30};
    age.print();  // Alice => 30

    KeyValuePair<int, std::string> code{200, "OK"};
    code.print();  // 200 => OK

    KeyValuePair<std::string, double> temp{"CPU", 67.5};
    temp.print();  // CPU => 67.5

    return 0;
}

Non-Type Template Parameters

Templates can accept values (not just types) as parameters. These are called non-type template parameters and must be compile-time constants.

#include <iostream>
#include <array>

// Array wrapper with a compile-time size
template <typename T, std::size_t N>
class FixedArray {
private:
    T data_[N];

public:
    T& operator[](std::size_t index) { return data_[index]; }
    const T& operator[](std::size_t index) const { return data_[index]; }

    std::size_t size() const { return N; }

    void fill(const T& value) {
        for (std::size_t i = 0; i < N; ++i) {
            data_[i] = value;
        }
    }

    void print() const {
        std::cout << "[";
        for (std::size_t i = 0; i < N; ++i) {
            std::cout << data_[i];
            if (i + 1 < N) std::cout << ", ";
        }
        std::cout << "]" << std::endl;
    }
};

int main() {
    FixedArray<int, 5> arr;
    arr.fill(42);
    arr[2] = 100;
    arr.print();  // [42, 42, 100, 42, 42]
    std::cout << "Size: " << arr.size() << std::endl;  // 5

    FixedArray<double, 3> darr;
    darr[0] = 1.1;
    darr[1] = 2.2;
    darr[2] = 3.3;
    darr.print();  // [1.1, 2.2, 3.3]

    return 0;
}

Tip: std::array<T, N> in the standard library is essentially this pattern — a fixed-size container parameterized by element type and count. Non-type parameters must be integral types, enumerations, or pointers/references. Floating-point values and class types are not allowed.

Default Type Parameters

You can provide default types for template parameters, making them optional at the point of instantiation.

#include <iostream>

// Default type: if not specified, uses double
template <typename T = double>
class NumericValue {
private:
    T value_;

public:
    NumericValue(T val) : value_{val} {}

    T get() const { return value_; }
    void set(T val) { value_ = val; }

    void print() const {
        std::cout << "Value: " << value_ << std::endl;
    }
};

int main() {
    // Explicit type
    NumericValue<int> n1{42};
    n1.print();  // Value: 42

    // Uses default type (double)
    NumericValue<> n2{3.14};
    n2.print();  // Value: 3.14

    return 0;
}

Member Function Definitions Outside the Class

When defining member functions outside the class body, each function needs the template declaration prefix and the class name must include the type parameter.

#include <iostream>

template <typename T>
class Box {
private:
    T contents_;

public:
    Box(T contents);
    T get() const;
    void set(T contents);
    void print() const;
};

// Outside definition: note the template prefix and Box<T> syntax
template <typename T>
Box<T>::Box(T contents) : contents_{contents} {}

template <typename T>
T Box<T>::get() const { return contents_; }

template <typename T>
void Box<T>::set(T contents) { contents_ = contents; }

template <typename T>
void Box<T>::print() const {
    std::cout << "Box contains: " << contents_ << std::endl;
}

int main() {
    Box<std::string> b{"treasure"};
    b.print();  // Box contains: treasure
    b.set("gold");
    b.print();  // Box contains: gold

    return 0;
}

Note: As with function templates, class template definitions must be in the header file (or included from the header). The compiler needs the full definition to instantiate the template. Common approaches: define everything in the .h file, or use a .hpp / .inl file included at the bottom of the header.
